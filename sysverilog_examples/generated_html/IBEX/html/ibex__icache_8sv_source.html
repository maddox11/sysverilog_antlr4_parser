200 //////////////////////////
201 // Instruction prefetch //

203
204 assign lookup_addr_aligned = {lookup_addr_ic0[ADDR_W-1:IC_LINE_W], {IC_LINE_W{1'b0}}};
205
206 // The prefetch address increments by one cache line for each granted request.
207 // This address is also updated if there is a branch that is not granted, since the target
208 // address (addr_i) is only valid for one cycle while branch_i is high.
209
210 // The captured branch target address is not forced to be aligned since the offset in the cache
211 // line must also be recorded for later use by the fill buffers.
212 assign prefetch_addr_d =
213 lookup_grant_ic0 ? (lookup_addr_aligned +
214 {{ADDR_W-IC_LINE_W-1{1'b0}}, 1'b1, {IC_LINE_W{1'b0}}}) :
215 addr_i;
216
217 assign prefetch_addr_en = branch_i | lookup_grant_ic0;
218
219 if (ResetAll) begin : g_prefetch_addr_ra
220 always_ff @(posedge clk_i or negedge rst_ni) begin
221 if (!rst_ni) begin
222 prefetch_addr_q <= '0;
223 end else if (prefetch_addr_en) begin
224 prefetch_addr_q <= prefetch_addr_d;
225 end
226 end
227 end else begin : g_prefetch_addr_nr
228 always_ff @(posedge clk_i) begin
229 if (prefetch_addr_en) begin
230 prefetch_addr_q <= prefetch_addr_d;
231 end
232 end
233 end

235 ////////////////////////
236 // Pipeline stage IC0 //
237 ////////////////////////
238
239 // Cache lookup
240 assign lookup_throttle = (fb_fill_level > FB_THRESHOLD[$clog2(NUM_FB)-1:0]);
241
242 assign lookup_req_ic0 = req_i & ~&fill_busy_q & (branch_i | ~lookup_throttle) &
243 ~ecc_write_req;
244 assign lookup_addr_ic0 = branch_i ? addr_i : prefetch_addr_q;
245 assign lookup_index_ic0 = lookup_addr_ic0[IC_INDEX_HI:IC_LINE_W];
246
247 // Cache write
248 assign fill_req_ic0 = (|fill_ram_req);
249 assign fill_index_ic0 = fill_ram_req_addr[IC_INDEX_HI:IC_LINE_W];
250 assign fill_tag_ic0 = {(~inval_write_req & ~ecc_write_req),
251 fill_ram_req_addr[ADDR_W-1:IC_INDEX_HI+1]};
252 assign fill_wdata_ic0 = fill_ram_req_data;
253
254 // Arbitrated signals - lookups have highest priority
255 assign lookup_grant_ic0 = lookup_req_ic0;
256 assign fill_grant_ic0 = fill_req_ic0 & ~lookup_req_ic0 & ~inval_write_req &
257 ~ecc_write_req;
258 // Qualified lookup grant to mask ram signals in IC1 if access was not made
259 assign lookup_actual_ic0 = lookup_grant_ic0 & icache_enable_i & ~inval_block_cache;
260
261 // Tagram
262 assign tag_req_ic0 = lookup_req_ic0 | fill_req_ic0 | inval_write_req | ecc_write_req;
263 assign tag_index_ic0 = inval_write_req ? inval_index_q :
264 ecc_write_req ? ecc_write_index :
265 fill_grant_ic0 ? fill_index_ic0 :
266 lookup_index_ic0;
267 assign tag_banks_ic0 = ecc_write_req ? ecc_write_ways :
268 fill_grant_ic0 ? fill_ram_req_way :
269 {IC_NUM_WAYS{1'b1}};
270 assign tag_write_ic0 = fill_grant_ic0 | inval_write_req | ecc_write_req;
271
272 // Dataram
273 assign data_req_ic0 = lookup_req_ic0 | fill_req_ic0;
274 assign data_index_ic0 = tag_index_ic0;
275 assign data_banks_ic0 = tag_banks_ic0;
276 assign data_write_ic0 = tag_write_ic0;
277
278 // Append ECC checkbits to write data if required
279 if (ICacheECC) begin : gen_ecc_wdata
280 // SEC_CM: ICACHE.MEM.INTEGRITY
281 // Tagram ECC
282 // Reuse the same ecc encoding module for larger cache sizes by padding with zeros
283 logic [21:0] tag_ecc_input_padded;
284 logic [27:0] tag_ecc_output_padded;
285 logic [22-IC_TAG_SIZE:0] unused_tag_ecc_output;
286
287 assign tag_ecc_input_padded = {{22-IC_TAG_SIZE{1'b0}},fill_tag_ic0};
288 assign unused_tag_ecc_output = tag_ecc_output_padded[21:IC_TAG_SIZE-1];
289
290 prim_secded_inv_28_22_enc tag_ecc_enc (
291 .data_i (tag_ecc_input_padded),
292 .data_o (tag_ecc_output_padded)
293 );
294
295 assign tag_wdata_ic0 = {tag_ecc_output_padded[27:22],tag_ecc_output_padded[IC_TAG_SIZE-1:0]};
296
297 // Dataram ECC
298 for (genvar bank = 0; bank < IC_LINE_BEATS; bank++) begin : gen_ecc_banks
299 prim_secded_inv_39_32_enc data_ecc_enc (
300 .data_i (fill_wdata_ic0[bank*BUS_SIZE+:BUS_SIZE]),
301 .data_o (data_wdata_ic0[bank*BusSizeECC+:BusSizeECC])
302 );
303 end
304
305 end else begin : gen_noecc_wdata
306 assign tag_wdata_ic0 = fill_tag_ic0;
307 assign data_wdata_ic0 = fill_wdata_ic0;
308 end
309
310 ////////////////
311 // IC0 -> IC1 //
312 ////////////////
313
314 // Tag RAMs outputs
315 assign ic_tag_req_o = {IC_NUM_WAYS{tag_req_ic0}} & tag_banks_ic0;
316 assign ic_tag_write_o = tag_write_ic0;
317 assign ic_tag_addr_o = tag_index_ic0;
318 assign ic_tag_wdata_o = tag_wdata_ic0;
319
320 // Tag RAMs inputs
321 assign tag_rdata_ic1 = ic_tag_rdata_i;
322
323 // Data RAMs outputs
324 assign ic_data_req_o = {IC_NUM_WAYS{data_req_ic0}} & data_banks_ic0;
325 assign ic_data_write_o = data_write_ic0;
326 assign ic_data_addr_o = data_index_ic0;
327 assign ic_data_wdata_o = data_wdata_ic0;
328
329 // Data RAMs inputs
330 assign data_rdata_ic1 = ic_data_rdata_i;
331
332 always_ff @(posedge clk_i or negedge rst_ni) begin
333 if (!rst_ni) begin
334 lookup_valid_ic1 <= 1'b0;
335 end else begin
336 lookup_valid_ic1 <= lookup_actual_ic0;
337 end
338 end
339
340 if (ResetAll) begin : g_lookup_addr_ra
341 always_ff @(posedge clk_i or negedge rst_ni) begin
342 if (!rst_ni) begin
343 lookup_addr_ic1 <= '0;
344 fill_in_ic1 <= '0;
345 end else if (lookup_grant_ic0) begin
346 lookup_addr_ic1 <= lookup_addr_ic0[ADDR_W-1:IC_INDEX_HI+1];
347 fill_in_ic1 <= fill_alloc_sel;
348 end
349 end
350 end else begin : g_lookup_addr_nr
351 always_ff @(posedge clk_i) begin
352 if (lookup_grant_ic0) begin
353 lookup_addr_ic1 <= lookup_addr_ic0[ADDR_W-1:IC_INDEX_HI+1];
354 fill_in_ic1 <= fill_alloc_sel;
355 end
356 end
357 end
358
359 ////////////////////////
360 // Pipeline stage IC1 //
361 ////////////////////////
362
363 // Tag matching
364 for (genvar way = 0; way < IC_NUM_WAYS; way++) begin : gen_tag_match
365 assign tag_match_ic1[way] = (tag_rdata_ic1[way][IC_TAG_SIZE-1:0] ==
366 {1'b1,lookup_addr_ic1[ADDR_W-1:IC_INDEX_HI+1]});
367 assign tag_invalid_ic1[way] = ~tag_rdata_ic1[way][IC_TAG_SIZE-1];
368 end
369
370 assign tag_hit_ic1 = |tag_match_ic1;
371
372 // Hit data mux
373 always_comb begin
374 hit_data_ecc_ic1 = 'b0;
375 for (int way = 0; way < IC_NUM_WAYS; way++) begin
376 if (tag_match_ic1[way]) begin
377 hit_data_ecc_ic1 |= data_rdata_ic1[way];
378 end
379 end
380 end
381
382 // Way selection for allocations to the cache (onehot signals)
383 // 1 first invalid way
384 // 2 global round-robin (pseudorandom) way
385 assign lowest_invalid_way_ic1[0] = tag_invalid_ic1[0];
386 assign round_robin_way_ic1[0] = round_robin_way_q[IC_NUM_WAYS-1];
387 for (genvar way = 1; way < IC_NUM_WAYS; way++) begin : gen_lowest_way
388 assign lowest_invalid_way_ic1[way] = tag_invalid_ic1[way] & ~|tag_invalid_ic1[way-1:0];
389 assign round_robin_way_ic1[way] = round_robin_way_q[way-1];
390 end
391
392 always_ff @(posedge clk_i or negedge rst_ni) begin
393 if (!rst_ni) begin
394 round_robin_way_q <= {{IC_NUM_WAYS-1{1'b0}}, 1'b1};
395 end else if (lookup_valid_ic1) begin
396 round_robin_way_q <= round_robin_way_ic1;
397 end
398 end
399
400 assign sel_way_ic1 = |tag_invalid_ic1 ? lowest_invalid_way_ic1 :
401 round_robin_way_q;
402
403 // ECC checking logic
404 if (ICacheECC) begin : gen_data_ecc_checking
405 // SEC_CM: ICACHE.MEM.INTEGRITY
406 logic [IC_NUM_WAYS-1:0] tag_err_ic1;
407 logic [IC_LINE_BEATS*2-1:0] data_err_ic1;
408 logic ecc_correction_write_d, ecc_correction_write_q;
409 logic [IC_NUM_WAYS-1:0] ecc_correction_ways_d, ecc_correction_ways_q;
410 logic [IC_INDEX_W-1:0] lookup_index_ic1, ecc_correction_index_q;
411
412 // Tag ECC checking
413 for (genvar way = 0; way < IC_NUM_WAYS; way++) begin : gen_tag_ecc
414 logic [1:0] tag_err_bank_ic1;
415 logic [27:0] tag_rdata_padded_ic1;
416
417 // Expand the tag rdata with extra padding if the tag size is less than the maximum
418 assign tag_rdata_padded_ic1 = {tag_rdata_ic1[way][TagSizeECC-1-:6],
419 {22-IC_TAG_SIZE{1'b0}},
420 tag_rdata_ic1[way][IC_TAG_SIZE-1:0]};
421
422 prim_secded_inv_28_22_dec data_ecc_dec (
423 .data_i (tag_rdata_padded_ic1),
424 .data_o (),
425 .syndrome_o (),
426 .err_o (tag_err_bank_ic1)
427 );
428 assign tag_err_ic1[way] = |tag_err_bank_ic1;
429 end
430
431 // Data ECC checking
432 // Note - could generate for all ways and mux after
433 for (genvar bank = 0; bank < IC_LINE_BEATS; bank++) begin : gen_ecc_banks
434 prim_secded_inv_39_32_dec data_ecc_dec (
435 .data_i (hit_data_ecc_ic1[bank*BusSizeECC+:BusSizeECC]),
436 .data_o (),
437 .syndrome_o (),
438 .err_o (data_err_ic1[bank*2+:2])
439 );
440
441 assign hit_data_ic1[bank*BUS_SIZE+:BUS_SIZE] =
442 hit_data_ecc_ic1[bank*BusSizeECC+:BUS_SIZE];
443
444 end
445
446 // Tag ECC across all ways is always expected to be correct so the check does not need to be
447 // qualified by hit or tag valid. Initial (invalid with correct ECC) tags are written on reset
448 // and all further tag writes produce correct ECC. For data ECC no initialisation is done on
449 // reset so unused data (in particular those ways that don't have a valid tag) may have
450 // incorrect ECC. We only check data ECC where tags indicate it is valid and we have hit on it.
451 assign ecc_err_ic1 = lookup_valid_ic1 & (((|data_err_ic1) & tag_hit_ic1) | (|tag_err_ic1));
452
453 // Error correction
454 // All ways will be invalidated on a tag error to prevent X-propagation from data_err_ic1 on
455 // spurious hits. Also prevents the same line being allocated twice when there was a true
456 // hit and a spurious hit.
457 assign ecc_correction_ways_d = {IC_NUM_WAYS{|tag_err_ic1}} |
458 (tag_match_ic1 & {IC_NUM_WAYS{|data_err_ic1}});
459 assign ecc_correction_write_d = ecc_err_ic1;
460
461 always_ff @(posedge clk_i or negedge rst_ni) begin
462 if (!rst_ni) begin
463 ecc_correction_write_q <= 1'b0;
464 end else begin
465 ecc_correction_write_q <= ecc_correction_write_d;
466 end
467 end
468
469 // The index is required in IC1 only when ECC is configured so is registered here
470 if (ResetAll) begin : g_lookup_ind_ra
471 always_ff @(posedge clk_i or negedge rst_ni) begin
472 if (!rst_ni) begin
473 lookup_index_ic1 <= '0;
474 end else if (lookup_grant_ic0) begin
475 lookup_index_ic1 <= lookup_addr_ic0[IC_INDEX_HI-:IC_INDEX_W];
476 end
477 end
478 end else begin : g_lookup_ind_nr
479 always_ff @(posedge clk_i) begin
480 if (lookup_grant_ic0) begin
481 lookup_index_ic1 <= lookup_addr_ic0[IC_INDEX_HI-:IC_INDEX_W];
482 end
483 end
484 end
485
486 // Store the ways with errors to be invalidated
487 if (ResetAll) begin : g_ecc_correction_ra
488 always_ff @(posedge clk_i or negedge rst_ni) begin
489 if (!rst_ni) begin
490 ecc_correction_ways_q <= '0;
491 ecc_correction_index_q <= '0;
492 end else if (ecc_err_ic1) begin
493 ecc_correction_ways_q <= ecc_correction_ways_d;
494 ecc_correction_index_q <= lookup_index_ic1;
495 end
496 end
497 end else begin : g_ecc_correction_nr
498 always_ff @(posedge clk_i) begin
499 if (ecc_err_ic1) begin
500 ecc_correction_ways_q <= ecc_correction_ways_d;
501 ecc_correction_index_q <= lookup_index_ic1;
502 end
503 end
504 end
505
506 assign ecc_write_req = ecc_correction_write_q;
507 assign ecc_write_ways = ecc_correction_ways_q;
508 assign ecc_write_index = ecc_correction_index_q;
509
510 assign ecc_error_o = ecc_err_ic1;
511 end else begin : gen_no_data_ecc
512 assign ecc_err_ic1 = 1'b0;
513 assign ecc_write_req = 1'b0;
514 assign ecc_write_ways = '0;
515 assign ecc_write_index = '0;
516 assign hit_data_ic1 = hit_data_ecc_ic1;
517
518 assign ecc_error_o = 1'b0;
519 end

521 ///////////////////////////////
522 // Cache allocation decision //
523 ///////////////////////////////
524
525 if (BranchCache) begin : gen_caching_logic
526
527 // Cache branch target + a number of subsequent lines
528 localparam int unsigned CACHE_AHEAD = 2;
529 localparam int unsigned CACHE_CNT_W = (CACHE_AHEAD == 1) ? 1 : $clog2(CACHE_AHEAD) + 1;
530 logic cache_cnt_dec;
531 logic [CACHE_CNT_W-1:0] cache_cnt_d, cache_cnt_q;
532
533 assign cache_cnt_dec = lookup_grant_ic0 & (|cache_cnt_q);
534 assign cache_cnt_d = branch_i ? CACHE_AHEAD[CACHE_CNT_W-1:0] :
535 (cache_cnt_q - {{CACHE_CNT_W-1{1'b0}},cache_cnt_dec});
536
537 always_ff @(posedge clk_i or negedge rst_ni) begin
538 if (!rst_ni) begin
539 cache_cnt_q <= '0;
540 end else begin
541 cache_cnt_q <= cache_cnt_d;
542 end
543 end
544
545 assign fill_cache_new = (branch_i | (|cache_cnt_q)) & icache_enable_i & ~inval_block_cache;
546 end else begin : gen_cache_all
547
548 // Cache all missing fetches
549 assign fill_cache_new = icache_enable_i & ~inval_block_cache;
550 end
551
552 //////////////////////////
553 // Fill buffer tracking //
554 //////////////////////////
555
556 always_comb begin
557 fb_fill_level = '0;
558 for (int i = 0; i < NUM_FB; i++) begin
559 if (fill_busy_q[i] & ~fill_stale_q[i]) begin
560 fb_fill_level += {{$clog2(NUM_FB) - 1{1'b0}}, 1'b1};
561 end
562 end
563 end
564
565 // Allocate a new buffer for every granted lookup
566 assign fill_new_alloc = lookup_grant_ic0;
567 // Track whether a speculative external request was made from IC0, and whether it was granted
568 // Speculative requests are only made for branches, or if the cache is disabled
569 assign fill_spec_req = (~icache_enable_i | branch_i) & ~|fill_ext_req;
570 assign fill_spec_done = fill_spec_req & instr_gnt_i;
571 assign fill_spec_hold = fill_spec_req & ~instr_gnt_i;
572
573 for (genvar fb = 0; fb < NUM_FB; fb++) begin : gen_fbs
574
575 /////////////////////////////
576 // Fill buffer allocations //
577 /////////////////////////////
578
579 // Allocate the lowest available buffer
580 if (fb == 0) begin : gen_fb_zero
581 assign fill_alloc_sel[fb] = ~fill_busy_q[fb];
582 end else begin : gen_fb_rest
583 assign fill_alloc_sel[fb] = ~fill_busy_q[fb] & (&fill_busy_q[fb-1:0]);
584 end
585
586 assign fill_alloc[fb] = fill_alloc_sel[fb] & fill_new_alloc;
587 assign fill_busy_d[fb] = fill_alloc[fb] | (fill_busy_q[fb] & ~fill_done[fb]);
588
589 // Track which other fill buffers are older than this one (for age-based arbitration)
590 // TODO sparsify
591 assign fill_older_d[fb] = (fill_alloc[fb] ? fill_busy_q : fill_older_q[fb]) & ~fill_done;
592
593 // A fill buffer can release once all its actions are completed
594 // all data written to the cache (unless hit or error)
595 assign fill_done[fb] = (fill_ram_done_q[fb] | fill_hit_q[fb] | ~fill_cache_q[fb] |
596 (|fill_err_q[fb])) &
597 // all data output unless stale due to intervening branch
598 (fill_out_done[fb] | fill_stale_q[fb] | branch_i) &
599 // all external requests completed
600 fill_rvd_done[fb];
601
602 /////////////////////////////////
603 // Fill buffer status tracking //
604 /////////////////////////////////
605
606 // Track staleness (requests become stale when a branch intervenes)
607 assign fill_stale_d[fb] = fill_busy_q[fb] & (branch_i | fill_stale_q[fb]);
608 // Track whether or not this request should allocate to the cache
609 // Any invalidation or disabling of the cache while the buffer is busy will stop allocation
610 assign fill_cache_d[fb] = (fill_alloc[fb] & fill_cache_new) |
611 (fill_cache_q[fb] & fill_busy_q[fb] &
612 icache_enable_i & ~icache_inval_i);
613 // Record whether the request hit in the cache
614 assign fill_hit_ic1[fb] = lookup_valid_ic1 & fill_in_ic1[fb] & tag_hit_ic1 & ~ecc_err_ic1;
615 assign fill_hit_d[fb] = fill_hit_ic1[fb] | (fill_hit_q[fb] & fill_busy_q[fb]);
616
617 ///////////////////////////////////////////
618 // Fill buffer external request tracking //
619 ///////////////////////////////////////////
620
621 // Make an external request
622 assign fill_ext_req[fb] = fill_busy_q[fb] & ~fill_ext_done_d[fb];
623
624 // Count the number of completed external requests (each line requires IC_LINE_BEATS requests)
625 assign fill_ext_cnt_d[fb] = fill_alloc[fb] ?
626 {{IC_LINE_BEATS_W{1'b0}},fill_spec_done} :
627 (fill_ext_cnt_q[fb] + {{IC_LINE_BEATS_W{1'b0}},
628 fill_ext_arb[fb] & instr_gnt_i});
629 // External request must be held until granted
630 assign fill_ext_hold_d[fb] = (fill_alloc[fb] & fill_spec_hold) |
631 (fill_ext_arb[fb] & ~instr_gnt_i);
632 // External requests are completed when the counter is filled or when the request is cancelled
633 assign fill_ext_done_d[fb] = (fill_ext_cnt_q[fb][IC_LINE_BEATS_W] |
634 // external requests are considered complete if the request hit
635 fill_hit_ic1[fb] | fill_hit_q[fb] |
636 // cancel if the line won't be cached and, it is stale
637 (~fill_cache_q[fb] & (branch_i | fill_stale_q[fb] |
638 // or we're already at the end of the line
639 fill_ext_beat[fb][IC_LINE_BEATS_W]))) &
640 // can't cancel while we are waiting for a grant on the bus
641 ~fill_ext_hold_q[fb] & fill_busy_q[fb];
642 // Track whether this fill buffer expects to receive beats of data
643 assign fill_rvd_exp[fb] = fill_busy_q[fb] & ~fill_rvd_done[fb];
644 // Count the number of rvalid beats received
645 assign fill_rvd_cnt_d[fb] = fill_alloc[fb] ? '0 :
646 (fill_rvd_cnt_q[fb] +
647 {{IC_LINE_BEATS_W{1'b0}},fill_rvd_arb[fb]});
648 // External data is complete when all issued external requests have received their data
649 assign fill_rvd_done[fb] = (fill_ext_done_q[fb] & ~fill_ext_hold_q[fb]) &
650 (fill_rvd_cnt_q[fb] == fill_ext_cnt_q[fb]);
651
652 //////////////////////////////////////
653 // Fill buffer data output tracking //
654 //////////////////////////////////////
655
656 // Send data to the IF stage for requests that are not stale, have not completed their
657 // data output, and have data available to send.
658 // Data is available if:
659 // - The request hit in the cache
660 // - Buffered data is available (fill_rvd_cnt_q is ahead of fill_out_cnt_q)
661 // - Data is available from the bus this cycle (fill_rvd_arb)
662 assign fill_out_req[fb] = fill_busy_q[fb] & ~fill_stale_q[fb] & ~fill_out_done[fb] &
663 (fill_hit_ic1[fb] | fill_hit_q[fb] |
664 (fill_rvd_beat[fb] > fill_out_cnt_q[fb]) | fill_rvd_arb[fb]);
665
666 // Calculate when a beat of data is output. Any ECC error squashes the output that cycle.
667 assign fill_out_grant[fb] = fill_out_arb[fb] & output_ready;
668
669 // Count the beats of data output to the IF stage
670 assign fill_out_cnt_d[fb] = fill_alloc[fb] ? {1'b0,lookup_addr_ic0[IC_LINE_W-1:BUS_W]} :
671 (fill_out_cnt_q[fb] +
672 {{IC_LINE_BEATS_W{1'b0}},fill_out_grant[fb]});
673 // Data output complete when the counter fills
674 assign fill_out_done[fb] = fill_out_cnt_q[fb][IC_LINE_BEATS_W];
675
676 //////////////////////////////////////
677 // Fill buffer ram request tracking //
678 //////////////////////////////////////
679
680 // make a fill request once all data beats received
681 assign fill_ram_req[fb] = fill_busy_q[fb] & fill_rvd_cnt_q[fb][IC_LINE_BEATS_W] &
682 // unless the request hit, was non-allocating or got an error
683 ~fill_hit_q[fb] & fill_cache_q[fb] & ~|fill_err_q[fb] &
684 // or the request was already completed
685 ~fill_ram_done_q[fb];
686
687 // Record when a cache allocation request has been completed
688 assign fill_ram_done_d[fb] = fill_ram_arb[fb] | (fill_ram_done_q[fb] & fill_busy_q[fb]);
689
690 //////////////////////////////
691 // Fill buffer line offsets //
692 //////////////////////////////
693
694 // When we branch into the middle of a line, the output count will not start from zero. This
695 // beat count is used to know which incoming rdata beats are relevant.
696 assign fill_ext_beat[fb] = {1'b0,fill_addr_q[fb][IC_LINE_W-1:BUS_W]} +
697 fill_ext_cnt_q[fb][IC_LINE_BEATS_W:0];
698 assign fill_ext_off[fb] = fill_ext_beat[fb][IC_LINE_BEATS_W-1:0];
699 assign fill_rvd_beat[fb] = {1'b0,fill_addr_q[fb][IC_LINE_W-1:BUS_W]} +
700 fill_rvd_cnt_q[fb][IC_LINE_BEATS_W:0];
701 assign fill_rvd_off[fb] = fill_rvd_beat[fb][IC_LINE_BEATS_W-1:0];
702
703 /////////////////////////////
704 // Fill buffer arbitration //
705 /////////////////////////////
706
707 // Age based arbitration - all these signals are one-hot
708 assign fill_ext_arb[fb] = fill_ext_req[fb] & ~|(fill_ext_req & fill_older_q[fb]);
709 assign fill_ram_arb[fb] = fill_ram_req[fb] & fill_grant_ic0 &
710 ~|(fill_ram_req & fill_older_q[fb]);
711 // Calculate which fill buffer is the oldest one which still needs to output data to IF
712 assign fill_data_sel[fb] = ~|(fill_busy_q & ~fill_out_done & ~fill_stale_q &
713 fill_older_q[fb]);
714 // Arbitrate the request which has data available to send, and is the oldest outstanding
715 assign fill_out_arb[fb] = fill_out_req[fb] & fill_data_sel[fb];
716 // Assign incoming rvalid data to the oldest fill buffer expecting it
717 assign fill_rvd_arb[fb] = instr_rvalid_i & fill_rvd_exp[fb] &
718 ~|(fill_rvd_exp & fill_older_q[fb]);
719
720 /////////////////////////////
721 // Fill buffer data muxing //
722 /////////////////////////////
723
724 // Output data muxing controls
725 // 1. Select data from the fill buffer data register
726 assign fill_data_reg[fb] = fill_busy_q[fb] & ~fill_stale_q[fb] &
727 ~fill_out_done[fb] & fill_data_sel[fb] &
728 // The incoming data is already ahead of the output count
729 ((fill_rvd_beat[fb] > fill_out_cnt_q[fb]) | fill_hit_q[fb] |
730 (|fill_err_q[fb]));
731 // 2. Select IC1 hit data
732 assign fill_data_hit[fb] = fill_busy_q[fb] & fill_hit_ic1[fb] & fill_data_sel[fb];
733 // 3. Select incoming instr_rdata_i
734 assign fill_data_rvd[fb] = fill_busy_q[fb] & fill_rvd_arb[fb] & ~fill_hit_q[fb] &
735 ~fill_hit_ic1[fb] & ~fill_stale_q[fb] & ~fill_out_done[fb] &
736 // The incoming data lines up with the output count
737 (fill_rvd_beat[fb] == fill_out_cnt_q[fb]) & fill_data_sel[fb];
738
739
740 ///////////////////////////
741 // Fill buffer registers //
742 ///////////////////////////
743
744 // Fill buffer general enable
745 assign fill_entry_en[fb] = fill_alloc[fb] | fill_busy_q[fb];
746
747 always_ff @(posedge clk_i or negedge rst_ni) begin
748 if (!rst_ni) begin
749 fill_busy_q[fb] <= 1'b0;
750 fill_older_q[fb] <= '0;
751 fill_stale_q[fb] <= 1'b0;
752 fill_cache_q[fb] <= 1'b0;
753 fill_hit_q[fb] <= 1'b0;
754 fill_ext_cnt_q[fb] <= '0;
755 fill_ext_hold_q[fb] <= 1'b0;
756 fill_ext_done_q[fb] <= 1'b0;
757 fill_rvd_cnt_q[fb] <= '0;
758 fill_ram_done_q[fb] <= 1'b0;
759 fill_out_cnt_q[fb] <= '0;
760 end else if (fill_entry_en[fb]) begin
761 fill_busy_q[fb] <= fill_busy_d[fb];
762 fill_older_q[fb] <= fill_older_d[fb];
763 fill_stale_q[fb] <= fill_stale_d[fb];
764 fill_cache_q[fb] <= fill_cache_d[fb];
765 fill_hit_q[fb] <= fill_hit_d[fb];
766 fill_ext_cnt_q[fb] <= fill_ext_cnt_d[fb];
767 fill_ext_hold_q[fb] <= fill_ext_hold_d[fb];
768 fill_ext_done_q[fb] <= fill_ext_done_d[fb];
769 fill_rvd_cnt_q[fb] <= fill_rvd_cnt_d[fb];
770 fill_ram_done_q[fb] <= fill_ram_done_d[fb];
771 fill_out_cnt_q[fb] <= fill_out_cnt_d[fb];
772 end
773 end
774
775 ////////////////////////////////////////
776 // Fill buffer address / data storage //
777 ////////////////////////////////////////
778
779 assign fill_addr_en[fb] = fill_alloc[fb];
780 assign fill_way_en[fb] = (lookup_valid_ic1 & fill_in_ic1[fb]);
781
782 if (ResetAll) begin : g_fill_addr_ra
783 always_ff @(posedge clk_i or negedge rst_ni) begin
784 if (!rst_ni) begin
785 fill_addr_q[fb] <= '0;
786 end else if (fill_addr_en[fb]) begin
787 fill_addr_q[fb] <= lookup_addr_ic0;
788 end
789 end
790 end else begin : g_fill_addr_nr
791 always_ff @(posedge clk_i) begin
792 if (fill_addr_en[fb]) begin
793 fill_addr_q[fb] <= lookup_addr_ic0;
794 end
795 end
796 end
797
798 always_ff @(posedge clk_i or negedge rst_ni) begin
799 if (!rst_ni) begin
800 fill_way_q[fb] <= '0;
801 end else if (fill_way_en[fb]) begin
802 fill_way_q[fb] <= sel_way_ic1;
803 end
804 end
805
806 // Data either comes from the cache or the bus. If there was an ECC error, we must take
807 // the incoming bus data since the cache hit data is corrupted.
808 assign fill_data_d[fb] = fill_hit_ic1[fb] ? hit_data_ic1 :
809 {IC_LINE_BEATS{instr_rdata_i}};
810
811 for (genvar b = 0; b < IC_LINE_BEATS; b++) begin : gen_data_buf
812 // Error tracking (per beat)
813 assign fill_err_d[fb][b] = (fill_rvd_arb[fb] & instr_err_i &
814 (fill_rvd_off[fb] == b[IC_LINE_BEATS_W-1:0])) |
815 // Hold the error once recorded
816 (fill_busy_q[fb] & fill_err_q[fb][b]);
817
818 always_ff @(posedge clk_i or negedge rst_ni) begin
819 if (!rst_ni) begin
820 fill_err_q[fb][b] <= '0;
821 end else if (fill_entry_en[fb]) begin
822 fill_err_q[fb][b] <= fill_err_d[fb][b];
823 end
824 end
825
826 // Enable the relevant part of the data register (or all for cache hits)
827 // Ignore incoming rvalid data when we already have cache hit data
828 assign fill_data_en[fb][b] = fill_hit_ic1[fb] |
829 (fill_rvd_arb[fb] & ~fill_hit_q[fb] &
830 (fill_rvd_off[fb] == b[IC_LINE_BEATS_W-1:0]));
831
832 if (ResetAll) begin : g_fill_data_ra
833 always_ff @(posedge clk_i or negedge rst_ni) begin
834 if (!rst_ni) begin
835 fill_data_q[fb][b*BUS_SIZE+:BUS_SIZE] <= '0;
836 end else if (fill_data_en[fb][b]) begin
837 fill_data_q[fb][b*BUS_SIZE+:BUS_SIZE] <= fill_data_d[fb][b*BUS_SIZE+:BUS_SIZE];
838 end
839 end
840 end else begin : g_fill_data_nr
841 always_ff @(posedge clk_i) begin
842 if (fill_data_en[fb][b]) begin
843 fill_data_q[fb][b*BUS_SIZE+:BUS_SIZE] <= fill_data_d[fb][b*BUS_SIZE+:BUS_SIZE];
844 end
845 end
846 end
847
848 end
849 end

851 ////////////////////////////////
852 // Fill buffer one-hot muxing //
853 ////////////////////////////////
854
855 // External req info
856 always_comb begin
857 fill_ext_req_addr = '0;
858 for (int i = 0; i < NUM_FB; i++) begin
859 if (fill_ext_arb[i]) begin
860 fill_ext_req_addr |= {fill_addr_q[i][ADDR_W-1:IC_LINE_W], fill_ext_off[i]};
861 end
862 end
863 end
864
865 // Cache req info
866 always_comb begin
867 fill_ram_req_addr = '0;
868 fill_ram_req_way = '0;
869 fill_ram_req_data = '0;
870 for (int i = 0; i < NUM_FB; i++) begin
871 if (fill_ram_arb[i]) begin
872 fill_ram_req_addr |= fill_addr_q[i];
873 fill_ram_req_way |= fill_way_q[i];
874 fill_ram_req_data |= fill_data_q[i];
875 end
876 end
877 end
878
879 // IF stage output data
880 always_comb begin
881 fill_out_data = '0;
882 fill_out_err = '0;
883 for (int i = 0; i < NUM_FB; i++) begin
884 if (fill_data_reg[i]) begin
885 fill_out_data |= fill_data_q[i];
886 // Ignore any speculative errors accumulated on cache hits
887 fill_out_err |= (fill_err_q[i] & ~{IC_LINE_BEATS{fill_hit_q[i]}});
888 end
889 end
890 end
891
892 ///////////////////////
893 // External requests //
894 ///////////////////////
895
896 assign instr_req = ((~icache_enable_i | branch_i) & lookup_grant_ic0) |
897 (|fill_ext_req);
898
899 assign instr_addr = |fill_ext_req ? fill_ext_req_addr :
900 lookup_addr_ic0[ADDR_W-1:BUS_W];
901
902 assign instr_req_o = instr_req;
903 assign instr_addr_o = {instr_addr[ADDR_W-1:BUS_W],{BUS_W{1'b0}}};
904
905 ////////////////////////
906 // Output data muxing //
907 ////////////////////////
908
909 // Mux between line-width data sources
910 assign line_data = |fill_data_hit ? hit_data_ic1 : fill_out_data;
911 assign line_err = |fill_data_hit ? {IC_LINE_BEATS{1'b0}} : fill_out_err;
912
913 // Mux the relevant beat of line data, based on the output address
914 always_comb begin
915 line_data_muxed = '0;
916 line_err_muxed = 1'b0;
917 for (int unsigned i = 0; i < IC_LINE_BEATS; i++) begin
918 // When data has been skidded, the output address is behind by one
919 if ((output_addr_q[IC_LINE_W-1:BUS_W] + {{IC_LINE_BEATS_W-1{1'b0}},skid_valid_q}) ==
920 i[IC_LINE_BEATS_W-1:0]) begin
921 line_data_muxed |= line_data[i*32+:32];
922 line_err_muxed |= line_err[i];
923 end
924 end
925 end
926
927 // Mux between incoming rdata and the muxed line data
928 assign output_data = |fill_data_rvd ? instr_rdata_i : line_data_muxed;
929 assign output_err = |fill_data_rvd ? instr_err_i : line_err_muxed;
930
931 // Output data is valid (from any of the three possible sources). Note that fill_out_arb
932 // must be used here rather than fill_out_req because data can become valid out of order
933 // (e.g. cache hit data can become available ahead of an older outstanding miss).
934 assign data_valid = |fill_out_arb;
935
936 // Skid buffer data
937 assign skid_data_d = output_data[31:16];
938
939 assign skid_en = data_valid & (ready_i | skid_ready);
940
941 if (ResetAll) begin : g_skid_data_ra
942 always_ff @(posedge clk_i or negedge rst_ni) begin
943 if (!rst_ni) begin
944 skid_data_q <= '0;
945 skid_err_q <= '0;
946 end else if (skid_en) begin
947 skid_data_q <= skid_data_d;
948 skid_err_q <= output_err;
949 end
950 end
951 end else begin : g_skid_data_nr
952 always_ff @(posedge clk_i) begin
953 if (skid_en) begin
954 skid_data_q <= skid_data_d;
955 skid_err_q <= output_err;
956 end
957 end
958 end
959
960 // The data in the skid buffer is ready if it's a complete compressed instruction or if there's
961 // an error (no need to wait for the second half)
962 assign skid_complete_instr = skid_valid_q & ((skid_data_q[1:0] != 2'b11) | skid_err_q);
963
964 // Data can be loaded into the skid buffer for an unaligned uncompressed instruction
965 assign skid_ready = output_addr_q[1] & ~skid_valid_q & (~output_compressed | output_err);
966
967 assign output_ready = (ready_i | skid_ready) & ~skid_complete_instr;
968
969 assign output_compressed = (rdata_o[1:0] != 2'b11);
970
971 assign skid_valid_d =
972 // Branches invalidate the skid buffer
973 branch_i ? 1'b0 :
974 // Once valid, the skid buffer stays valid until a compressed instruction realigns the stream
975 (skid_valid_q ? ~(ready_i & ((skid_data_q[1:0] != 2'b11) | skid_err_q)) :
976 // The skid buffer becomes valid when:
977 // - we branch to an unaligned uncompressed instruction
978 (data_valid &
979 (((output_addr_q[1] & (~output_compressed | output_err)) |
980 // - a compressed instruction misaligns the stream
981 (~output_addr_q[1] & output_compressed & ~output_err & ready_i)))));
982
983 always_ff @(posedge clk_i or negedge rst_ni) begin
984 if (!rst_ni) begin
985 skid_valid_q <= 1'b0;
986 end else begin
987 skid_valid_q <= skid_valid_d;
988 end
989 end
990
991 // Signal that valid data is available to the IF stage
992 // Note that if the first half of an unaligned instruction reports an error, we do not need
993 // to wait for the second half
994 // Compressed instruction completely satisfied by skid buffer
995 assign output_valid = skid_complete_instr |
996 // Output data available and, output stream aligned, or skid data available,
997 (data_valid & (~output_addr_q[1] | skid_valid_q |
998 // or this is an error or an unaligned compressed instruction
999 output_err | (output_data[17:16] != 2'b11)));
1000
1001 // Update the address on branches and every time an instruction is driven
1002 assign output_addr_en = branch_i | (ready_i & valid_o);
1003
1004 // Increment the address by two every time a compressed instruction is popped
1005 assign addr_incr_two = output_compressed & ~err_o;
1006
1007 // Next IF stage PC
1008 assign output_addr_incr = (output_addr_q[31:1] +
1009 // Increment address by 4 or 2
1010 {29'd0, ~addr_incr_two, addr_incr_two});
1011
1012 // Redirect the address on branches
1013 assign output_addr_d = branch_i ? addr_i[31:1] : output_addr_incr;
1014
1015 if (ResetAll) begin : g_output_addr_ra
1016 always_ff @(posedge clk_i or negedge rst_ni) begin
1017 if (!rst_ni) begin
1018 output_addr_q <= '0;
1019 end else if (output_addr_en) begin
1020 output_addr_q <= output_addr_d;
1021 end
1022 end
1023 end else begin : g_output_addr_nr
1024 always_ff @(posedge clk_i) begin
1025 if (output_addr_en) begin
1026 output_addr_q <= output_addr_d;
1027 end
1028 end
1029 end
1030
1031 // Mux the data from BUS_SIZE to halfword
1032 // This muxing realigns data when instruction words are split across BUS_W e.g.
1033 // word 1 |----|*h1*|
1034 // word 0 |*h0*|----| --> |*h1*|*h0*|
1035 // 31 15 0 31 15 0
1036 always_comb begin
1037 output_data_lo = '0;
1038 for (int unsigned i = 0; i < IC_OUTPUT_BEATS; i++) begin
1039 if (output_addr_q[BUS_W-1:1] == i[BUS_W-2:0]) begin
1040 output_data_lo |= output_data[i*16+:16];
1041 end
1042 end
1043 end
1044
1045 always_comb begin
1046 output_data_hi = '0;
1047 for (int unsigned i = 0; i < IC_OUTPUT_BEATS - 1; i++) begin
1048 if (output_addr_q[BUS_W-1:1] == i[BUS_W-2:0]) begin
1049 output_data_hi |= output_data[(i+1)*16+:16];
1050 end
1051 end
1052 if (&output_addr_q[BUS_W-1:1]) begin
1053 output_data_hi |= output_data[15:0];
1054 end
1055 end

1057 assign valid_o = output_valid;

1066 // Invalidations //
1067 ///////////////////
1068
1069 // Invalidation (writing all entries in the tag RAM with an invalid tag) occurs straight out of
1070 // reset and after any invalidation request (signalled via icache_inval_i). An invalidation
1071 // request coming whilst another is writing tags causes the invalidation to start again. This
1072 // ensures a new scramble key is requested where a previous one is in use.
1073 // TODO: Ditch this behaviour for non-secure ibex?
1074 always_comb begin
1075 inval_state_d = inval_state_q;
1076 inval_index_d = inval_index_q;
1077 inval_index_en = 1'b0;
1078 inval_write_req = 1'b0;
1079 ic_scr_key_req_o = 1'b0;
1080
1081 // Prevent other cache activity (cache lookups and cache allocations) whilst an invalidation is
1082 // in progress. Set to 1 by default as the only time we don't block is when the state machine is
1083 // IDLE.
1084 inval_block_cache = 1'b1;
1085
1086 unique case (inval_state_q)
1087 OUT_OF_RESET: begin
1088 // Initial state, this initialises the tag RAMs out of reset before the icache can be used
1089 inval_state_d = AWAIT_SCRAMBLE_KEY;
1090
1091 if (~ic_scr_key_valid_i) begin
1092 ic_scr_key_req_o = 1'b1;
1093 end
1094 end
1095 AWAIT_SCRAMBLE_KEY: begin
1096 // When invalidating a new scrambling key is requested on all invalidation requests. Wait
1097 // for that new key to be available before beginning with the actual invalidation (cannot
1098 // write to the tag RAM until we have the new scrambling key that will be used). Ignore any
1099 // requests in this phase (once a scramble key request has started we cannot request a new
1100 // one until the on-going request is done).
1101 if (ic_scr_key_valid_i) begin
1102 inval_state_d = INVAL_CACHE;
1103 inval_index_d = '0;
1104 inval_index_en = 1'b1;
1105 end
1106 end
1107 INVAL_CACHE: begin
1108 // Actually invalidate the cache. Write every entry in the tag RAM with an invalid tag. Once
1109 // all are written we're done.
1110 inval_write_req = 1'b1;
1111 inval_index_d = (inval_index_q + {{IC_INDEX_W-1{1'b0}},1'b1});
1112 inval_index_en = 1'b1;
1113
1114 if (icache_inval_i) begin
1115 // If a new invalidaiton requests comes in go back to the beginning with a new scramble
1116 // key
1117 ic_scr_key_req_o = 1'b1;
1118 inval_state_d = AWAIT_SCRAMBLE_KEY;
1119 end else if (&inval_index_q) begin
1120 // When the final index is written we're done
1121 inval_state_d = IDLE;
1122 end
1123 end
1124 IDLE: begin
1125 // Usual running state
1126 if (icache_inval_i) begin
1127 ic_scr_key_req_o = 1'b1;
1128 inval_state_d = AWAIT_SCRAMBLE_KEY;
1129 end else begin
1130 // Allow other cache activies whilst in IDLE and no invalidation has been requested
1131 inval_block_cache = 1'b0;
1132 end
1133 end
1134 default: ;
1135 endcase
1136 end
1137
1138 assign inval_active = inval_state_q != IDLE;
1139
1140 always_ff @(posedge clk_i or negedge rst_ni) begin
1141 if (!rst_ni) begin
1142 inval_state_q <= OUT_OF_RESET;
1143 end else begin
1144 inval_state_q <= inval_state_d;
1145 end
1146 end
1147
1148 if (ResetAll) begin : g_inval_index_ra
1149 always_ff @(posedge clk_i or negedge rst_ni) begin
1150 if (!rst_ni) begin
1151 inval_index_q <= '0;
1152 end else if (inval_index_en) begin
1153 inval_index_q <= inval_index_d;
1154 end
1155 end
1156 end else begin : g_inval_index_nr
1157 always_ff @(posedge clk_i) begin
1158 if (inval_index_en) begin
1159 inval_index_q <= inval_index_d;
1160 end
1161 end
1162 end
