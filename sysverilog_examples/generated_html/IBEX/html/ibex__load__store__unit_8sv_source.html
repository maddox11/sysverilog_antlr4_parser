7/**
8 * Load Store Unit
9 *
10 * Load Store Unit, used to eliminate multiple access during processor stalls,
11 * and to align bytes and halfwords.
12 */
13
14`include "prim_assert.sv"
15`include "dv_fcov_macros.svh"
16
17module ibex_load_store_unit #(
18 parameter bit MemECC = 1'b0,
19 parameter int unsigned MemDataWidth = MemECC ? 32 + 7 : 32
20) (
21 input logic clk_i,
22 input logic rst_ni,
23
24 // data interface
25 output logic data_req_o,
26 input logic data_gnt_i,
27 input logic data_rvalid_i,
28 input logic data_bus_err_i,
29 input logic data_pmp_err_i,
30
31 output logic [31:0] data_addr_o,
32 output logic data_we_o,
33 output logic [3:0] data_be_o,
34 output logic [MemDataWidth-1:0] data_wdata_o,
35 input logic [MemDataWidth-1:0] data_rdata_i,
36
37 // signals to/from ID/EX stage
38 input logic lsu_we_i, // write enable -> from ID/EX
39 input logic [1:0] lsu_type_i, // data type: word, half word, byte -> from ID/EX
40 input logic [31:0] lsu_wdata_i, // data to write to memory -> from ID/EX
41 input logic lsu_sign_ext_i, // sign extension -> from ID/EX
42
43 output logic [31:0] lsu_rdata_o, // requested data -> to ID/EX
44 output logic lsu_rdata_valid_o,
45 input logic lsu_req_i, // data request -> from ID/EX
46
47 input logic [31:0] adder_result_ex_i, // address computed in ALU -> from ID/EX
48
49 output logic addr_incr_req_o, // request address increment for
50 // misaligned accesses -> to ID/EX
51 output logic [31:0] addr_last_o, // address of last transaction -> to controller
52 // -> mtval
53 // -> AGU for misaligned accesses
54
55 output logic lsu_req_done_o, // Signals that data request is complete
56 // (only need to await final data
57 // response) -> to ID/EX
58
59 output logic lsu_resp_valid_o, // LSU has response from transaction -> to ID/EX
60
61 // exception signals
62 output logic load_err_o,
63 output logic load_resp_intg_err_o,
64 output logic store_err_o,
65 output logic store_resp_intg_err_o,
66
67 output logic busy_o,
68
69 output logic perf_load_o,
70 output logic perf_store_o
71);
72
73 logic [31:0] data_addr;
74 logic [31:0] data_addr_w_aligned;
75 logic [31:0] addr_last_q, addr_last_d;
76
77 logic addr_update;
78 logic ctrl_update;
79 logic rdata_update;
80 logic [31:8] rdata_q;
81 logic [1:0] rdata_offset_q;
82 logic [1:0] data_type_q;
83 logic data_sign_ext_q;
84 logic data_we_q;
85
86 logic [1:0] data_offset; // mux control for data to be written to memory
87
88 logic [3:0] data_be;
89 logic [31:0] data_wdata;
90
91 logic [31:0] data_rdata_ext;
92
93 logic [31:0] rdata_w_ext; // word realignment for misaligned loads
94 logic [31:0] rdata_h_ext; // sign extension for half words
95 logic [31:0] rdata_b_ext; // sign extension for bytes
96
97 logic split_misaligned_access;
98 logic handle_misaligned_q, handle_misaligned_d; // high after receiving grant for first
99 // part of a misaligned access
100 logic pmp_err_q, pmp_err_d;
101 logic lsu_err_q, lsu_err_d;
102 logic data_intg_err, data_or_pmp_err;
103
104 typedef enum logic [2:0] {
105 IDLE, WAIT_GNT_MIS, WAIT_RVALID_MIS, WAIT_GNT,
106 WAIT_RVALID_MIS_GNTS_DONE
107 } ls_fsm_e;
108
109 ls_fsm_e ls_fsm_cs, ls_fsm_ns;
110
111 assign data_addr = adder_result_ex_i;
112 assign data_offset = data_addr[1:0];
113
114 ///////////////////
115 // BE generation //
116 ///////////////////
117
118 always_comb begin
119 unique case (lsu_type_i) // Data type 00 Word, 01 Half word, 11,10 byte
120 2'b00: begin // Writing a word
121 if (!handle_misaligned_q) begin // first part of potentially misaligned transaction
122 unique case (data_offset)
123 2'b00: data_be = 4'b1111;
124 2'b01: data_be = 4'b1110;
125 2'b10: data_be = 4'b1100;
126 2'b11: data_be = 4'b1000;
127 default: data_be = 4'b1111;
128 endcase // case (data_offset)
129 end else begin // second part of misaligned transaction
130 unique case (data_offset)
131 2'b00: data_be = 4'b0000; // this is not used, but included for completeness
132 2'b01: data_be = 4'b0001;
133 2'b10: data_be = 4'b0011;
134 2'b11: data_be = 4'b0111;
135 default: data_be = 4'b1111;
136 endcase // case (data_offset)
137 end
138 end
139
140 2'b01: begin // Writing a half word
141 if (!handle_misaligned_q) begin // first part of potentially misaligned transaction
142 unique case (data_offset)
143 2'b00: data_be = 4'b0011;
144 2'b01: data_be = 4'b0110;
145 2'b10: data_be = 4'b1100;
146 2'b11: data_be = 4'b1000;
147 default: data_be = 4'b1111;
148 endcase // case (data_offset)
149 end else begin // second part of misaligned transaction
150 data_be = 4'b0001;
151 end
152 end
153
154 2'b10,
155 2'b11: begin // Writing a byte
156 unique case (data_offset)
157 2'b00: data_be = 4'b0001;
158 2'b01: data_be = 4'b0010;
159 2'b10: data_be = 4'b0100;
160 2'b11: data_be = 4'b1000;
161 default: data_be = 4'b1111;
162 endcase // case (data_offset)
163 end
164
165 default: data_be = 4'b1111;
166 endcase // case (lsu_type_i)
167 end
168
169 /////////////////////
170 // WData alignment //
171 /////////////////////
172
173 // prepare data to be written to the memory
174 // we handle misaligned accesses, half word and byte accesses here
175 always_comb begin
176 unique case (data_offset)
177 2'b00: data_wdata = lsu_wdata_i[31:0];
178 2'b01: data_wdata = {lsu_wdata_i[23:0], lsu_wdata_i[31:24]};
179 2'b10: data_wdata = {lsu_wdata_i[15:0], lsu_wdata_i[31:16]};
180 2'b11: data_wdata = {lsu_wdata_i[ 7:0], lsu_wdata_i[31: 8]};
181 default: data_wdata = lsu_wdata_i[31:0];
182 endcase // case (data_offset)
183 end
184
185 /////////////////////
186 // RData alignment //
187 /////////////////////
188
189 // register for unaligned rdata
190 always_ff @(posedge clk_i or negedge rst_ni) begin
191 if (!rst_ni) begin
192 rdata_q <= '0;
193 end else if (rdata_update) begin
194 rdata_q <= data_rdata_i[31:8];
195 end
196 end
197
198 // registers for transaction control
199 always_ff @(posedge clk_i or negedge rst_ni) begin
200 if (!rst_ni) begin
201 rdata_offset_q <= 2'h0;
202 data_type_q <= 2'h0;
203 data_sign_ext_q <= 1'b0;
204 data_we_q <= 1'b0;
205 end else if (ctrl_update) begin
206 rdata_offset_q <= data_offset;
207 data_type_q <= lsu_type_i;
208 data_sign_ext_q <= lsu_sign_ext_i;
209 data_we_q <= lsu_we_i;
210 end
211 end
212
213 // Store last address for mtval + AGU for misaligned transactions. Do not update in case of
214 // errors, mtval needs the (first) failing address. Where an aligned access or the first half of
215 // a misaligned access sees an error provide the calculated access address. For the second half of
216 // a misaligned access provide the word aligned address of the second half.
217 assign addr_last_d = addr_incr_req_o ? data_addr_w_aligned : data_addr;
218
219 always_ff @(posedge clk_i or negedge rst_ni) begin
220 if (!rst_ni) begin
221 addr_last_q <= '0;
222 end else if (addr_update) begin
223 addr_last_q <= addr_last_d;
224 end
225 end
226
227 // take care of misaligned words
228 always_comb begin
229 unique case (rdata_offset_q)
230 2'b00: rdata_w_ext = data_rdata_i[31:0];
231 2'b01: rdata_w_ext = {data_rdata_i[ 7:0], rdata_q[31:8]};
232 2'b10: rdata_w_ext = {data_rdata_i[15:0], rdata_q[31:16]};
233 2'b11: rdata_w_ext = {data_rdata_i[23:0], rdata_q[31:24]};
234 default: rdata_w_ext = data_rdata_i[31:0];
235 endcase
236 end
237
238 ////////////////////
239 // Sign extension //
240 ////////////////////
241
242 // sign extension for half words
243 always_comb begin
244 unique case (rdata_offset_q)
245 2'b00: begin
246 if (!data_sign_ext_q) begin
247 rdata_h_ext = {16'h0000, data_rdata_i[15:0]};
248 end else begin
249 rdata_h_ext = {{16{data_rdata_i[15]}}, data_rdata_i[15:0]};
250 end
251 end
252
253 2'b01: begin
254 if (!data_sign_ext_q) begin
255 rdata_h_ext = {16'h0000, data_rdata_i[23:8]};
256 end else begin
257 rdata_h_ext = {{16{data_rdata_i[23]}}, data_rdata_i[23:8]};
258 end
259 end
260
261 2'b10: begin
262 if (!data_sign_ext_q) begin
263 rdata_h_ext = {16'h0000, data_rdata_i[31:16]};
264 end else begin
265 rdata_h_ext = {{16{data_rdata_i[31]}}, data_rdata_i[31:16]};
266 end
267 end
268
269 2'b11: begin
270 if (!data_sign_ext_q) begin
271 rdata_h_ext = {16'h0000, data_rdata_i[7:0], rdata_q[31:24]};
272 end else begin
273 rdata_h_ext = {{16{data_rdata_i[7]}}, data_rdata_i[7:0], rdata_q[31:24]};
274 end
275 end
276
277 default: rdata_h_ext = {16'h0000, data_rdata_i[15:0]};
278 endcase // case (rdata_offset_q)
279 end
280
281 // sign extension for bytes
282 always_comb begin
283 unique case (rdata_offset_q)
284 2'b00: begin
285 if (!data_sign_ext_q) begin
286 rdata_b_ext = {24'h00_0000, data_rdata_i[7:0]};
287 end else begin
288 rdata_b_ext = {{24{data_rdata_i[7]}}, data_rdata_i[7:0]};
289 end
290 end
291
292 2'b01: begin
293 if (!data_sign_ext_q) begin
294 rdata_b_ext = {24'h00_0000, data_rdata_i[15:8]};
295 end else begin
296 rdata_b_ext = {{24{data_rdata_i[15]}}, data_rdata_i[15:8]};
297 end
298 end
299
300 2'b10: begin
301 if (!data_sign_ext_q) begin
302 rdata_b_ext = {24'h00_0000, data_rdata_i[23:16]};
303 end else begin
304 rdata_b_ext = {{24{data_rdata_i[23]}}, data_rdata_i[23:16]};
305 end
306 end
307
308 2'b11: begin
309 if (!data_sign_ext_q) begin
310 rdata_b_ext = {24'h00_0000, data_rdata_i[31:24]};
311 end else begin
312 rdata_b_ext = {{24{data_rdata_i[31]}}, data_rdata_i[31:24]};
313 end
314 end
315
316 default: rdata_b_ext = {24'h00_0000, data_rdata_i[7:0]};
317 endcase // case (rdata_offset_q)
318 end
319
320 // select word, half word or byte sign extended version
321 always_comb begin
322 unique case (data_type_q)
323 2'b00: data_rdata_ext = rdata_w_ext;
324 2'b01: data_rdata_ext = rdata_h_ext;
325 2'b10,2'b11: data_rdata_ext = rdata_b_ext;
326 default: data_rdata_ext = rdata_w_ext;
327 endcase // case (data_type_q)
328 end
329
330 ///////////////////////////////
331 // Read data integrity check //
332 ///////////////////////////////
333
334 // SEC_CM: BUS.INTEGRITY
335 if (MemECC) begin : g_mem_rdata_ecc
336 logic [1:0] ecc_err;
337 logic [MemDataWidth-1:0] data_rdata_buf;
338
339 prim_buf #(.Width(MemDataWidth)) u_prim_buf_instr_rdata (
340 .in_i (data_rdata_i),
341 .out_o(data_rdata_buf)
342 );
343
344 prim_secded_inv_39_32_dec u_data_intg_dec (
345 .data_i (data_rdata_buf),
346 .data_o (),
347 .syndrome_o (),
348 .err_o (ecc_err)
349 );
350
351 // Don't care if error is correctable or not, they're all treated the same
352 assign data_intg_err = |ecc_err;
353 end else begin : g_no_mem_data_ecc
354 assign data_intg_err = 1'b0;
355 end
356
357 /////////////
358 // LSU FSM //
359 /////////////
360
361 // check for misaligned accesses that need to be split into two word-aligned accesses
362 assign split_misaligned_access =
363 ((lsu_type_i == 2'b00) && (data_offset != 2'b00)) || // misaligned word access
364 ((lsu_type_i == 2'b01) && (data_offset == 2'b11)); // misaligned half-word access
365
366 // FSM
367 always_comb begin
368 ls_fsm_ns = ls_fsm_cs;
369
370 data_req_o = 1'b0;
371 addr_incr_req_o = 1'b0;
372 handle_misaligned_d = handle_misaligned_q;
373 pmp_err_d = pmp_err_q;
374 lsu_err_d = lsu_err_q;
375
376 addr_update = 1'b0;
377 ctrl_update = 1'b0;
378 rdata_update = 1'b0;
379
380 perf_load_o = 1'b0;
381 perf_store_o = 1'b0;
382
383 unique case (ls_fsm_cs)
384
385 IDLE: begin
386 pmp_err_d = 1'b0;
387 if (lsu_req_i) begin
388 data_req_o = 1'b1;
389 pmp_err_d = data_pmp_err_i;
390 lsu_err_d = 1'b0;
391 perf_load_o = ~lsu_we_i;
392 perf_store_o = lsu_we_i;
393
394 if (data_gnt_i) begin
395 ctrl_update = 1'b1;
396 addr_update = 1'b1;
397 handle_misaligned_d = split_misaligned_access;
398 ls_fsm_ns = split_misaligned_access ? WAIT_RVALID_MIS : IDLE;
399 end else begin
400 ls_fsm_ns = split_misaligned_access ? WAIT_GNT_MIS : WAIT_GNT;
401 end
402 end
403 end
404
405 WAIT_GNT_MIS: begin
406 data_req_o = 1'b1;
407 // data_pmp_err_i is valid during the address phase of a request. An error will block the
408 // external request and so a data_gnt_i might never be signalled. The registered version
409 // pmp_err_q is only updated for new address phases and so can be used in WAIT_GNT* and
410 // WAIT_RVALID* states
411 if (data_gnt_i || pmp_err_q) begin
412 addr_update = 1'b1;
413 ctrl_update = 1'b1;
414 handle_misaligned_d = 1'b1;
415 ls_fsm_ns = WAIT_RVALID_MIS;
416 end
417 end
418
419 WAIT_RVALID_MIS: begin
420 // push out second request
421 data_req_o = 1'b1;
422 // tell ID/EX stage to update the address
423 addr_incr_req_o = 1'b1;
424
425 // first part rvalid is received, or gets a PMP error
426 if (data_rvalid_i || pmp_err_q) begin
427 // Update the PMP error for the second part
428 pmp_err_d = data_pmp_err_i;
429 // Record the error status of the first part
430 lsu_err_d = data_bus_err_i | pmp_err_q;
431 // Capture the first rdata for loads
432 rdata_update = ~data_we_q;
433 // If already granted, wait for second rvalid
434 ls_fsm_ns = data_gnt_i ? IDLE : WAIT_GNT;
435 // Update the address for the second part, if no error
436 addr_update = data_gnt_i & ~(data_bus_err_i | pmp_err_q);
437 // clear handle_misaligned if second request is granted
438 handle_misaligned_d = ~data_gnt_i;
439 end else begin
440 // first part rvalid is NOT received
441 if (data_gnt_i) begin
442 // second grant is received
443 ls_fsm_ns = WAIT_RVALID_MIS_GNTS_DONE;
444 handle_misaligned_d = 1'b0;
445 end
446 end
447 end
448
449 WAIT_GNT: begin
450 // tell ID/EX stage to update the address
451 addr_incr_req_o = handle_misaligned_q;
452 data_req_o = 1'b1;
453 if (data_gnt_i || pmp_err_q) begin
454 ctrl_update = 1'b1;
455 // Update the address, unless there was an error
456 addr_update = ~lsu_err_q;
457 ls_fsm_ns = IDLE;
458 handle_misaligned_d = 1'b0;
459 end
460 end
461
462 WAIT_RVALID_MIS_GNTS_DONE: begin
463 // tell ID/EX stage to update the address (to make sure the
464 // second address can be captured correctly for mtval and PMP checking)
465 addr_incr_req_o = 1'b1;
466 // Wait for the first rvalid, second request is already granted
467 if (data_rvalid_i) begin
468 // Update the pmp error for the second part
469 pmp_err_d = data_pmp_err_i;
470 // The first part cannot see a PMP error in this state
471 lsu_err_d = data_bus_err_i;
472 // Now we can update the address for the second part if no error
473 addr_update = ~data_bus_err_i;
474 // Capture the first rdata for loads
475 rdata_update = ~data_we_q;
476 // Wait for second rvalid
477 ls_fsm_ns = IDLE;
478 end
479 end
480
481 default: begin
482 ls_fsm_ns = IDLE;
483 end
484 endcase
485 end
486
487 assign lsu_req_done_o = (lsu_req_i | (ls_fsm_cs != IDLE)) & (ls_fsm_ns == IDLE);
488
489 // registers for FSM
490 always_ff @(posedge clk_i or negedge rst_ni) begin
491 if (!rst_ni) begin
492 ls_fsm_cs <= IDLE;
493 handle_misaligned_q <= '0;
494 pmp_err_q <= '0;
495 lsu_err_q <= '0;
496 end else begin
497 ls_fsm_cs <= ls_fsm_ns;
498 handle_misaligned_q <= handle_misaligned_d;
499 pmp_err_q <= pmp_err_d;
500 lsu_err_q <= lsu_err_d;
501 end
502 end
503
504 /////////////
505 // Outputs //
506 /////////////
507
508 assign data_or_pmp_err = lsu_err_q | data_bus_err_i | pmp_err_q;
509 assign lsu_resp_valid_o = (data_rvalid_i | pmp_err_q) & (ls_fsm_cs == IDLE);
510 assign lsu_rdata_valid_o =
511 (ls_fsm_cs == IDLE) & data_rvalid_i & ~data_or_pmp_err & ~data_we_q & ~data_intg_err;
512
513 // output to register file
514 assign lsu_rdata_o = data_rdata_ext;
515
516 // output data address must be word aligned
517 assign data_addr_w_aligned = {data_addr[31:2], 2'b00};
518
519 // output to data interface
520 assign data_addr_o = data_addr_w_aligned;
521 assign data_we_o = lsu_we_i;
522 assign data_be_o = data_be;
523
524 /////////////////////////////////////
525 // Write data integrity generation //
526 /////////////////////////////////////
527
528 // SEC_CM: BUS.INTEGRITY
529 if (MemECC) begin : g_mem_wdata_ecc
530 prim_secded_inv_39_32_enc u_data_gen (
531 .data_i (data_wdata),
532 .data_o (data_wdata_o)
533 );
534 end else begin : g_no_mem_wdata_ecc
535 assign data_wdata_o = data_wdata;
536 end
537
538 // output to ID stage: mtval + AGU for misaligned transactions
539 assign addr_last_o = addr_last_q;
540
541 // Signal a load or store error depending on the transaction type outstanding
542 assign load_err_o = data_or_pmp_err & ~data_we_q & lsu_resp_valid_o;
543 assign store_err_o = data_or_pmp_err & data_we_q & lsu_resp_valid_o;
544 // Integrity errors are their own category for timing reasons. load_err_o is factored directly
545 // into data_req_o to enable synchronous exception on load errors without performance loss (An
546 // upcoming load cannot request until the current load has seen its response, so the earliest
547 // point the new request can be sent is the same cycle the response is seen). If load_err_o isn't
548 // factored into data_req_o there would have to be a stall cycle between all back to back loads.
549 // The data_intg_err signal is generated combinatorially from the incoming data_rdata_i. Were it
550 // to be factored into load_err_o there would be a feedthrough path from data_rdata_i to
551 // data_req_o which is undesirable.
552 assign load_resp_intg_err_o = data_intg_err & data_rvalid_i & ~data_we_q;
553 assign store_resp_intg_err_o = data_intg_err & data_rvalid_i & data_we_q;
554
555 assign busy_o = (ls_fsm_cs != IDLE);
556
557 //////////
558 // FCOV //
559 //////////
560 `ifndef DV_FCOV_DISABLE
561 // Set when awaiting the response for the second half of a misaligned access
562 logic fcov_mis_2_en_d, fcov_mis_2_en_q;
563
564 // fcov_mis_rvalid_1: Set when the response is received to the first half of a misaligned access,
565 // fcov_mis_rvalid_2: Set when response is received for the second half
566 logic fcov_mis_rvalid_1, fcov_mis_rvalid_2;
567
568 // Set when the first half of a misaligned access saw a bus errror
569 logic fcov_mis_bus_err_1_d, fcov_mis_bus_err_1_q;
570
571 assign fcov_mis_rvalid_1 = ls_fsm_cs inside {WAIT_RVALID_MIS, WAIT_RVALID_MIS_GNTS_DONE} &&
572 data_rvalid_i;
573
574 assign fcov_mis_rvalid_2 = ls_fsm_cs inside {IDLE} && fcov_mis_2_en_q && data_rvalid_i;
575
576 assign fcov_mis_2_en_d = fcov_mis_rvalid_2 ? 1'b0 : // clr
577 fcov_mis_rvalid_1 ? 1'b1 : // set
578 fcov_mis_2_en_q ;
579
580 assign fcov_mis_bus_err_1_d = fcov_mis_rvalid_2 ? 1'b0 : // clr
581 fcov_mis_rvalid_1 && data_bus_err_i ? 1'b1 : // set
582 fcov_mis_bus_err_1_q ;
583
584 always_ff @(posedge clk_i or negedge rst_ni) begin
585 if (!rst_ni) begin
586 fcov_mis_2_en_q <= 1'b0;
587 fcov_mis_bus_err_1_q <= 1'b0;
588 end else begin
589 fcov_mis_2_en_q <= fcov_mis_2_en_d;
590 fcov_mis_bus_err_1_q <= fcov_mis_bus_err_1_d;
591 end
592 end
593 `endif

595 `DV_FCOV_SIGNAL(logic, ls_error_exception, (load_err_o| store_err_o) & ~pmp_err_q)
596 `DV_FCOV_SIGNAL(logic, ls_pmp_exception, (load_err_o| store_err_o) & pmp_err_q)
597 `DV_FCOV_SIGNAL(logic, ls_first_req, lsu_req_i & (ls_fsm_cs== IDLE))
598 `DV_FCOV_SIGNAL(logic, ls_second_req,
599 (ls_fsm_cs inside {WAIT_RVALID_MIS}) & data_req_o & addr_incr_req_o)
