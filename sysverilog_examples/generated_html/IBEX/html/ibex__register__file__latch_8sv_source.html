6/**
7 * RISC-V register file
8 *
9 * Register file with 31 or 15x 32 bit wide registers. Register 0 is fixed to 0.
10 * This register file is based on latches and is thus smaller than the flip-flop
11 * based RF. It requires a target technology-specific clock gating cell. Use this
12 * register file when targeting ASIC synthesis or event-based simulators.
13 */
14module ibex_register_file_latch #(
15 parameter bit RV32E = 0,
16 parameter int unsigned DataWidth = 32,
17 parameter bit DummyInstructions = 0,
18 parameter bit WrenCheck = 0,
19 parameter bit RdataMuxCheck = 0,
20 parameter logic [DataWidth-1:0] WordZeroVal = '0
21) (
22 // Clock and Reset
23 input logic clk_i,
24 input logic rst_ni,
25
26 input logic test_en_i,
27 input logic dummy_instr_id_i,
28 input logic dummy_instr_wb_i,
29
30 //Read port R1
31 input logic [4:0] raddr_a_i,
32 output logic [DataWidth-1:0] rdata_a_o,
33
34 //Read port R2
35 input logic [4:0] raddr_b_i,
36 output logic [DataWidth-1:0] rdata_b_o,
37
38 // Write port W1
39 input logic [4:0] waddr_a_i,
40 input logic [DataWidth-1:0] wdata_a_i,
41 input logic we_a_i,
42
43 // This indicates whether spurious WE or non-one-hot encoded raddr are detected.
44 output logic err_o
45);
46
47 localparam int unsigned ADDR_WIDTH = RV32E ? 4 : 5;
48 localparam int unsigned NUM_WORDS = 2**ADDR_WIDTH;
49
50 logic [DataWidth-1:0] mem[NUM_WORDS];
51
52 logic [NUM_WORDS-1:0] waddr_onehot_a;
53
54 logic oh_raddr_a_err, oh_raddr_b_err, oh_we_err;
55
56 logic [NUM_WORDS-1:1] mem_clocks;
57 logic [DataWidth-1:0] wdata_a_q;
58
59 // internal addresses
60 logic [ADDR_WIDTH-1:0] raddr_a_int, raddr_b_int, waddr_a_int;
61
62 assign raddr_a_int = raddr_a_i[ADDR_WIDTH-1:0];
63 assign raddr_b_int = raddr_b_i[ADDR_WIDTH-1:0];
64 assign waddr_a_int = waddr_a_i[ADDR_WIDTH-1:0];
65
66 logic clk_int;
67
68 assign err_o = oh_raddr_a_err || oh_raddr_b_err || oh_we_err;
69
70 //////////
71 // READ //
72 //////////
73 if (RdataMuxCheck) begin : gen_rdata_mux_check
74 // Encode raddr_a/b into one-hot encoded signals.
75 logic [NUM_WORDS-1:0] raddr_onehot_a, raddr_onehot_b;
76 logic [NUM_WORDS-1:0] raddr_onehot_a_buf, raddr_onehot_b_buf;
77 prim_onehot_enc #(
78 .OneHotWidth(NUM_WORDS)
79 ) u_prim_onehot_enc_raddr_a (
80 .in_i (raddr_a_int),
81 .en_i (1'b1),
82 .out_o (raddr_onehot_a)
83 );
84
85 prim_onehot_enc #(
86 .OneHotWidth(NUM_WORDS)
87 ) u_prim_onehot_enc_raddr_b (
88 .in_i (raddr_b_int),
89 .en_i (1'b1),
90 .out_o (raddr_onehot_b)
91 );
92
93 // Buffer the one-hot encoded signals so that the checkers
94 // are not optimized.
95 prim_buf #(
96 .Width(NUM_WORDS)
97 ) u_prim_buf_raddr_a (
98 .in_i(raddr_onehot_a),
99 .out_o(raddr_onehot_a_buf)
100 );
101
102 prim_buf #(
103 .Width(NUM_WORDS)
104 ) u_prim_buf_raddr_b (
105 .in_i(raddr_onehot_b),
106 .out_o(raddr_onehot_b_buf)
107 );
108
109 // SEC_CM: DATA_REG_SW.GLITCH_DETECT
110 // Check the one-hot encoded signals for glitches.
111 prim_onehot_check #(
112 .AddrWidth(ADDR_WIDTH),
113 .OneHotWidth(NUM_WORDS),
114 .AddrCheck(1),
115 // When AddrCheck=1 also EnableCheck needs to be 1.
116 .EnableCheck(1)
117 ) u_prim_onehot_check_raddr_a (
118 .clk_i,
119 .rst_ni,
120 .oh_i (raddr_onehot_a_buf),
121 .addr_i (raddr_a_int),
122 // Set enable=1 as address is always valid.
123 .en_i (1'b1),
124 .err_o (oh_raddr_a_err)
125 );
126
127 prim_onehot_check #(
128 .AddrWidth(ADDR_WIDTH),
129 .OneHotWidth(NUM_WORDS),
130 .AddrCheck(1),
131 // When AddrCheck=1 also EnableCheck needs to be 1.
132 .EnableCheck(1)
133 ) u_prim_onehot_check_raddr_b (
134 .clk_i,
135 .rst_ni,
136 .oh_i (raddr_onehot_b_buf),
137 .addr_i (raddr_b_int),
138 // Set enable=1 as address is always valid.
139 .en_i (1'b1),
140 .err_o (oh_raddr_b_err)
141 );
142
143 // MUX register to rdata_a/b_o according to raddr_a/b_onehot.
144 prim_onehot_mux #(
145 .Width(DataWidth),
146 .Inputs(NUM_WORDS)
147 ) u_rdata_a_mux (
148 .clk_i,
149 .rst_ni,
150 .in_i (mem),
151 .sel_i (raddr_onehot_a),
152 .out_o (rdata_a_o)
153 );
154
155 prim_onehot_mux #(
156 .Width(DataWidth),
157 .Inputs(NUM_WORDS)
158 ) u_rdata_b_mux (
159 .clk_i,
160 .rst_ni,
161 .in_i (mem),
162 .sel_i (raddr_onehot_b),
163 .out_o (rdata_b_o)
164 );
165 end else begin : gen_no_rdata_mux_check
166 assign rdata_a_o = mem[raddr_a_int];
167 assign rdata_b_o = mem[raddr_b_int];
168 assign oh_raddr_a_err = 1'b0;
169 assign oh_raddr_b_err = 1'b0;
170 end
171
172 ///////////
173 // WRITE //
174 ///////////
175 // Global clock gating
176 prim_clock_gating cg_we_global (
177 .clk_i ( clk_i ),
178 .en_i ( we_a_i ),
179 .test_en_i ( test_en_i ),
180 .clk_o ( clk_int )
181 );
182
183 // Sample input data
184 // Use clk_int here, since otherwise we don't want to write anything anyway.
185 always_ff @(posedge clk_int or negedge rst_ni) begin : sample_wdata
186 if (!rst_ni) begin
187 wdata_a_q <= WordZeroVal;
188 end else begin
189 if (we_a_i) begin
190 wdata_a_q <= wdata_a_i;
191 end
192 end
193 end
194
195 // Write address decoding
196 always_comb begin : wad
197 for (int i = 0; i < NUM_WORDS; i++) begin : wad_word_iter
198 if (we_a_i && (waddr_a_int == 5'(i))) begin
199 waddr_onehot_a[i] = 1'b1;
200 end else begin
201 waddr_onehot_a[i] = 1'b0;
202 end
203 end
204 end
205
206 // SEC_CM: DATA_REG_SW.GLITCH_DETECT
207 // This checks for spurious WE strobes on the regfile.
208 if (WrenCheck) begin : gen_wren_check
209 // Buffer the decoded write enable bits so that the checker
210 // is not optimized into the address decoding logic.
211 logic [NUM_WORDS-1:0] waddr_onehot_a_buf;
212 prim_buf #(
213 .Width(NUM_WORDS)
214 ) u_prim_buf (
215 .in_i(waddr_onehot_a),
216 .out_o(waddr_onehot_a_buf)
217 );
218
219 prim_onehot_check #(
220 .AddrWidth(ADDR_WIDTH),
221 .AddrCheck(1),
222 .EnableCheck(1)
223 ) u_prim_onehot_check (
224 .clk_i,
225 .rst_ni,
226 .oh_i(waddr_onehot_a_buf),
227 .addr_i(waddr_a_i),
228 .en_i(we_a_i),
229 .err_o(oh_we_err)
230 );
231 end else begin : gen_no_wren_check
232 logic unused_strobe;
233 assign unused_strobe = waddr_onehot_a[0]; // this is never read from in this case
234 assign oh_we_err = 1'b0;
235 end
236
237 // Individual clock gating (if integrated clock-gating cells are available)
238 for (genvar x = 1; x < NUM_WORDS; x++) begin : gen_cg_word_iter
239 prim_clock_gating cg_i (
240 .clk_i ( clk_int ),
241 .en_i ( waddr_onehot_a[x] ),
242 .test_en_i ( test_en_i ),
243 .clk_o ( mem_clocks[x] )
244 );
245 end
246
247 // Actual write operation:
248 // Generate the sequential process for the NUM_WORDS words of the memory.
249 // The process is synchronized with the clocks mem_clocks[i], i = 1, ..., NUM_WORDS-1.
250 for (genvar i = 1; i < NUM_WORDS; i++) begin : g_rf_latches
251 always_latch begin
252 if (mem_clocks[i]) begin
253 mem[i] = wdata_a_q;
254 end
255 end
256 end
257
258 // With dummy instructions enabled, R0 behaves as a real register but will always return 0 for
259 // real instructions.
260 if (DummyInstructions) begin : g_dummy_r0
261 // SEC_CM: CTRL_FLOW.UNPREDICTABLE
262 logic we_r0_dummy;
263 logic r0_clock;
264 logic [DataWidth-1:0] mem_r0;
265
266 // Write enable for dummy R0 register (waddr_a_i will always be 0 for dummy instructions)
267 assign we_r0_dummy = we_a_i & dummy_instr_wb_i;
268
269 // R0 clock gate
270 prim_clock_gating cg_i (
271 .clk_i ( clk_int ),
272 .en_i ( we_r0_dummy ),
273 .test_en_i ( test_en_i ),
274 .clk_o ( r0_clock )
275 );
276
277 always_latch begin : latch_wdata
278 if (r0_clock) begin
279 mem_r0 = wdata_a_q;
280 end
281 end
282
283 // Output the dummy data for dummy instructions, otherwise R0 reads as zero
284 assign mem[0] = dummy_instr_id_i ? mem_r0 : WordZeroVal;
285
286 end else begin : g_normal_r0
287 logic unused_dummy_instr;
288 assign unused_dummy_instr = dummy_instr_id_i ^ dummy_instr_wb_i;
289
290 assign mem[0] = WordZeroVal;
291 end

300endmodule
