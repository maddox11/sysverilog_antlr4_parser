6/**
7 * Trace executed instructions in simulation
8 *
9 * This tracer takes execution information from the RISC-V Verification Interface (RVFI) and
10 * produces a text file with a human-readable trace.
11 *
12 * All traced instructions are written to a log file. By default, the log file is named
13 * trace_core_<HARTID>.log, with <HARTID> being the 8 digit hart ID of the core being traced.
14 *
15 * The file name base, defaulting to "trace_core" can be set using the "ibex_tracer_file_base"
16 * plusarg passed to the simulation, e.g. "+ibex_tracer_file_base=ibex_my_trace". The exact syntax
17 * of passing plusargs to a simulation depends on the simulator.
18 *
19 * The creation of the instruction trace is enabled by default but can be disabled for a simulation.
20 * This behaviour is controlled by the plusarg "ibex_tracer_enable". Use "ibex_tracer_enable=0" to
21 * disable the tracer.
22 *
23 * The trace contains six columns, separated by tabs:
24 * - The simulation time
25 * - The clock cycle count since reset
26 * - The program counter (PC)
27 * - The instruction
28 * - The decoded instruction in the same format as objdump, together with the accessed registers and
29 * read/written memory values. Jumps and branches show the target address.
30 * This column may be omitted if the instruction does not decode into a long form.
31 * - Accessed registers and memory locations.
32 *
33 * Significant effort is spent to make the decoding produced by this tracer as similar as possible
34 * to the one produced by objdump. This simplifies the correlation between the static program
35 * information from the objdump-generated disassembly, and the runtime information from this tracer.
36 */

110 function automatic void printbuffer_dumpline(int fh);
111 string rvfi_insn_str;
112
113 // Write compressed instructions as four hex digits (16 bit word), and
114 // uncompressed ones as 8 hex digits (32 bit words).
115 if (insn_is_compressed) begin
116 rvfi_insn_str = $sformatf("%h", rvfi_insn[15:0]);
117 end else begin
118 rvfi_insn_str = $sformatf("%h", rvfi_insn);
119 end
120
121 $fwrite(fh, "%15t\t%d\t%h\t%s\t%s\t",
122 $time, cycle, rvfi_pc_rdata, rvfi_insn_str, decoded_str);
123
124 if ((data_accessed & RS1) != 0) begin
125 $fwrite(fh, " %s:0x%08x", reg_addr_to_str(rvfi_rs1_addr), rvfi_rs1_rdata);
126 end
127 if ((data_accessed & RS2) != 0) begin
128 $fwrite(fh, " %s:0x%08x", reg_addr_to_str(rvfi_rs2_addr), rvfi_rs2_rdata);
129 end
130 if ((data_accessed & RS3) != 0) begin
131 $fwrite(fh, " %s:0x%08x", reg_addr_to_str(rvfi_rs3_addr), rvfi_rs3_rdata);
132 end
133 if ((data_accessed & RD) != 0) begin
134 $fwrite(fh, " %s=0x%08x", reg_addr_to_str(rvfi_rd_addr), rvfi_rd_wdata);
135 end
136 if ((data_accessed & MEM) != 0) begin
137 $fwrite(fh, " PA:0x%08x", rvfi_mem_addr);
138
139 if (rvfi_mem_wmask != 4'b0000) begin
140 $fwrite(fh, " store:0x%08x", rvfi_mem_wdata);
141 end
142 if (rvfi_mem_rmask != 4'b0000) begin
143 $fwrite(fh, " load:0x%08x", rvfi_mem_rdata);
144 end
145 end
146
147 $fwrite(fh, "\n");
148 endfunction

487 function automatic void decode_csr_insn(input string mnemonic);
488 logic [11:0] csr;
489 string csr_name;
490 csr = rvfi_insn[31:20];
491 csr_name = get_csr_name(csr);

493 data_accessed = RD;
494
495 if (!rvfi_insn[14]) begin
496 data_accessed |= RS1;
497 decoded_str = $sformatf("%s\tx%0d,%s,x%0d",
498 mnemonic, rvfi_rd_addr, csr_name, rvfi_rs1_addr);
499 end else begin
500 decoded_str = $sformatf("%s\tx%0d,%s,%0d",
501 mnemonic, rvfi_rd_addr, csr_name, {27'b0, rvfi_insn[19:15]});
502 end

505 function automatic void decode_cr_insn(input string mnemonic);
506 if (rvfi_rs2_addr == 5'b0) begin
507 if (rvfi_insn[12] == 1'b1) begin
508 // C.JALR
509 data_accessed = RS1 | RD;
510 end else begin
511 // C.JR
512 data_accessed = RS1;
513 end
514 decoded_str = $sformatf("%s\tx%0d", mnemonic, rvfi_rs1_addr);
515 end else begin
516 data_accessed = RS1 | RS2 | RD; // RS1 == RD
517 decoded_str = $sformatf("%s\tx%0d,x%0d", mnemonic, rvfi_rd_addr, rvfi_rs2_addr);
518 end

571 function automatic void decode_cb_insn(input string mnemonic);
572 logic [7:0] imm;
573 logic [31:0] jump_target;
574 if (rvfi_insn[15:13] == 3'b110 || rvfi_insn[15:13] == 3'b111) begin
575 // C.BNEZ and C.BEQZ
576 // We cannot use rvfi_pc_wdata for conditional jumps.
577 imm = {rvfi_insn[12], rvfi_insn[6:5], rvfi_insn[2], rvfi_insn[11:10], rvfi_insn[4:3]};
578 jump_target = rvfi_pc_rdata + 32'($signed({imm, 1'b0}));
579 data_accessed = RS1;
580 decoded_str = $sformatf("%s\tx%0d,%0x", mnemonic, rvfi_rs1_addr, jump_target);
581 end else if (rvfi_insn[15:13] == 3'b100) begin
582 // C.ANDI
583 imm = {{2{rvfi_insn[12]}}, rvfi_insn[12], rvfi_insn[6:2]};
584 data_accessed = RS1 | RD; // RS1 == RD
585 decoded_str = $sformatf("%s\tx%0d,%0d", mnemonic, rvfi_rd_addr, $signed(imm));
586 end else begin
587 imm = {rvfi_insn[12], rvfi_insn[6:2], 2'b00};
588 data_accessed = RS1;
589 decoded_str = $sformatf("%s\tx%0d,0x%0x", mnemonic, rvfi_rs1_addr, imm);
590 end
591 endfunction

606 function automatic void decode_compressed_load_insn(input string mnemonic);

633 function automatic void decode_load_insn();
634 string mnemonic;
635
636 /*
637 Gives wrong results in Verilator < 4.020.
638 See https://github.com/lowRISC/ibex/issues/372 and
639 https://www.veripool.org/issues/1536-Verilator-Misoptimization-in-if-and-case-with-default-statement-inside-a-function
640
641 unique case (rvfi_insn[14:12])
642 3'b000: mnemonic = "lb";
643 3'b001: mnemonic = "lh";
644 3'b010: mnemonic = "lw";
645 3'b100: mnemonic = "lbu";
646 3'b101: mnemonic = "lhu";
647 default: begin
648 decode_mnemonic("INVALID");
649 return;
650 end
651 endcase
652 */
653 logic [2:0] size;
654 size = rvfi_insn[14:12];
655 if (size == 3'b000) begin
656 mnemonic = "lb";
657 end else if (size == 3'b001) begin
658 mnemonic = "lh";
659 end else if (size == 3'b010) begin
660 mnemonic = "lw";
661 end else if (size == 3'b100) begin
662 mnemonic = "lbu";
663 end else if (size == 3'b101) begin
664 mnemonic = "lhu";
665 end else begin
666 decode_mnemonic("INVALID");
667 return;
668 end
669
670
671 data_accessed = RD | RS1 | MEM;
672 decoded_str = $sformatf("%s\tx%0d,%0d(x%0d)", mnemonic, rvfi_rd_addr,
673 $signed({{20 {rvfi_insn[31]}}, rvfi_insn[31:20]}), rvfi_rs1_addr);
674 endfunction

676 function automatic void decode_store_insn();
677 string mnemonic;
678
679 unique case (rvfi_insn[13:12])
680 2'b00: mnemonic = "sb";
681 2'b01: mnemonic = "sh";
682 2'b10: mnemonic = "sw";
683 default: begin
684 decode_mnemonic("INVALID");
685 return;
686 end
687 endcase
688
689 if (!rvfi_insn[14]) begin
690 // regular store
691 data_accessed = RS1 | RS2 | MEM;
692 decoded_str = $sformatf("%s\tx%0d,%0d(x%0d)",
693 mnemonic,
694 rvfi_rs2_addr,
695 $signed({{20{rvfi_insn[31]}}, rvfi_insn[31:25], rvfi_insn[11:7]}),
696 rvfi_rs1_addr);
697 end else begin
698 decode_mnemonic("INVALID");
699 end
700 endfunction

702 function automatic string get_fence_description(logic [3:0] bits);
703 string desc = "";
704 if (bits[3]) begin
705 desc = {desc, "i"};
706 end
707 if (bits[2]) begin
708 desc = {desc, "o"};
709 end
710 if (bits[1]) begin
711 desc = {desc, "r"};
712 end
713 if (bits[0]) begin
714 desc = {desc, "w"};
715 end
716 return desc;
717 endfunction
