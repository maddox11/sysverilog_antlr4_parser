188 assign load_err_d = load_err_i;
189 assign store_err_d = store_err_i;
190
191 // Decoder doesn't take instr_valid into account, factor it in here.
192 assign ecall_insn = ecall_insn_i & instr_valid_i;
193 assign mret_insn = mret_insn_i & instr_valid_i;
194 assign dret_insn = dret_insn_i & instr_valid_i;
195 assign wfi_insn = wfi_insn_i & instr_valid_i;
196 assign ebrk_insn = ebrk_insn_i & instr_valid_i;
197 assign csr_pipe_flush = csr_pipe_flush_i & instr_valid_i;
198 assign instr_fetch_err = instr_fetch_err_i & instr_valid_i;
199
200 // This is recorded in the illegal_insn_q flop to help timing. Specifically
201 // it is needed to break the path from ibex_cs_registers/illegal_csr_insn_o
202 // to pc_set_o. Clear when controller is in FLUSH so it won't remain set
203 // once illegal instruction is handled.
204 // illegal_insn_i only set when instr_valid_i is set.
205 assign illegal_insn_d = illegal_insn_i & (ctrl_fsm_cs != FLUSH);
206
207 `ASSERT(IllegalInsnOnlyIfInsnValid, illegal_insn_i |-> instr_valid_i)
208
209 // exception requests
210 // requests are flopped in exc_req_q. This is cleared when controller is in
211 // the FLUSH state so the cycle following exc_req_q won't remain set for an
212 // exception request that has just been handled.
213 // All terms in this expression are qualified by instr_valid_i
214 assign exc_req_d = (ecall_insn | ebrk_insn | illegal_insn_d | instr_fetch_err) &
215 (ctrl_fsm_cs != FLUSH);
216
217 // LSU exception requests
218 assign exc_req_lsu = store_err_i | load_err_i;
219
220 assign id_exception_o = exc_req_d & ~wb_exception_o;
221
222 // special requests: special instructions, pipeline flushes, exceptions...
223 // All terms in these expressions are qualified by instr_valid_i except exc_req_lsu which can come
224 // from the Writeback stage with no instr_valid_i from the ID stage
225
226 // These special requests only cause a pipeline flush and in particular don't cause a PC change
227 // that is outside the normal execution flow
228 assign special_req_flush_only = wfi_insn | csr_pipe_flush;
229
230 // These special requests cause a change in PC
231 assign special_req_pc_change = mret_insn | dret_insn | exc_req_d | exc_req_lsu;
232
233 // generic special request signal, applies to all instructions
234 assign special_req = special_req_pc_change | special_req_flush_only;
235
236 // Is there an instruction in ID or WB that has yet to complete?
237 assign id_wb_pending = instr_valid_i | ~ready_wb_i;

239 // Logic to determine which exception takes priority where multiple are possible.
240 if (WritebackStage) begin : g_wb_exceptions
241 always_comb begin
242 instr_fetch_err_prio = 0;
243 illegal_insn_prio = 0;
244 ecall_insn_prio = 0;
245 ebrk_insn_prio = 0;
246 store_err_prio = 0;
247 load_err_prio = 0;
248
249 // Note that with the writeback stage store/load errors occur on the instruction in writeback,
250 // all other exception/faults occur on the instruction in ID/EX. The faults from writeback
251 // must take priority as that instruction is architecurally ordered before the one in ID/EX.
252 if (store_err_q) begin
253 store_err_prio = 1'b1;
254 end else if (load_err_q) begin
255 load_err_prio = 1'b1;
256 end else if (instr_fetch_err) begin
257 instr_fetch_err_prio = 1'b1;
258 end else if (illegal_insn_q) begin
259 illegal_insn_prio = 1'b1;
260 end else if (ecall_insn) begin
261 ecall_insn_prio = 1'b1;
262 end else if (ebrk_insn) begin
263 ebrk_insn_prio = 1'b1;
264 end
265 end
266
267 // Instruction in writeback is generating an exception so instruction in ID must not execute
268 assign wb_exception_o = load_err_q | store_err_q | load_err_i | store_err_i;
269 end else begin : g_no_wb_exceptions
270 always_comb begin
271 instr_fetch_err_prio = 0;
272 illegal_insn_prio = 0;
273 ecall_insn_prio = 0;
274 ebrk_insn_prio = 0;
275 store_err_prio = 0;
276 load_err_prio = 0;
277
278 if (instr_fetch_err) begin
279 instr_fetch_err_prio = 1'b1;
280 end else if (illegal_insn_q) begin
281 illegal_insn_prio = 1'b1;
282 end else if (ecall_insn) begin
283 ecall_insn_prio = 1'b1;
284 end else if (ebrk_insn) begin
285 ebrk_insn_prio = 1'b1;
286 end else if (store_err_q) begin
287 store_err_prio = 1'b1;
288 end else if (load_err_q) begin
289 load_err_prio = 1'b1;
290 end
291 end
292 assign wb_exception_o = 1'b0;
293 end

295 `ASSERT_IF(IbexExceptionPrioOnehot,
296 $onehot({instr_fetch_err_prio,
297 illegal_insn_prio,
298 ecall_insn_prio,
299 ebrk_insn_prio,
300 store_err_prio,
301 load_err_prio}),
302 (ctrl_fsm_cs== FLUSH) & exc_req_q)

308 // Internal interrupt control
309 // All internal interrupts act as an NMI and go to the NMI vector. mcause is set based upon
310 // irq_nm_int_cause.
311
312 if (MemECC) begin : g_intg_irq_int
313 logic mem_resp_intg_err_irq_pending_q, mem_resp_intg_err_irq_pending_d;
314 logic [31:0] mem_resp_intg_err_addr_q, mem_resp_intg_err_addr_d;
315 logic mem_resp_intg_err_irq_set, mem_resp_intg_err_irq_clear;
316 logic entering_nmi;
317
318 assign entering_nmi = nmi_mode_d & ~nmi_mode_q;
319
320 // Load integerity error internal interrupt
321 always_comb begin
322 mem_resp_intg_err_addr_d = mem_resp_intg_err_addr_q;
323 mem_resp_intg_err_irq_set = 1'b0;
324 mem_resp_intg_err_irq_clear = 1'b0;
325
326 if (mem_resp_intg_err_irq_pending_q) begin
327 // Clear ECC error interrupt when it is handled. External NMI takes a higher priority so
328 // don't clear the ECC error interrupt if an external NMI is present.
329 if (entering_nmi & !irq_nm_ext_i) begin
330 mem_resp_intg_err_irq_clear = 1'b1;
331 end
332 end else if (mem_resp_intg_err_i) begin
333 // When an ECC error is seen set the ECC error interrupt and capture the address that saw
334 // the error. If there is already an ecc error IRQ pending ignore any ECC errors coming in.
335 mem_resp_intg_err_addr_d = lsu_addr_last_i;
336 mem_resp_intg_err_irq_set = 1'b1;
337 end
338 end
339
340 assign mem_resp_intg_err_irq_pending_d =
341 (mem_resp_intg_err_irq_pending_q & ~mem_resp_intg_err_irq_clear) | mem_resp_intg_err_irq_set;
342
343 always_ff @(posedge clk_i or negedge rst_ni) begin
344 if (!rst_ni) begin
345 mem_resp_intg_err_irq_pending_q <= 1'b0;
346 mem_resp_intg_err_addr_q <= '0;
347 end else begin
348 mem_resp_intg_err_irq_pending_q <= mem_resp_intg_err_irq_pending_d;
349 mem_resp_intg_err_addr_q <= mem_resp_intg_err_addr_d;
350 end
351 end
352
353 // As integrity error is the only internal interrupt implement, set irq_nm_* signals directly
354 // within this generate block.
355 assign irq_nm_int = mem_resp_intg_err_irq_pending_q;
356 assign irq_nm_int_cause = NMI_INT_CAUSE_ECC;
357 assign irq_nm_int_mtval = mem_resp_intg_err_addr_q;
358 end else begin : g_no_intg_irq_int
359 logic unused_mem_resp_intg_err_i;
360
361 assign unused_mem_resp_intg_err_i = mem_resp_intg_err_i;
362
363 // No integrity checking on incoming load data so no internal interrupts
364 assign irq_nm_int = 1'b0;
365 assign irq_nm_int_cause = nmi_int_cause_e'(0);
366 assign irq_nm_int_mtval = '0;
367 end

369
370 // Enter debug mode due to an external debug_req_i or because the core is in
371 // single step mode (dcsr.step == 1). Single step must be qualified with
372 // instruction valid otherwise the core will immediately enter debug mode
373 // due to a recently flushed IF (or a delay in an instruction returning from
374 // memory) before it has had anything to single step.
375 // Also enter debug mode on a trigger match (hardware breakpoint)
376
377 // Set `do_single_step_q` when a valid instruction is seen outside of debug mode and core is in
378 // single step mode. The first valid instruction on debug mode entry will clear it. Hold its value
379 // when there is no valid instruction so `do_single_step_d` remains asserted until debug mode is
380 // entered.
381 assign do_single_step_d = instr_valid_i ? ~debug_mode_q & debug_single_step_i : do_single_step_q;
382 // Enter debug mode due to:
383 // * external `debug_req_i`
384 // * core in single step mode (dcsr.step == 1).
385 // * trigger match (hardware breakpoint)
386 //
387 // `debug_req_i` and `do_single_step_d` request debug mode with priority. This results in a debug
388 // mode entry even if the controller goes to `FLUSH` in preparation for handling an exception or
389 // interrupt. `trigger_match_i` is not a priority entry into debug mode as it must be ignored
390 // where control flow changes such that the instruction causing the trigger is no longer being
391 // executed.
392 assign enter_debug_mode_prio_d = (debug_req_i | do_single_step_d) & ~debug_mode_q;
393 assign enter_debug_mode = enter_debug_mode_prio_d | (trigger_match_i & ~debug_mode_q);
394
395 // Set when an ebreak should enter debug mode rather than jump to exception
396 // handler
397 assign ebreak_into_debug = priv_mode_i == PRIV_LVL_M ? debug_ebreakm_i :
398 priv_mode_i == PRIV_LVL_U ? debug_ebreaku_i :
399 1'b0;
400
401 // NMI can be produced from an external (irq_nm_i top level input) or an internal (within
402 // ibex_core) source. For internal sources the cause is specified via irq_nm_int_cause.
403 assign irq_nm = irq_nm_ext_i | irq_nm_int;
404
405 // MIE bit only applies when in M mode
406 assign irq_enabled = csr_mstatus_mie_i | (priv_mode_i == PRIV_LVL_U);
407
408 // Interrupts including NMI are ignored,
409 // - while in debug mode,
410 // - while in NMI mode (nested NMIs are not supported, NMI has highest priority and
411 // cannot be interrupted by regular interrupts),
412 // - while single stepping.
413 assign handle_irq = ~debug_mode_q & ~debug_single_step_i & ~nmi_mode_q &
414 (irq_nm | (irq_pending_i & irq_enabled));
415
416 // generate ID of fast interrupts, highest priority to lowest ID
417 always_comb begin : gen_mfip_id
418 mfip_id = 4'd0;
419
420 for (int i = 14; i >= 0; i--) begin
421 if (irqs_i.irq_fast[i]) begin
422 mfip_id = i[3:0];
423 end
424 end
425 end
426
427 assign unused_irq_timer = irqs_i.irq_timer;
428
429 // Record the debug cause outside of the FSM
430 // The decision to enter debug_mode and the write of the cause to DCSR happen
431 // in seperate steps within the FSM. Hence, there are a small number of cycles
432 // where a change in external stimulus can cause the cause to be recorded incorrectly.
433 assign debug_cause_d = trigger_match_i ? DBG_CAUSE_TRIGGER :
434 ebrk_insn_prio & ebreak_into_debug ? DBG_CAUSE_EBREAK :
435 debug_req_i ? DBG_CAUSE_HALTREQ :
436 do_single_step_d ? DBG_CAUSE_STEP :
437 DBG_CAUSE_NONE ;
438
439 always_ff @(posedge clk_i or negedge rst_ni) begin
440 if (!rst_ni) begin
441 debug_cause_q <= DBG_CAUSE_NONE;
442 end else begin
443 debug_cause_q <= debug_cause_d;
444 end
445 end
446
447 assign debug_cause_o = debug_cause_q;

449 /////////////////////
450 // Core controller //
451 /////////////////////
452
453 always_comb begin
454 // Default values
455 instr_req_o = 1'b1;
456
457 csr_save_if_o = 1'b0;
458 csr_save_id_o = 1'b0;
459 csr_save_wb_o = 1'b0;
460 csr_restore_mret_id_o = 1'b0;
461 csr_restore_dret_id_o = 1'b0;
462 csr_save_cause_o = 1'b0;
463 csr_mtval_o = '0;
464
465 // The values of pc_mux and exc_pc_mux are only relevant if pc_set is set. Some of the states
466 // below always set pc_mux and exc_pc_mux but only set pc_set if certain conditions are met.
467 // This avoid having to factor those conditions into the pc_mux and exc_pc_mux select signals
468 // helping timing.
469 pc_mux_o = PC_BOOT;
470 pc_set_o = 1'b0;
471 nt_branch_mispredict_o = 1'b0;
472
473 exc_pc_mux_o = EXC_PC_IRQ;
474 exc_cause_o = ExcCauseInsnAddrMisa; // = 6'h00
475
476 ctrl_fsm_ns = ctrl_fsm_cs;
477
478 ctrl_busy_o = 1'b1;
479
480 halt_if = 1'b0;
481 retain_id = 1'b0;
482 flush_id = 1'b0;
483
484 debug_csr_save_o = 1'b0;
485 debug_mode_d = debug_mode_q;
486 debug_mode_entering_o = 1'b0;
487 nmi_mode_d = nmi_mode_q;
488
489 perf_tbranch_o = 1'b0;
490 perf_jump_o = 1'b0;
491
492 controller_run_o = 1'b0;
493
494 unique case (ctrl_fsm_cs)
495 RESET: begin
496 instr_req_o = 1'b0;
497 pc_mux_o = PC_BOOT;
498 pc_set_o = 1'b1;
499 ctrl_fsm_ns = BOOT_SET;
500 end
501
502 BOOT_SET: begin
503 // copy boot address to instr fetch address
504 instr_req_o = 1'b1;
505 pc_mux_o = PC_BOOT;
506 pc_set_o = 1'b1;
507
508 ctrl_fsm_ns = FIRST_FETCH;
509 end
510
511 WAIT_SLEEP: begin
512 ctrl_busy_o = 1'b0;
513 instr_req_o = 1'b0;
514 halt_if = 1'b1;
515 flush_id = 1'b1;
516 ctrl_fsm_ns = SLEEP;
517 end
518
519 SLEEP: begin
520 // instruction in IF stage is already valid
521 // we begin execution when an interrupt has arrived
522 instr_req_o = 1'b0;
523 halt_if = 1'b1;
524 flush_id = 1'b1;
525
526 // normal execution flow
527 // in debug mode or single step mode we leave immediately (wfi=nop)
528 if (irq_nm || irq_pending_i || debug_req_i || debug_mode_q || debug_single_step_i) begin
529 ctrl_fsm_ns = FIRST_FETCH;
530 end else begin
531 // Make sure clock remains disabled.
532 ctrl_busy_o = 1'b0;
533 end
534 end
535
536 FIRST_FETCH: begin
537 // Stall because of IF miss
538 if (id_in_ready_o) begin
539 ctrl_fsm_ns = DECODE;
540 end
541
542 // handle interrupts
543 if (handle_irq) begin
544 // We are handling an interrupt. Set halt_if to tell IF not to give
545 // us any more instructions before it redirects to the handler, but
546 // don't set flush_id: we must allow this instruction to complete
547 // (since it might have outstanding loads or stores).
548 ctrl_fsm_ns = IRQ_TAKEN;
549 halt_if = 1'b1;
550 end
551
552 // enter debug mode
553 if (enter_debug_mode) begin
554 ctrl_fsm_ns = DBG_TAKEN_IF;
555 // Halt IF only for now, ID will be flushed in DBG_TAKEN_IF as the
556 // ID state is needed for correct debug mode entry
557 halt_if = 1'b1;
558 end
559 end
560
561 DECODE: begin
562 // normal operating mode of the ID stage, in case of debug and interrupt requests,
563 // priorities are as follows (lower number == higher priority)
564 // 1. currently running (multicycle) instructions and exceptions caused by these
565 // 2. debug requests
566 // 3. interrupt requests
567
568 controller_run_o = 1'b1;
569
570 // Set PC mux for branch and jump here to ease timing. Value is only relevant if pc_set_o is
571 // also set. Setting the mux value here avoids factoring in special_req and instr_valid_i
572 // which helps timing.
573 pc_mux_o = PC_JUMP;
574
575
576 // Get ready for special instructions, exceptions, pipeline flushes
577 if (special_req) begin
578 // Halt IF but don't flush ID. This leaves a valid instruction in
579 // ID so controller can determine appropriate action in the
580 // FLUSH state.
581 retain_id = 1'b1;
582
583 // Wait for the writeback stage to either be ready for a new instruction or raise its own
584 // exception before going to FLUSH. If the instruction in writeback raises an exception it
585 // must take priority over any exception from an instruction in ID/EX. Only once the
586 // writeback stage is ready can we be certain that won't happen. Without a writeback
587 // stage ready_wb_i == 1 so the FSM will always go directly to FLUSH.
588
589 if (ready_wb_i | wb_exception_o) begin
590 ctrl_fsm_ns = FLUSH;
591 end
592 end
593
594 if (branch_set_i || jump_set_i) begin
595 // Only set the PC if the branch predictor hasn't already done the branch for us
596 pc_set_o = BranchPredictor ? ~instr_bp_taken_i : 1'b1;

598 perf_tbranch_o = branch_set_i;
599 perf_jump_o = jump_set_i;
600 end

602 if (BranchPredictor) begin

605 // signal a mispredict.
606 nt_branch_mispredict_o = 1'b1;
607 end

610 // If entering debug mode or handling an IRQ the core needs to wait until any instruction in
611 // ID or WB has finished executing. Stall IF during that time.
612 if ((enter_debug_mode || handle_irq) && (stall || id_wb_pending)) begin

618 // enter debug mode
619 ctrl_fsm_ns = DBG_TAKEN_IF;
620 // Halt IF only for now, ID will be flushed in DBG_TAKEN_IF as the
621 // ID state is needed for correct debug mode entry
622 halt_if = 1'b1;
623 end else if (handle_irq) begin
624 // handle interrupt (not in debug mode)
625 ctrl_fsm_ns = IRQ_TAKEN;
626 // We are handling an interrupt (not in debug mode). Set halt_if to
627 // tell IF not to give us any more instructions before it redirects
628 // to the handler, but don't set flush_id: we must allow this
629 // instruction to complete (since it might have outstanding loads
630 // or stores).
631 halt_if = 1'b1;
632 end
633 end

637 IRQ_TAKEN: begin
638 pc_mux_o = PC_EXC;
639 exc_pc_mux_o = EXC_PC_IRQ;

641 if (handle_irq) begin
642 pc_set_o = 1'b1;
643
644 csr_save_if_o = 1'b1;
645 csr_save_cause_o = 1'b1;

647 // Prioritise interrupts as required by the architecture
648 if (irq_nm && !nmi_mode_q) begin
649 exc_cause_o =
650 irq_nm_ext_i ? ExcCauseIrqNm :
651 '{irq_ext: 1'b0, irq_int: 1'b1, lower_cause: irq_nm_int_cause};
652
653 if (irq_nm_int & !irq_nm_ext_i) begin
654 csr_mtval_o = irq_nm_int_mtval;
655 end
656
657 nmi_mode_d = 1'b1; // enter NMI mode
658 end else if (irqs_i.irq_fast != 15'b0) begin
659 // generate exception cause ID from fast interrupt ID:
660 // - first bit distinguishes interrupts from exceptions,
661 // - second bit adds 16 to fast interrupt ID
662 // for example ExcCauseIrqFast0 = {1'b1, 5'd16}
663 exc_cause_o = '{irq_ext: 1'b1, irq_int: 1'b0, lower_cause: {1'b1, mfip_id}};
664 end else if (irqs_i.irq_external) begin
665 exc_cause_o = ExcCauseIrqExternalM;
666 end else if (irqs_i.irq_software) begin
667 exc_cause_o = ExcCauseIrqSoftwareM;
668 end else begin // irqs_i.irq_timer
669 exc_cause_o = ExcCauseIrqTimerM;
670 end
671 end
672
673 ctrl_fsm_ns = DECODE;
674 end
675
676 DBG_TAKEN_IF: begin
677 pc_mux_o = PC_EXC;
678 exc_pc_mux_o = EXC_PC_DBD;
679
680 // enter debug mode and save PC in IF to dpc
681 // jump to debug exception handler in debug memory
682 flush_id = 1'b1;
683 pc_set_o = 1'b1;
684
685 csr_save_if_o = 1'b1;
686 debug_csr_save_o = 1'b1;
687
688 csr_save_cause_o = 1'b1;
689
690 // enter debug mode
691 debug_mode_d = 1'b1;
692 debug_mode_entering_o = 1'b1;
693
694 ctrl_fsm_ns = DECODE;
695 end
696
697 DBG_TAKEN_ID: begin
698 // enter debug mode and save PC in ID to dpc, used when encountering
699 // 1. EBREAK during debug mode
700 // 2. EBREAK with forced entry into debug mode (ebreakm or ebreaku set).
701 // regular ebreak's go through FLUSH.
702 //
703 // for 1. do not update dcsr and dpc, for 2. do so
704 // jump to debug exception handler in debug memory
705 flush_id = 1'b1;
706 pc_mux_o = PC_EXC;
707 pc_set_o = 1'b1;
708 exc_pc_mux_o = EXC_PC_DBD;
709
710 // update dcsr and dpc
711 if (ebreak_into_debug && !debug_mode_q) begin // ebreak with forced entry
712
713 // dpc (set to the address of the EBREAK, i.e. set to PC in ID stage)
714 csr_save_cause_o = 1'b1;
715 csr_save_id_o = 1'b1;
716
717 // dcsr
718 debug_csr_save_o = 1'b1;
719 end
720
721 // enter debug mode
722 debug_mode_d = 1'b1;
723 debug_mode_entering_o = 1'b1;
724
725 ctrl_fsm_ns = DECODE;
726 end
727
728 FLUSH: begin
729 // flush the pipeline
730 halt_if = 1'b1;
731 flush_id = 1'b1;
732 ctrl_fsm_ns = DECODE;
733
734 // As pc_mux and exc_pc_mux can take various values in this state they aren't set early
735 // here.
736
737 // exceptions: set exception PC, save PC and exception cause
738 // exc_req_lsu is high for one clock cycle only (in DECODE)
739 if (exc_req_q || store_err_q || load_err_q) begin
740 pc_set_o = 1'b1;
741 pc_mux_o = PC_EXC;
742 exc_pc_mux_o = debug_mode_q ? EXC_PC_DBG_EXC : EXC_PC_EXC;
743
744 if (WritebackStage) begin : g_writeback_mepc_save
745 // With the writeback stage present whether an instruction accessing memory will cause
746 // an exception is only known when it is in writeback. So when taking such an exception
747 // epc must come from writeback.
748 csr_save_id_o = ~(store_err_q | load_err_q);
749 csr_save_wb_o = store_err_q | load_err_q;
750 end else begin : g_no_writeback_mepc_save
751 csr_save_id_o = 1'b0;
752 end
753
754 csr_save_cause_o = 1'b1;
755
756 // Exception/fault prioritisation logic will have set exactly 1 X_prio signal
757 unique case (1'b1)
758 instr_fetch_err_prio: begin
759 exc_cause_o = ExcCauseInstrAccessFault;
760 csr_mtval_o = instr_fetch_err_plus2_i ? (pc_id_i + 32'd2) : pc_id_i;
761 end
762 illegal_insn_prio: begin
763 exc_cause_o = ExcCauseIllegalInsn;
764 csr_mtval_o = instr_is_compressed_i ? {16'b0, instr_compressed_i} : instr_i;
765 end
766 ecall_insn_prio: begin
767 exc_cause_o = (priv_mode_i == PRIV_LVL_M) ? ExcCauseEcallMMode :
768 ExcCauseEcallUMode;
769 end
770 ebrk_insn_prio: begin
771 if (debug_mode_q | ebreak_into_debug) begin
772 // EBREAK enters debug mode when dcsr.ebreakm or dcsr.ebreaku is set and we're in
773 // M or U mode respectively. If we're already in debug mode we re-enter debug mode.
774
775 pc_set_o = 1'b0;
776 csr_save_id_o = 1'b0;
777 csr_save_cause_o = 1'b0;
778 ctrl_fsm_ns = DBG_TAKEN_ID;
779 flush_id = 1'b0;
780 end else begin
781 // If EBREAK won't enter debug mode (dcsr.ebreakm/u not set) then raise a breakpoint
782 // exception.
783 exc_cause_o = ExcCauseBreakpoint;
784 end
785 end
786 store_err_prio: begin
787 exc_cause_o = ExcCauseStoreAccessFault;
788 csr_mtval_o = lsu_addr_last_i;
789 end
790 load_err_prio: begin
791 exc_cause_o = ExcCauseLoadAccessFault;
792 csr_mtval_o = lsu_addr_last_i;
793 end
794 default: ;
795 endcase
796 end else begin
797 // special instructions and pipeline flushes
798 if (mret_insn) begin
799 pc_mux_o = PC_ERET;
800 pc_set_o = 1'b1;
801 csr_restore_mret_id_o = 1'b1;
802 if (nmi_mode_q) begin
803 nmi_mode_d = 1'b0; // exit NMI mode
804 end
805 end else if (dret_insn) begin
806 pc_mux_o = PC_DRET;
807 pc_set_o = 1'b1;
808 debug_mode_d = 1'b0;
809 csr_restore_dret_id_o = 1'b1;
810 end else if (wfi_insn) begin
811 ctrl_fsm_ns = WAIT_SLEEP;
812 end
813 end // exc_req_q
814
815 // Entering debug mode due to either single step or debug_req. Ensure
816 // registers are set for exception but then enter debug handler rather
817 // than exception handler
818 // Leave all other signals as is to ensure CSRs and PC get set as if
819 // core was entering exception handler, entry to debug mode will then
820 // see the appropriate state and setup dpc correctly.
821
822 // If an EBREAK instruction is causing us to enter debug mode on the
823 // same cycle as a debug_req or single step, honor the EBREAK and
824 // proceed to DBG_TAKEN_ID, as it has the highest priority.
825 //
826 // cause==EBREAK -> prio 3 (highest)
827 // cause==debug_req -> prio 2
828 // cause==step -> prio 1 (lowest)
829 if (enter_debug_mode_prio_q && !(ebrk_insn_prio && ebreak_into_debug)) begin
830 ctrl_fsm_ns = DBG_TAKEN_IF;
831 end
832 end // FLUSH
833
834 default: begin
835 instr_req_o = 1'b0;
836 ctrl_fsm_ns = RESET;
837 end
838 endcase
839
840 if (~instr_exec_i) begin
841 // Hold halt_if high when instr_exec_i is low to stop accepting instructions from the IF stage
842 halt_if = 1'b1;
843 end
844 end

846 assign flush_id_o = flush_id;
847
848 // signal to CSR when in debug mode
849 assign debug_mode_o = debug_mode_q;
850
851 // signal to CSR when in an NMI handler (for nested exception handling)
852 assign nmi_mode_o = nmi_mode_q;
853
854 ///////////////////
855 // Stall control //
856 ///////////////////
857
858 // If high current instruction cannot complete this cycle. Either because it needs more cycles to
859 // finish (stall_id_i) or because the writeback stage cannot accept it yet (stall_wb_i). If there
860 // is no writeback stage stall_wb_i is a constant 0.
861 assign stall = stall_id_i | stall_wb_i;
862
863 // signal to IF stage that ID stage is ready for next instr
864 assign id_in_ready_o = ~stall & ~halt_if & ~retain_id;
865
866 // kill instr in IF-ID pipeline reg that are done, or if a
867 // multicycle instr causes an exception for example
868 // retain_id is another kind of stall, where the instr_valid bit must remain
869 // set (unless flush_id is set also). It cannot be factored directly into
870 // stall as this causes a combinational loop.
871 assign instr_valid_clear_o = ~(stall | retain_id) | flush_id;
872
873 // update registers
874 always_ff @(posedge clk_i or negedge rst_ni) begin : update_regs
875 if (!rst_ni) begin
876 ctrl_fsm_cs <= RESET;
877 nmi_mode_q <= 1'b0;
878 do_single_step_q <= 1'b0;
879 debug_mode_q <= 1'b0;
880 enter_debug_mode_prio_q <= 1'b0;
881 load_err_q <= 1'b0;
882 store_err_q <= 1'b0;
883 exc_req_q <= 1'b0;
884 illegal_insn_q <= 1'b0;
885 end else begin
886 ctrl_fsm_cs <= ctrl_fsm_ns;
887 nmi_mode_q <= nmi_mode_d;
888 do_single_step_q <= do_single_step_d;
889 debug_mode_q <= debug_mode_d;
890 enter_debug_mode_prio_q <= enter_debug_mode_prio_d;
891 load_err_q <= load_err_d;
892 store_err_q <= store_err_d;
893 exc_req_q <= exc_req_d;
894 illegal_insn_q <= illegal_insn_d;
895 end
896 end

929 // If something that causes a jump into an exception handler is seen that jump must occur before
930 // the next instruction executes. The logic tracks whether a jump into an exception handler is
931 // expected. Assertions check the jump occurs.
932
933 logic exception_req, exception_req_pending, exception_req_accepted, exception_req_done;
934 logic exception_pc_set, seen_exception_pc_set, expect_exception_pc_set;
935 logic exception_req_needs_pc_set;

937 assign exception_req = (special_req | enter_debug_mode | handle_irq);
938 // Any exception rquest will cause a transition out of DECODE, once the controller transitions
939 // back into DECODE we're done handling the request.
940 assign exception_req_done =
941 exception_req_pending & (ctrl_fsm_cs != DECODE) & (ctrl_fsm_ns == DECODE);
942
943 assign exception_req_needs_pc_set = enter_debug_mode | handle_irq | special_req_pc_change;
944
945 // An exception PC set uses specific PC types
946 assign exception_pc_set =
947 exception_req_pending & (pc_set_o & (pc_mux_o inside {PC_EXC, PC_ERET, PC_DRET}));
948
949 always @(posedge clk_i or negedge rst_ni) begin
950 if (!rst_ni) begin
951 exception_req_pending <= 1'b0;
952 exception_req_accepted <= 1'b0;
953 expect_exception_pc_set <= 1'b0;
954 seen_exception_pc_set <= 1'b0;
955 end else begin
956 // Keep `exception_req_pending` asserted once an exception_req is seen until it is done
957 exception_req_pending <= (exception_req_pending | exception_req) & ~exception_req_done;
958
959 // The exception req has been accepted once the controller transitions out of decode
960 exception_req_accepted <= (exception_req_accepted & ~exception_req_done) |
961 (exception_req & ctrl_fsm_ns != DECODE);
962
963 // Set `expect_exception_pc_set` if exception req needs one and keep it asserted until
964 // exception req is done
965 expect_exception_pc_set <= (expect_exception_pc_set | exception_req_needs_pc_set) &
966 ~exception_req_done;
967
968 // Keep `seen_exception_pc_set` asserted once an exception PC set is seen until the
969 // exception req is done
970 seen_exception_pc_set <= (seen_exception_pc_set | exception_pc_set) & ~exception_req_done;
971 end
972 end

975 // DECODE

979 // pending or it is done this cycle.

984 // excepting those to move into debug mode.

992 // cycle or a previous one).

1001 // If entering or exiting debug mode, the pipeline must be flushed. This is because Ibex
1002 // currently does not support some of the pipeline stages being in debug mode; either all or
