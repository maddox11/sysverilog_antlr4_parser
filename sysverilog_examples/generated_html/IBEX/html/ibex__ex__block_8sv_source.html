6/**
7 * Execution stage
8 *
9 * Execution block: Hosts ALU and MUL/DIV unit
10 */
11module ibex_ex_block #(
12 parameter ibex_pkg::rv32m_e RV32M = ibex_pkg::RV32MFast,
13 parameter ibex_pkg::rv32b_e RV32B = ibex_pkg::RV32BNone,
14 parameter bit BranchTargetALU = 0
15) (
16 input logic clk_i,
17 input logic rst_ni,
18
19 // ALU
20 input ibex_pkg::alu_op_e alu_operator_i,
21 input logic [31:0] alu_operand_a_i,
22 input logic [31:0] alu_operand_b_i,
23 input logic alu_instr_first_cycle_i,
24
25 // Branch Target ALU
26 // All of these signals are unusued when BranchTargetALU == 0
27 input logic [31:0] bt_a_operand_i,
28 input logic [31:0] bt_b_operand_i,
29
30 // Multiplier/Divider
31 input ibex_pkg::md_op_e multdiv_operator_i,
32 input logic mult_en_i, // dynamic enable signal, for FSM control
33 input logic div_en_i, // dynamic enable signal, for FSM control
34 input logic mult_sel_i, // static decoder output, for data muxes
35 input logic div_sel_i, // static decoder output, for data muxes
36 input logic [1:0] multdiv_signed_mode_i,
37 input logic [31:0] multdiv_operand_a_i,
38 input logic [31:0] multdiv_operand_b_i,
39 input logic multdiv_ready_id_i,
40 input logic data_ind_timing_i,
41
42 // intermediate val reg
43 output logic [1:0] imd_val_we_o,
44 output logic [33:0] imd_val_d_o[2],
45 input logic [33:0] imd_val_q_i[2],
46
47 // Outputs
48 output logic [31:0] alu_adder_result_ex_o, // to LSU
49 output logic [31:0] result_ex_o,
50 output logic [31:0] branch_target_o, // to IF
51 output logic branch_decision_o, // to ID
52
53 output logic ex_valid_o // EX has valid output
54);
55
56 import ibex_pkg::*;
57
58 logic [31:0] alu_result, multdiv_result;
59
60 logic [32:0] multdiv_alu_operand_b, multdiv_alu_operand_a;
61 logic [33:0] alu_adder_result_ext;
62 logic alu_cmp_result, alu_is_equal_result;
63 logic multdiv_valid;
64 logic multdiv_sel;
65 logic [31:0] alu_imd_val_q[2];
66 logic [31:0] alu_imd_val_d[2];
67 logic [ 1:0] alu_imd_val_we;
68 logic [33:0] multdiv_imd_val_d[2];
69 logic [ 1:0] multdiv_imd_val_we;
70
71 /*
72 The multdiv_i output is never selected if RV32M=RV32MNone
73 At synthesis time, all the combinational and sequential logic
74 from the multdiv_i module are eliminated
75 */
76 if (RV32M != RV32MNone) begin : gen_multdiv_m
77 assign multdiv_sel = mult_sel_i | div_sel_i;
78 end else begin : gen_multdiv_no_m
79 assign multdiv_sel = 1'b0;
80 end
81
82 // Intermediate Value Register Mux
83 assign imd_val_d_o[0] = multdiv_sel ? multdiv_imd_val_d[0] : {2'b0, alu_imd_val_d[0]};
84 assign imd_val_d_o[1] = multdiv_sel ? multdiv_imd_val_d[1] : {2'b0, alu_imd_val_d[1]};
85 assign imd_val_we_o = multdiv_sel ? multdiv_imd_val_we : alu_imd_val_we;
86
87 assign alu_imd_val_q = '{imd_val_q_i[0][31:0], imd_val_q_i[1][31:0]};
88
89 assign result_ex_o = multdiv_sel ? multdiv_result : alu_result;
90
91 // branch handling
92 assign branch_decision_o = alu_cmp_result;
93
94 if (BranchTargetALU) begin : g_branch_target_alu
95 logic [32:0] bt_alu_result;
96 logic unused_bt_carry;
97
98 assign bt_alu_result = bt_a_operand_i + bt_b_operand_i;
99
100 assign unused_bt_carry = bt_alu_result[32];
101 assign branch_target_o = bt_alu_result[31:0];
102 end else begin : g_no_branch_target_alu
103 // Unused bt_operand signals cause lint errors, this avoids them
104 logic [31:0] unused_bt_a_operand, unused_bt_b_operand;
105
106 assign unused_bt_a_operand = bt_a_operand_i;
107 assign unused_bt_b_operand = bt_b_operand_i;
108
109 assign branch_target_o = alu_adder_result_ex_o;
110 end
111
112 /////////
113 // ALU //
114 /////////
115
116 ibex_alu #(
117 .RV32B(RV32B)
118 ) alu_i (
119 .operator_i (alu_operator_i),
120 .operand_a_i (alu_operand_a_i),
121 .operand_b_i (alu_operand_b_i),
122 .instr_first_cycle_i(alu_instr_first_cycle_i),
123 .imd_val_q_i (alu_imd_val_q),
124 .imd_val_we_o (alu_imd_val_we),
125 .imd_val_d_o (alu_imd_val_d),
126 .multdiv_operand_a_i(multdiv_alu_operand_a),
127 .multdiv_operand_b_i(multdiv_alu_operand_b),
128 .multdiv_sel_i (multdiv_sel),
129 .adder_result_o (alu_adder_result_ex_o),
130 .adder_result_ext_o (alu_adder_result_ext),
131 .result_o (alu_result),
132 .comparison_result_o(alu_cmp_result),
133 .is_equal_result_o (alu_is_equal_result)
134 );
135
136 ////////////////
137 // Multiplier //
138 ////////////////
139
140 if (RV32M == RV32MSlow) begin : gen_multdiv_slow
141 ibex_multdiv_slow multdiv_i (
142 .clk_i (clk_i),
143 .rst_ni (rst_ni),
144 .mult_en_i (mult_en_i),
145 .div_en_i (div_en_i),
146 .mult_sel_i (mult_sel_i),
147 .div_sel_i (div_sel_i),
148 .operator_i (multdiv_operator_i),
149 .signed_mode_i (multdiv_signed_mode_i),
150 .op_a_i (multdiv_operand_a_i),
151 .op_b_i (multdiv_operand_b_i),
152 .alu_adder_ext_i (alu_adder_result_ext),
153 .alu_adder_i (alu_adder_result_ex_o),
154 .equal_to_zero_i (alu_is_equal_result),
155 .data_ind_timing_i (data_ind_timing_i),
156 .valid_o (multdiv_valid),
157 .alu_operand_a_o (multdiv_alu_operand_a),
158 .alu_operand_b_o (multdiv_alu_operand_b),
159 .imd_val_q_i (imd_val_q_i),
160 .imd_val_d_o (multdiv_imd_val_d),
161 .imd_val_we_o (multdiv_imd_val_we),
162 .multdiv_ready_id_i(multdiv_ready_id_i),
163 .multdiv_result_o (multdiv_result)
164 );
165 end else if (RV32M == RV32MFast || RV32M == RV32MSingleCycle) begin : gen_multdiv_fast
166 ibex_multdiv_fast #(
167 .RV32M(RV32M)
168 ) multdiv_i (
169 .clk_i (clk_i),
170 .rst_ni (rst_ni),
171 .mult_en_i (mult_en_i),
172 .div_en_i (div_en_i),
173 .mult_sel_i (mult_sel_i),
174 .div_sel_i (div_sel_i),
175 .operator_i (multdiv_operator_i),
176 .signed_mode_i (multdiv_signed_mode_i),
177 .op_a_i (multdiv_operand_a_i),
178 .op_b_i (multdiv_operand_b_i),
179 .alu_operand_a_o (multdiv_alu_operand_a),
180 .alu_operand_b_o (multdiv_alu_operand_b),
181 .alu_adder_ext_i (alu_adder_result_ext),
182 .alu_adder_i (alu_adder_result_ex_o),
183 .equal_to_zero_i (alu_is_equal_result),
184 .data_ind_timing_i (data_ind_timing_i),
185 .imd_val_q_i (imd_val_q_i),
186 .imd_val_d_o (multdiv_imd_val_d),
187 .imd_val_we_o (multdiv_imd_val_we),
188 .multdiv_ready_id_i(multdiv_ready_id_i),
189 .valid_o (multdiv_valid),
190 .multdiv_result_o (multdiv_result)
191 );
192 end
193
194 // Multiplier/divider may require multiple cycles. The ALU output is valid in the same cycle
195 // unless the intermediate result register is being written (which indicates this isn't the
196 // final cycle of ALU operation).
197 assign ex_valid_o = multdiv_sel ? multdiv_valid : ~(|alu_imd_val_we);
198
199 `ifdef INC_ASSERT
200 // This is intended to be accessed via hierarchal references so isn't output from this module nor
201 // used in any logic in this module
202 logic sva_multdiv_fsm_idle;
203
204 if (RV32M == RV32MSlow) begin : gen_multdiv_sva_idle_slow
205 assign sva_multdiv_fsm_idle = gen_multdiv_slow.multdiv_i.sva_fsm_idle;
206 end else if (RV32M == RV32MFast || RV32M == RV32MSingleCycle) begin : gen_multdiv_sva_idle_fast
207 assign sva_multdiv_fsm_idle = gen_multdiv_fast.multdiv_i.sva_fsm_idle;
208 end else begin : gen_multdiv_sva_idle_none
209 assign sva_multdiv_fsm_idle = 1'b1;
210 end
211
212 // Mark the sva_multdiv_fsm_idle as unused to avoid lint issues
213 logic unused_sva_multdiv_fsm_idle;
214 assign unused_sva_multdiv_fsm_idle = sva_multdiv_fsm_idle;
215 `endif
216
217endmodule
