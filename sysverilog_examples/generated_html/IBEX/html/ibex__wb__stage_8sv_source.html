5/**
6 * Writeback Stage
7 *
8 * Writeback is an optional third pipeline stage. It writes data back to the register file that was
9 * produced in the ID/EX stage or awaits a response to a load/store (LSU writes direct to register
10 * file for load data). If the writeback stage is not present (WritebackStage == 0) this acts as
11 * a simple passthrough to write data direct to the register file.
12 */
13
14`include "prim_assert.sv"
15`include "dv_fcov_macros.svh"
16
17module ibex_wb_stage #(
18 parameter bit ResetAll = 1'b0,
19 parameter bit WritebackStage = 1'b0,
20 parameter bit DummyInstructions = 1'b0
21) (
22 input logic clk_i,
23 input logic rst_ni,
24
25 input logic en_wb_i,
26 input ibex_pkg::wb_instr_type_e instr_type_wb_i,
27 input logic [31:0] pc_id_i,
28 input logic instr_is_compressed_id_i,
29 input logic instr_perf_count_id_i,
30
31 output logic ready_wb_o,
32 output logic rf_write_wb_o,
33 output logic outstanding_load_wb_o,
34 output logic outstanding_store_wb_o,
35 output logic [31:0] pc_wb_o,
36 output logic perf_instr_ret_wb_o,
37 output logic perf_instr_ret_compressed_wb_o,
38 output logic perf_instr_ret_wb_spec_o,
39 output logic perf_instr_ret_compressed_wb_spec_o,
40
41 input logic [4:0] rf_waddr_id_i,
42 input logic [31:0] rf_wdata_id_i,
43 input logic rf_we_id_i,
44
45 input logic dummy_instr_id_i,
46
47 input logic [31:0] rf_wdata_lsu_i,
48 input logic rf_we_lsu_i,
49
50 output logic [31:0] rf_wdata_fwd_wb_o,
51
52 output logic [4:0] rf_waddr_wb_o,
53 output logic [31:0] rf_wdata_wb_o,
54 output logic rf_we_wb_o,
55
56 output logic dummy_instr_wb_o,
57
58 input logic lsu_resp_valid_i,
59 input logic lsu_resp_err_i,
60
61 output logic instr_done_wb_o
62);
63
64 import ibex_pkg::*;
65
66 // 0 == RF write from ID
67 // 1 == RF write from LSU
68 logic [31:0] rf_wdata_wb_mux [2];
69 logic [1:0] rf_wdata_wb_mux_we;
70
71 if (WritebackStage) begin : g_writeback_stage
72 logic [31:0] rf_wdata_wb_q;
73 logic rf_we_wb_q;
74 logic [4:0] rf_waddr_wb_q;
75
76 logic wb_done;
77
78 logic wb_valid_q;
79 logic [31:0] wb_pc_q;
80 logic wb_compressed_q;
81 logic wb_count_q;
82 wb_instr_type_e wb_instr_type_q;
83
84 logic wb_valid_d;
85
86 // Stage becomes valid if an instruction enters for ID/EX and valid is cleared when instruction
87 // is done
88 assign wb_valid_d = (en_wb_i & ready_wb_o) | (wb_valid_q & ~wb_done);
89
90 // Writeback for non load/store instructions always completes in a cycle (so instantly done)
91 // Writeback for load/store must wait for response to be received by the LSU
92 // Signal only relevant if wb_valid_q set
93 assign wb_done = (wb_instr_type_q == WB_INSTR_OTHER) | lsu_resp_valid_i;
94
95 always_ff @(posedge clk_i or negedge rst_ni) begin
96 if (!rst_ni) begin
97 wb_valid_q <= 1'b0;
98 end else begin
99 wb_valid_q <= wb_valid_d;
100 end
101 end
102
103 if (ResetAll) begin : g_wb_regs_ra
104 always_ff @(posedge clk_i or negedge rst_ni) begin
105 if (!rst_ni) begin
106 rf_we_wb_q <= '0;
107 rf_waddr_wb_q <= '0;
108 rf_wdata_wb_q <= '0;
109 wb_instr_type_q <= wb_instr_type_e'(0);
110 wb_pc_q <= '0;
111 wb_compressed_q <= '0;
112 wb_count_q <= '0;
113 end else if (en_wb_i) begin
114 rf_we_wb_q <= rf_we_id_i;
115 rf_waddr_wb_q <= rf_waddr_id_i;
116 rf_wdata_wb_q <= rf_wdata_id_i;
117 wb_instr_type_q <= instr_type_wb_i;
118 wb_pc_q <= pc_id_i;
119 wb_compressed_q <= instr_is_compressed_id_i;
120 wb_count_q <= instr_perf_count_id_i;
121 end
122 end
123 end else begin : g_wb_regs_nr
124 always_ff @(posedge clk_i) begin
125 if (en_wb_i) begin
126 rf_we_wb_q <= rf_we_id_i;
127 rf_waddr_wb_q <= rf_waddr_id_i;
128 rf_wdata_wb_q <= rf_wdata_id_i;
129 wb_instr_type_q <= instr_type_wb_i;
130 wb_pc_q <= pc_id_i;
131 wb_compressed_q <= instr_is_compressed_id_i;
132 wb_count_q <= instr_perf_count_id_i;
133 end
134 end
135 end
136
137 assign rf_waddr_wb_o = rf_waddr_wb_q;
138 assign rf_wdata_wb_mux[0] = rf_wdata_wb_q;
139 assign rf_wdata_wb_mux_we[0] = rf_we_wb_q & wb_valid_q;
140
141 assign ready_wb_o = ~wb_valid_q | wb_done;
142
143 // Instruction in writeback will be writing to register file if either rf_we is set or writeback
144 // is awaiting load data. This is used for determining RF read hazards in ID/EX
145 assign rf_write_wb_o = wb_valid_q & (rf_we_wb_q | (wb_instr_type_q == WB_INSTR_LOAD));
146
147 assign outstanding_load_wb_o = wb_valid_q & (wb_instr_type_q == WB_INSTR_LOAD);
148 assign outstanding_store_wb_o = wb_valid_q & (wb_instr_type_q == WB_INSTR_STORE);
149
150 assign pc_wb_o = wb_pc_q;
151
152 assign instr_done_wb_o = wb_valid_q & wb_done;
153
154 // Increment instruction retire counters for valid instructions which are not lsu errors.
155 // Speculative versions of the signals do not factor in exceptions and whether the instruction
156 // is done yet. These are used to get correct values for instructions reading the relevant
157 // performance counters in the ID stage.
158 assign perf_instr_ret_wb_spec_o = wb_count_q;
159 assign perf_instr_ret_compressed_wb_spec_o = perf_instr_ret_wb_spec_o & wb_compressed_q;
160 assign perf_instr_ret_wb_o = instr_done_wb_o & wb_count_q &
161 ~(lsu_resp_valid_i & lsu_resp_err_i);
162 assign perf_instr_ret_compressed_wb_o = perf_instr_ret_wb_o & wb_compressed_q;
163
164 // Forward data that will be written to the RF back to ID to resolve data hazards. The flopped
165 // rf_wdata_wb_q is used rather than rf_wdata_wb_o as the latter includes read data from memory
166 // that returns too late to be used on the forwarding path.
167 assign rf_wdata_fwd_wb_o = rf_wdata_wb_q;
168
169 assign rf_wdata_wb_mux_we[1] = rf_we_lsu_i;
170
171 if (DummyInstructions) begin : g_dummy_instr_wb
172 logic dummy_instr_wb_q;
173
174 if (ResetAll) begin : g_dummy_instr_wb_regs_ra
175 always_ff @(posedge clk_i or negedge rst_ni) begin
176 if (!rst_ni) begin
177 dummy_instr_wb_q <= 1'b0;
178 end else if (en_wb_i) begin
179 dummy_instr_wb_q <= dummy_instr_id_i;
180 end
181 end
182 end else begin : g_dummy_instr_wb_regs_nr
183 always_ff @(posedge clk_i) begin
184 if (en_wb_i) begin
185 dummy_instr_wb_q <= dummy_instr_id_i;
186 end
187 end
188 end
189
190 assign dummy_instr_wb_o = dummy_instr_wb_q;
191 end else begin : g_no_dummy_instr_wb
192 logic unused_dummy_instr_id;
193 assign unused_dummy_instr_id = dummy_instr_id_i;
194
195 assign dummy_instr_wb_o = 1'b0;
196 end
197 end else begin : g_bypass_wb
198 // without writeback stage just pass through register write signals
199 assign rf_waddr_wb_o = rf_waddr_id_i;
200 assign rf_wdata_wb_mux[0] = rf_wdata_id_i;
201 assign rf_wdata_wb_mux_we[0] = rf_we_id_i;
202 assign rf_wdata_wb_mux_we[1] = rf_we_lsu_i;
203
204 assign dummy_instr_wb_o = dummy_instr_id_i;
205
206 // Increment instruction retire counters for valid instructions which are not lsu errors.
207 // The speculative signals are always 0 when no writeback stage is present as the raw counter
208 // values will be correct.
209 assign perf_instr_ret_wb_spec_o = 1'b0;
210 assign perf_instr_ret_compressed_wb_spec_o = 1'b0;
211 assign perf_instr_ret_wb_o = instr_perf_count_id_i & en_wb_i &
212 ~(lsu_resp_valid_i & lsu_resp_err_i);
213 assign perf_instr_ret_compressed_wb_o = perf_instr_ret_wb_o & instr_is_compressed_id_i;
214
215 // ready needs to be constant 1 without writeback stage (otherwise ID/EX stage will stall)
216 assign ready_wb_o = 1'b1;
217
218 // Unused Writeback stage only IO & wiring
219 // Assign inputs and internal wiring to unused signals to satisfy lint checks
220 // Tie-off outputs to constant values
221 logic unused_clk;
222 logic unused_rst;
223 wb_instr_type_e unused_instr_type_wb;
224 logic [31:0] unused_pc_id;
225 logic unused_dummy_instr_id;
226
227 assign unused_clk = clk_i;
228 assign unused_rst = rst_ni;
229 assign unused_instr_type_wb = instr_type_wb_i;
230 assign unused_pc_id = pc_id_i;
231 assign unused_dummy_instr_id = dummy_instr_id_i;
232
233 assign outstanding_load_wb_o = 1'b0;
234 assign outstanding_store_wb_o = 1'b0;
235 assign pc_wb_o = '0;
236 assign rf_write_wb_o = 1'b0;
237 assign rf_wdata_fwd_wb_o = 32'b0;
238 assign instr_done_wb_o = 1'b0;
239 end
240
241 assign rf_wdata_wb_mux[1] = rf_wdata_lsu_i;
242
243 // RF write data can come from ID results (all RF writes that aren't because of loads will come
244 // from here) or the LSU (RF writes for load data)
245 assign rf_wdata_wb_o = ({32{rf_wdata_wb_mux_we[0]}} & rf_wdata_wb_mux[0]) |
246 ({32{rf_wdata_wb_mux_we[1]}} & rf_wdata_wb_mux[1]);
247 assign rf_we_wb_o = |rf_wdata_wb_mux_we;
248
249 `DV_FCOV_SIGNAL_GEN_IF(logic, wb_valid, g_writeback_stage.wb_valid_q, WritebackStage)
250
251 `ASSERT(RFWriteFromOneSourceOnly, $onehot0(rf_wdata_wb_mux_we))
252endmodule
