6/**
7 * Fetch Fifo for 32 bit memory interface
8 *
9 * input port: send address and data to the FIFO
10 * clear_i clears the FIFO for the following cycle, including any new request
11 */
12
13`include "prim_assert.sv"
14
15module ibex_fetch_fifo #(
16 parameter int unsigned NUM_REQS = 2,
17 parameter bit ResetAll = 1'b0
18) (
19 input logic clk_i,
20 input logic rst_ni,
21
22 // control signals
23 input logic clear_i, // clears the contents of the FIFO
24 output logic [NUM_REQS-1:0] busy_o,
25
26 // input port
27 input logic in_valid_i,
28 input logic [31:0] in_addr_i,
29 input logic [31:0] in_rdata_i,
30 input logic in_err_i,
31
32 // output port
33 output logic out_valid_o,
34 input logic out_ready_i,
35 output logic [31:0] out_addr_o,
36 output logic [31:0] out_rdata_o,
37 output logic out_err_o,
38 output logic out_err_plus2_o
39);
40
41 localparam int unsigned DEPTH = NUM_REQS+1;
42
43 // index 0 is used for output
44 logic [DEPTH-1:0] [31:0] rdata_d, rdata_q;
45 logic [DEPTH-1:0] err_d, err_q;
46 logic [DEPTH-1:0] valid_d, valid_q;
47 logic [DEPTH-1:0] lowest_free_entry;
48 logic [DEPTH-1:0] valid_pushed, valid_popped;
49 logic [DEPTH-1:0] entry_en;
50
51 logic pop_fifo;
52 logic [31:0] rdata, rdata_unaligned;
53 logic err, err_unaligned, err_plus2;
54 logic valid, valid_unaligned;
55
56 logic aligned_is_compressed, unaligned_is_compressed;
57
58 logic addr_incr_two;
59 logic [31:1] instr_addr_next;
60 logic [31:1] instr_addr_d, instr_addr_q;
61 logic instr_addr_en;
62 logic unused_addr_in;
63
64 /////////////////
65 // Output port //
66 /////////////////
67
68 assign rdata = valid_q[0] ? rdata_q[0] : in_rdata_i;
69 assign err = valid_q[0] ? err_q[0] : in_err_i;
70 assign valid = valid_q[0] | in_valid_i;
71
72 // The FIFO contains word aligned memory fetches, but the instructions contained in each entry
73 // might be half-word aligned (due to compressed instructions)
74 // e.g.
75 // | 31 16 | 15 0 |
76 // FIFO entry 0 | Instr 1 [15:0] | Instr 0 [15:0] |
77 // FIFO entry 1 | Instr 2 [15:0] | Instr 1 [31:16] |
78 //
79 // The FIFO also has a direct bypass path, so a complete instruction might be made up of data
80 // from the FIFO and new incoming data.
81 //
82
83 // Construct the output data for an unaligned instruction
84 assign rdata_unaligned = valid_q[1] ? {rdata_q[1][15:0], rdata[31:16]} :
85 {in_rdata_i[15:0], rdata[31:16]};
86
87 // If entry[1] is valid, an error can come from entry[0] or entry[1], unless the
88 // instruction in entry[0] is compressed (entry[1] is a new instruction)
89 // If entry[1] is not valid, and entry[0] is, an error can come from entry[0] or the incoming
90 // data, unless the instruction in entry[0] is compressed
91 // If entry[0] is not valid, the error must come from the incoming data
92 assign err_unaligned = valid_q[1] ? ((err_q[1] & ~unaligned_is_compressed) | err_q[0]) :
93 ((valid_q[0] & err_q[0]) |
94 (in_err_i & (~valid_q[0] | ~unaligned_is_compressed)));
95
96 // Record when an error is caused by the second half of an unaligned 32bit instruction.
97 // Only needs to be correct when unaligned and if err_unaligned is set
98 assign err_plus2 = valid_q[1] ? (err_q[1] & ~err_q[0]) :
99 (in_err_i & valid_q[0] & ~err_q[0]);
100
101 // An uncompressed unaligned instruction is only valid if both parts are available
102 assign valid_unaligned = valid_q[1] ? 1'b1 :
103 (valid_q[0] & in_valid_i);
104
105 // If there is an error, rdata is unknown
106 assign unaligned_is_compressed = (rdata[17:16] != 2'b11) & ~err;
107 assign aligned_is_compressed = (rdata[ 1: 0] != 2'b11) & ~err;
108
109 ////////////////////////////////////////
110 // Instruction aligner (if unaligned) //
111 ////////////////////////////////////////
112
113 always_comb begin
114 if (out_addr_o[1]) begin
115 // unaligned case
116 out_rdata_o = rdata_unaligned;
117 out_err_o = err_unaligned;
118 out_err_plus2_o = err_plus2;
119
120 if (unaligned_is_compressed) begin
121 out_valid_o = valid;
122 end else begin
123 out_valid_o = valid_unaligned;
124 end
125 end else begin
126 // aligned case
127 out_rdata_o = rdata;
128 out_err_o = err;
129 out_err_plus2_o = 1'b0;
130 out_valid_o = valid;
131 end
132 end
133
134 /////////////////////////
135 // Instruction address //
136 /////////////////////////
137
138 // Update the address on branches and every time an instruction is driven
139 assign instr_addr_en = clear_i | (out_ready_i & out_valid_o);
140
141 // Increment the address by two every time a compressed instruction is popped
142 assign addr_incr_two = instr_addr_q[1] ? unaligned_is_compressed :
143 aligned_is_compressed;
144
145 assign instr_addr_next = (instr_addr_q[31:1] +
146 // Increment address by 4 or 2
147 {29'd0,~addr_incr_two,addr_incr_two});
148
149 assign instr_addr_d = clear_i ? in_addr_i[31:1] :
150 instr_addr_next;
151
152 if (ResetAll) begin : g_instr_addr_ra
153 always_ff @(posedge clk_i or negedge rst_ni) begin
154 if (!rst_ni) begin
155 instr_addr_q <= '0;
156 end else if (instr_addr_en) begin
157 instr_addr_q <= instr_addr_d;
158 end
159 end
160 end else begin : g_instr_addr_nr
161 always_ff @(posedge clk_i) begin
162 if (instr_addr_en) begin
163 instr_addr_q <= instr_addr_d;
164 end
165 end
166 end
167
168 // Output PC of current instruction
169 assign out_addr_o = {instr_addr_q, 1'b0};
170
171 // The LSB of the address is unused, since all addresses are halfword aligned
172 assign unused_addr_in = in_addr_i[0];
173
174 /////////////////
175 // FIFO status //
176 /////////////////
177
178 // Indicate the fill level of fifo-entries. This is used to determine when a new request can be
179 // made on the bus. The prefetch buffer only needs to know about the upper entries which overlap
180 // with NUM_REQS.
181 assign busy_o = valid_q[DEPTH-1:DEPTH-NUM_REQS];
182
183 /////////////////////
184 // FIFO management //
185 /////////////////////
186
187 // Since an entry can contain unaligned instructions, popping an entry can leave the entry valid
188 assign pop_fifo = out_ready_i & out_valid_o & (~aligned_is_compressed | out_addr_o[1]);
189
190 for (genvar i = 0; i < (DEPTH - 1); i++) begin : g_fifo_next
191 // Calculate lowest free entry (write pointer)
192 if (i == 0) begin : g_ent0
193 assign lowest_free_entry[i] = ~valid_q[i];
194 end else begin : g_ent_others
195 assign lowest_free_entry[i] = ~valid_q[i] & valid_q[i-1];
196 end
197
198 // An entry is set when an incoming request chooses the lowest available entry
199 assign valid_pushed[i] = (in_valid_i & lowest_free_entry[i]) |
200 valid_q[i];
201 // Popping the FIFO shifts all entries down
202 assign valid_popped[i] = pop_fifo ? valid_pushed[i+1] : valid_pushed[i];
203 // All entries are wiped out on a clear
204 assign valid_d[i] = valid_popped[i] & ~clear_i;
205
206 // data flops are enabled if there is new data to shift into it, or
207 assign entry_en[i] = (valid_pushed[i+1] & pop_fifo) |
208 // a new request is incoming and this is the lowest free entry
209 (in_valid_i & lowest_free_entry[i] & ~pop_fifo);
210
211 // take the next entry or the incoming data
212 assign rdata_d[i] = valid_q[i+1] ? rdata_q[i+1] : in_rdata_i;
213 assign err_d [i] = valid_q[i+1] ? err_q [i+1] : in_err_i;
214 end
215 // The top entry is similar but with simpler muxing
216 assign lowest_free_entry[DEPTH-1] = ~valid_q[DEPTH-1] & valid_q[DEPTH-2];
217 assign valid_pushed [DEPTH-1] = valid_q[DEPTH-1] | (in_valid_i & lowest_free_entry[DEPTH-1]);
218 assign valid_popped [DEPTH-1] = pop_fifo ? 1'b0 : valid_pushed[DEPTH-1];
219 assign valid_d [DEPTH-1] = valid_popped[DEPTH-1] & ~clear_i;
220 assign entry_en[DEPTH-1] = in_valid_i & lowest_free_entry[DEPTH-1];
221 assign rdata_d [DEPTH-1] = in_rdata_i;
222 assign err_d [DEPTH-1] = in_err_i;
223
224 ////////////////////
225 // FIFO registers //
226 ////////////////////
227
228 always_ff @(posedge clk_i or negedge rst_ni) begin
229 if (!rst_ni) begin
230 valid_q <= '0;
231 end else begin
232 valid_q <= valid_d;
233 end
234 end
235
236 for (genvar i = 0; i < DEPTH; i++) begin : g_fifo_regs
237 if (ResetAll) begin : g_rdata_ra
238 always_ff @(posedge clk_i or negedge rst_ni) begin
239 if (!rst_ni) begin
240 rdata_q[i] <= '0;
241 err_q[i] <= '0;
242 end else if (entry_en[i]) begin
243 rdata_q[i] <= rdata_d[i];
244 err_q[i] <= err_d[i];
245 end
246 end
247 end else begin : g_rdata_nr
248 always_ff @(posedge clk_i) begin
249 if (entry_en[i]) begin
250 rdata_q[i] <= rdata_d[i];
251 err_q[i] <= err_d[i];
252 end
253 end
254 end
255 end
256
257 ////////////////
258 // Assertions //
259 ////////////////
260
261 // Must not push and pop simultaneously when FIFO full.
262 `ASSERT(IbexFetchFifoPushPopFull,
263 (in_valid_i&& pop_fifo) |-> (!valid_q[DEPTH-1] || clear_i))
264
265 // Must not push to FIFO when full.
266 `ASSERT(IbexFetchFifoPushFull,
267 (in_valid_i) |-> (!valid_q[DEPTH-1] || clear_i))
268
269endmodule
