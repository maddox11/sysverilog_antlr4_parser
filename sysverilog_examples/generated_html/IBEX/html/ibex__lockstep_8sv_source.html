118 //////////////////////
119 // Reset generation //
120 //////////////////////
121
122 // Upon reset, the comparison is stopped and the shadow core is reset, both immediately. A
123 // counter is started. After LockstepOffset clock cycles:
124 // - The counter is stopped.
125 // - The reset of the shadow core is synchronously released.
126 // The comparison is started in the following clock cycle.
127
128 logic [LockstepOffsetW-1:0] rst_shadow_cnt;
129 logic rst_shadow_cnt_err;
130 ibex_mubi_t rst_shadow_set_d, rst_shadow_set_q;
131 logic rst_shadow_n, rst_shadow_set_single_bit;
132 ibex_mubi_t enable_cmp_d, enable_cmp_q;
133
134 // This counter primitive starts counting to LockstepOffset after a system
135 // reset. The counter value saturates at LockstepOffset.
136 prim_count #(
137 .Width (LockstepOffsetW ),
138 .ResetValue (LockstepOffsetW'(1'b0) )
139 ) u_rst_shadow_cnt (
140 .clk_i (clk_i ),
141 .rst_ni (rst_ni ),
142 .clr_i (1'b0 ),
143 .set_i (1'b0 ),
144 .set_cnt_i ('0 ),
145 .incr_en_i (1'b1 ),
146 .decr_en_i (1'b0 ),
147 .step_i (LockstepOffsetW'(1'b1) ),
148 .commit_i (1'b1 ),
149 .cnt_o (rst_shadow_cnt ),
150 .cnt_after_commit_o ( ),
151 .err_o (rst_shadow_cnt_err )
152 );
153
154 // When the LockstepOffset counter value is reached, activate the lockstep
155 // comparison. We do not explicitly check whether rst_shadow_set_q forms a valid
156 // multibit signal as this value is implicitly checked by the enable_cmp
157 // comparison below.
158 assign rst_shadow_set_d =
159 (rst_shadow_cnt >= LockstepOffsetW'(LockstepOffset - 1)) ? IbexMuBiOn : IbexMuBiOff;
160
161 // Enable lockstep comparison.
162 assign enable_cmp_d = rst_shadow_set_q;
163
164 // This assignment is needed in order to avoid "Warning-IMPERFECTSCH" messages.
165 // TODO: Remove when updating Verilator #2134.
166 assign rst_shadow_set_single_bit = rst_shadow_set_q[0];
167
168 // The primitives below are used to place size-only constraints in order to prevent
169 // synthesis optimizations and preserve anchor points for constraining backend tools.
170 prim_flop #(
171 .Width(IbexMuBiWidth),
172 .ResetValue(IbexMuBiOff)
173 ) u_prim_rst_shadow_set_flop (
174 .clk_i (clk_i),
175 .rst_ni(rst_ni),
176 .d_i (rst_shadow_set_d),
177 .q_o (rst_shadow_set_q)
178 );
179
180 prim_flop #(
181 .Width(IbexMuBiWidth),
182 .ResetValue(IbexMuBiOff)
183 ) u_prim_enable_cmp_flop (
184 .clk_i (clk_i),
185 .rst_ni(rst_ni),
186 .d_i (enable_cmp_d),
187 .q_o (enable_cmp_q)
188 );
189
190 prim_clock_mux2 #(
191 .NoFpgaBufG(1'b1)
192 ) u_prim_rst_shadow_n_mux2 (
193 .clk0_i(rst_shadow_set_single_bit),
194 .clk1_i(scan_rst_ni),
195 .sel_i (test_en_i),
196 .clk_o (rst_shadow_n)
197 );

250 // Delay the inputs

252 if (!rst_ni) begin
253 for (int unsigned i = 0; i < LockstepOffset; i++) begin
254 shadow_inputs_q[i] <= delayed_inputs_t'('0);
255 shadow_tag_rdata_q[i] <= '{default: 0};
256 shadow_data_rdata_q[i] <= '{default: 0};

259 for (int unsigned i = 0; i < LockstepOffset - 1; i++) begin
260 shadow_inputs_q[i] <= shadow_inputs_q[i+1];
261 shadow_tag_rdata_q[i] <= shadow_tag_rdata_q[i+1];
262 shadow_data_rdata_q[i] <= shadow_data_rdata_q[i+1];
263 end
264 shadow_inputs_q[LockstepOffset-1] <= shadow_inputs_in;
265 shadow_tag_rdata_q[LockstepOffset-1] <= ic_tag_rdata_i;
266 shadow_data_rdata_q[LockstepOffset-1] <= ic_data_rdata_i;
267 end
268 end
