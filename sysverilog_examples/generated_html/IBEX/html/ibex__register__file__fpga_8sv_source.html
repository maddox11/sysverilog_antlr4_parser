6/**
7 * RISC-V register file
8 *
9 * Register file with 31 or 15x 32 bit wide registers. Register 0 is fixed to 0.
10 *
11 * This register file is designed to make FPGA synthesis tools infer RAM primitives. For Xilinx
12 * FPGA architectures, it will produce RAM32M primitives. Other vendors have not yet been tested.
13 */
14module ibex_register_file_fpga #(
15 parameter bit RV32E = 0,
16 parameter int unsigned DataWidth = 32,
17 parameter bit DummyInstructions = 0,
18 parameter bit WrenCheck = 0,
19 parameter bit RdataMuxCheck = 0,
20 parameter logic [DataWidth-1:0] WordZeroVal = '0
21) (
22 // Clock and Reset
23 input logic clk_i,
24 input logic rst_ni,
25
26 input logic test_en_i,
27 input logic dummy_instr_id_i,
28 input logic dummy_instr_wb_i,
29
30 //Read port R1
31 input logic [ 4:0] raddr_a_i,
32 output logic [DataWidth-1:0] rdata_a_o,
33 //Read port R2
34 input logic [ 4:0] raddr_b_i,
35 output logic [DataWidth-1:0] rdata_b_o,
36 // Write port W1
37 input logic [ 4:0] waddr_a_i,
38 input logic [DataWidth-1:0] wdata_a_i,
39 input logic we_a_i,
40
41 // This indicates whether spurious WE or non-one-hot encoded raddr are detected.
42 output logic err_o
43);
44
45 localparam int ADDR_WIDTH = RV32E ? 4 : 5;
46 localparam int NUM_WORDS = 2 ** ADDR_WIDTH;
47
48 logic [DataWidth-1:0] mem[NUM_WORDS];
49 logic we; // write enable if writing to any register other than R0
50
51 logic [DataWidth-1:0] mem_o_a, mem_o_b;
52
53 // WE strobe and one-hot encoded raddr alert.
54 logic oh_raddr_a_err, oh_raddr_b_err, oh_we_err;
55 assign err_o = oh_raddr_a_err || oh_raddr_b_err || oh_we_err;
56
57 if (RdataMuxCheck) begin : gen_rdata_mux_check
58 // Encode raddr_a/b into one-hot encoded signals.
59 logic [NUM_WORDS-1:0] raddr_onehot_a, raddr_onehot_b;
60 logic [NUM_WORDS-1:0] raddr_onehot_a_buf, raddr_onehot_b_buf;
61 prim_onehot_enc #(
62 .OneHotWidth(NUM_WORDS)
63 ) u_prim_onehot_enc_raddr_a (
64 .in_i (raddr_a_i),
65 .en_i (1'b1),
66 .out_o (raddr_onehot_a)
67 );
68
69 prim_onehot_enc #(
70 .OneHotWidth(NUM_WORDS)
71 ) u_prim_onehot_enc_raddr_b (
72 .in_i (raddr_b_i),
73 .en_i (1'b1),
74 .out_o (raddr_onehot_b)
75 );
76
77 // Buffer the one-hot encoded signals so that the checkers
78 // are not optimized.
79 prim_buf #(
80 .Width(NUM_WORDS)
81 ) u_prim_buf_raddr_a (
82 .in_i (raddr_onehot_a),
83 .out_o(raddr_onehot_a_buf)
84 );
85
86 prim_buf #(
87 .Width(NUM_WORDS)
88 ) u_prim_buf_raddr_b (
89 .in_i (raddr_onehot_b),
90 .out_o(raddr_onehot_b_buf)
91 );
92
93 // SEC_CM: DATA_REG_SW.GLITCH_DETECT
94 // Check the one-hot encoded signals for glitches.
95 prim_onehot_check #(
96 .AddrWidth(ADDR_WIDTH),
97 .OneHotWidth(NUM_WORDS),
98 .AddrCheck(1),
99 // When AddrCheck=1 also EnableCheck needs to be 1.
100 .EnableCheck(1)
101 ) u_prim_onehot_check_raddr_a (
102 .clk_i,
103 .rst_ni,
104 .oh_i (raddr_onehot_a_buf),
105 .addr_i (raddr_a_i),
106 // Set enable=1 as address is always valid.
107 .en_i (1'b1),
108 .err_o (oh_raddr_a_err)
109 );
110
111 prim_onehot_check #(
112 .AddrWidth(ADDR_WIDTH),
113 .OneHotWidth(NUM_WORDS),
114 .AddrCheck(1),
115 // When AddrCheck=1 also EnableCheck needs to be 1.
116 .EnableCheck(1)
117 ) u_prim_onehot_check_raddr_b (
118 .clk_i,
119 .rst_ni,
120 .oh_i (raddr_onehot_b_buf),
121 .addr_i (raddr_b_i),
122 // Set enable=1 as address is always valid.
123 .en_i (1'b1),
124 .err_o (oh_raddr_b_err)
125 );
126
127 // MUX register to rdata_a/b_o according to raddr_a/b_onehot.
128 prim_onehot_mux #(
129 .Width(DataWidth),
130 .Inputs(NUM_WORDS)
131 ) u_rdata_a_mux (
132 .clk_i,
133 .rst_ni,
134 .in_i (mem),
135 .sel_i (raddr_onehot_a),
136 .out_o (mem_o_a)
137 );
138
139 prim_onehot_mux #(
140 .Width(DataWidth),
141 .Inputs(NUM_WORDS)
142 ) u_rdata_b_mux (
143 .clk_i,
144 .rst_ni,
145 .in_i (mem),
146 .sel_i (raddr_onehot_b),
147 .out_o (mem_o_b)
148 );
149
150 assign rdata_a_o = (raddr_a_i == '0) ? WordZeroVal : mem_o_a;
151 assign rdata_b_o = (raddr_b_i == '0) ? WordZeroVal : mem_o_b;
152 end else begin : gen_no_rdata_mux_check
153 assign rdata_a_o = (raddr_a_i == '0) ? WordZeroVal : mem[raddr_a_i];
154 assign rdata_b_o = (raddr_b_i == '0) ? WordZeroVal : mem[raddr_b_i];
155
156 assign oh_raddr_a_err = 1'b0;
157 assign oh_raddr_b_err = 1'b0;
158 end
159
160 // we select
161 assign we = (waddr_a_i == '0) ? 1'b0 : we_a_i;
162
163 // SEC_CM: DATA_REG_SW.GLITCH_DETECT
164 // This checks for spurious WE strobes on the regfile.
165 if (WrenCheck) begin : gen_wren_check
166 // Since the FPGA uses a memory macro, there is only one write-enable strobe to check.
167 assign oh_we_err = we && !we_a_i;
168 end else begin : gen_no_wren_check
169 assign oh_we_err = 1'b0;
170 end
171
172 // Note that the SystemVerilog LRM requires variables on the LHS of assignments within
173 // "always_ff" to not be written to by any other process. However, to enable the initialization
174 // of the inferred RAM32M primitives with non-zero values, below "initial" procedure is needed.
175 // Therefore, we use "always" instead of the generally preferred "always_ff" for the synchronous
176 // write procedure.
177 always @(posedge clk_i) begin : sync_write
178 if (we == 1'b1) begin
179 mem[waddr_a_i] <= wdata_a_i;
180 end
181 end : sync_write
182
183 // Make sure we initialize the BRAM with the correct register reset value.
184 initial begin
185 for (int k = 0; k < NUM_WORDS; k++) begin
186 mem[k] = WordZeroVal;
187 end
188 end
189
190 // Reset not used in this register file version
191 logic unused_rst_ni;
192 assign unused_rst_ni = rst_ni;
193
194 // Dummy instruction changes not relevant for FPGA implementation
195 logic unused_dummy_instr;
196 assign unused_dummy_instr = dummy_instr_id_i ^ dummy_instr_wb_i;
197 // Test enable signal not used in FPGA implementation
198 logic unused_test_en;
199 assign unused_test_en = test_en_i;
200
201endmodule
