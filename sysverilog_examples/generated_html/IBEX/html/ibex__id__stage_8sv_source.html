512 /////////////////////////////////
513 // CSR-related pipeline flushes //
514 /////////////////////////////////
515 always_comb begin : csr_pipeline_flushes
516 csr_pipe_flush = 1'b0;
517
518 // A pipeline flush is needed to let the controller react after modifying certain CSRs:
519 // - When enabling interrupts, pending IRQs become visible to the controller only during
520 // the next cycle. If during that cycle the core disables interrupts again, it does not
521 // see any pending IRQs and consequently does not start to handle interrupts.
522 // - When modifying any PMP CSR, PMP check of the next instruction might get invalidated.
523 // Hence, a pipeline flush is needed to instantiate another PMP check with the updated CSRs.
524 // - When modifying debug CSRs.
525 if (csr_op_en_o == 1'b1 && (csr_op_o == CSR_OP_WRITE || csr_op_o == CSR_OP_SET)) begin
526 if (csr_num_e'(instr_rdata_i[31:20]) == CSR_MSTATUS ||
527 csr_num_e'(instr_rdata_i[31:20]) == CSR_MIE ||
528 csr_num_e'(instr_rdata_i[31:20]) == CSR_MSECCFG ||
529 // To catch all PMPCFG/PMPADDR registers, get the shared top most 7 bits.
530 instr_rdata_i[31:25] == 7'h1D) begin
531 csr_pipe_flush = 1'b1;
532 end
533 end else if (csr_op_en_o == 1'b1 && csr_op_o != CSR_OP_READ) begin
534 if (csr_num_e'(instr_rdata_i[31:20]) == CSR_DCSR ||
535 csr_num_e'(instr_rdata_i[31:20]) == CSR_DPC ||
536 csr_num_e'(instr_rdata_i[31:20]) == CSR_DSCRATCH0 ||
537 csr_num_e'(instr_rdata_i[31:20]) == CSR_DSCRATCH1) begin
538 csr_pipe_flush = 1'b1;
539 end
540 end
541 end

778 typedef enum logic { FIRST_CYCLE, MULTI_CYCLE } id_fsm_e;

789 // ID/EX stage can be in two states, FIRST_CYCLE and MULTI_CYCLE. An instruction enters
790 // MULTI_CYCLE if it requires multiple cycles to complete regardless of stalls and other
791 // considerations. An instruction may be held in FIRST_CYCLE if it's unable to begin executing
792 // (this is controlled by instr_executing).
793
794 always_comb begin
795 id_fsm_d = id_fsm_q;
796 rf_we_raw = rf_we_dec;
797 stall_multdiv = 1'b0;
798 stall_jump = 1'b0;
799 stall_branch = 1'b0;
800 stall_alu = 1'b0;
801 branch_set_raw_d = 1'b0;
802 branch_not_set = 1'b0;
803 jump_set_raw = 1'b0;
804 perf_branch_o = 1'b0;
805
806 if (instr_executing_spec) begin
807 unique case (id_fsm_q)
808 FIRST_CYCLE: begin
809 unique case (1'b1)
810 lsu_req_dec: begin
811 if (!WritebackStage) begin
812 // LSU operation
813 id_fsm_d = MULTI_CYCLE;
814 end else begin
815 if(~lsu_req_done_i) begin
816 id_fsm_d = MULTI_CYCLE;
817 end
818 end
819 end
820 multdiv_en_dec: begin
821 // MUL or DIV operation
822 if (~ex_valid_i) begin
823 // When single-cycle multiply is configured mul can finish in the first cycle so
824 // only enter MULTI_CYCLE state if a result isn't immediately available
825 id_fsm_d = MULTI_CYCLE;
826 rf_we_raw = 1'b0;
827 stall_multdiv = 1'b1;
828 end
829 end
830 branch_in_dec: begin
831 // cond branch operation
832 // All branches take two cycles in fixed time execution mode, regardless of branch
833 // condition.
834 // SEC_CM: CORE.DATA_REG_SW.SCA
835 id_fsm_d = (data_ind_timing_i || (!BranchTargetALU && branch_decision_i)) ?
836 MULTI_CYCLE : FIRST_CYCLE;
837 stall_branch = (~BranchTargetALU & branch_decision_i) | data_ind_timing_i;
838 branch_set_raw_d = (branch_decision_i | data_ind_timing_i);
839
840 if (BranchPredictor) begin
841 branch_not_set = ~branch_decision_i;
842 end
843
844 perf_branch_o = 1'b1;
845 end
846 jump_in_dec: begin
847 // uncond branch operation
848 // BTALU means jumps only need one cycle
849 id_fsm_d = BranchTargetALU ? FIRST_CYCLE : MULTI_CYCLE;
850 stall_jump = ~BranchTargetALU;
851 jump_set_raw = jump_set_dec;
852 end
853 alu_multicycle_dec: begin
854 stall_alu = 1'b1;
855 id_fsm_d = MULTI_CYCLE;
856 rf_we_raw = 1'b0;
857 end
858 default: begin
859 id_fsm_d = FIRST_CYCLE;
860 end
861 endcase
862 end
863
864 MULTI_CYCLE: begin
865 if(multdiv_en_dec) begin
866 rf_we_raw = rf_we_dec & ex_valid_i;
867 end
868
869 if (multicycle_done & ready_wb_i) begin
870 id_fsm_d = FIRST_CYCLE;
871 end else begin
872 stall_multdiv = multdiv_en_dec;
873 stall_branch = branch_in_dec;
874 stall_jump = jump_in_dec;
875 end
876 end
877
878 default: begin
879 id_fsm_d = FIRST_CYCLE;
880 end
881 endcase
882 end
883 end
