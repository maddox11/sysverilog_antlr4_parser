5module ibex_counter #(
6 parameter int CounterWidth = 32,
7 // When set `counter_val_upd_o` provides an incremented version of the counter value, otherwise
8 // the output is hard-wired to 0. This is required to allow Xilinx DSP inference to work
9 // correctly. When `ProvideValUpd` is set no DSPs are inferred.
10 parameter bit ProvideValUpd = 0
11) (
12 input logic clk_i,
13 input logic rst_ni,
14
15 input logic counter_inc_i,
16 input logic counterh_we_i,
17 input logic counter_we_i,
18 input logic [31:0] counter_val_i,
19 output logic [63:0] counter_val_o,
20 output logic [63:0] counter_val_upd_o
21);
22
23 logic [63:0] counter;
24 logic [CounterWidth-1:0] counter_upd;
25 logic [63:0] counter_load;
26 logic we;
27 logic [CounterWidth-1:0] counter_d;
28
29 // Increment
30 assign counter_upd = counter[CounterWidth-1:0] + {{CounterWidth - 1{1'b0}}, 1'b1};
31
32 // Update
33 always_comb begin
34 // Write
35 we = counter_we_i | counterh_we_i;
36 counter_load[63:32] = counter[63:32];
37 counter_load[31:0] = counter_val_i;
38 if (counterh_we_i) begin
39 counter_load[63:32] = counter_val_i;
40 counter_load[31:0] = counter[31:0];
41 end
42
43 // Next value logic
44 if (we) begin
45 counter_d = counter_load[CounterWidth-1:0];
46 end else if (counter_inc_i) begin
47 counter_d = counter_upd[CounterWidth-1:0];
48 end else begin
49 counter_d = counter[CounterWidth-1:0];
50 end
51 end
52
53 `ifdef FPGA_XILINX
54 // On Xilinx FPGAs, 48-bit DSPs are available that can be used for the
55 // counter. Hence, use Xilinx specific flop implementation. The datatype for
56 // UseDsp is on purpose int as with string Xilinx throws an error for the
57 // use_dsp pragma.
58 localparam int UseDsp = CounterWidth < 49 ? "yes" : "no";
59 (* use_dsp = UseDsp *) logic [CounterWidth-1:0] counter_q;
60 `else
61 localparam int UseDsp = "no";
62 logic [CounterWidth-1:0] counter_q;
63 `endif
64
65 if (UseDsp == "yes") begin : g_cnt_dsp
66 // Use sync. reset for DSP.
67 always_ff @(posedge clk_i) begin
68 if (!rst_ni) begin
69 counter_q <= '0;
70 end else begin
71 counter_q <= counter_d;
72 end
73 end
74 end else begin : g_cnt_no_dsp
75 // Use async. reset for flop.
76 always_ff @(posedge clk_i or negedge rst_ni) begin
77 if (!rst_ni) begin
78 counter_q <= '0;
79 end else begin
80 counter_q <= counter_d;
81 end
82 end
83 end
84
85
86 if (CounterWidth < 64) begin : g_counter_narrow
87 logic [63:CounterWidth] unused_counter_load;
88
89 assign counter[CounterWidth-1:0] = counter_q;
90 assign counter[63:CounterWidth] = '0;
91
92 if (ProvideValUpd) begin : g_counter_val_upd_o
93 assign counter_val_upd_o[CounterWidth-1:0] = counter_upd;
94 end else begin : g_no_counter_val_upd_o
95 assign counter_val_upd_o[CounterWidth-1:0] = '0;
96 end
97 assign counter_val_upd_o[63:CounterWidth] = '0;
98 assign unused_counter_load = counter_load[63:CounterWidth];
99 end else begin : g_counter_full
100 assign counter = counter_q;
101
102 if (ProvideValUpd) begin : g_counter_val_upd_o
103 assign counter_val_upd_o = counter_upd;
104 end else begin : g_no_counter_val_upd_o
105 assign counter_val_upd_o = '0;
106 end
107 end
108
109 assign counter_val_o = counter;
110
111endmodule
