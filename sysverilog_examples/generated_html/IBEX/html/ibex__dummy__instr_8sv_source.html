5/**
6 * Dummy instruction module
7 *
8 * Provides pseudo-randomly inserted fake instructions for secure code obfuscation
9 */
10
11// SEC_CM: CTRL_FLOW.UNPREDICTABLE
12module ibex_dummy_instr import ibex_pkg::*; #(
13 parameter lfsr_seed_t RndCnstLfsrSeed = RndCnstLfsrSeedDefault,
14 parameter lfsr_perm_t RndCnstLfsrPerm = RndCnstLfsrPermDefault
15) (
16 // Clock and reset
17 input logic clk_i,
18 input logic rst_ni,
19
20 // Interface to CSRs
21 input logic dummy_instr_en_i,
22 input logic [2:0] dummy_instr_mask_i,
23 input logic dummy_instr_seed_en_i,
24 input logic [31:0] dummy_instr_seed_i,
25
26 // Interface to IF stage
27 input logic fetch_valid_i,
28 input logic id_in_ready_i,
29 output logic insert_dummy_instr_o,
30 output logic [31:0] dummy_instr_data_o
31);
32
33 localparam int unsigned TIMEOUT_CNT_W = 5;
34 localparam int unsigned OP_W = 5;
35
36 typedef enum logic [1:0] {
37 DUMMY_ADD = 2'b00,
38 DUMMY_MUL = 2'b01,
39 DUMMY_DIV = 2'b10,
40 DUMMY_AND = 2'b11
41 } dummy_instr_e;
42
43 typedef struct packed {
44 dummy_instr_e instr_type;
45 logic [OP_W-1:0] op_b;
46 logic [OP_W-1:0] op_a;
47 logic [TIMEOUT_CNT_W-1:0] cnt;
48 } lfsr_data_t;
49 localparam int unsigned LFSR_OUT_W = $bits(lfsr_data_t);
50
51 lfsr_data_t lfsr_data;
52 logic [TIMEOUT_CNT_W-1:0] dummy_cnt_incr, dummy_cnt_threshold;
53 logic [TIMEOUT_CNT_W-1:0] dummy_cnt_d, dummy_cnt_q;
54 logic dummy_cnt_en;
55 logic lfsr_en;
56 logic [LFSR_OUT_W-1:0] lfsr_state;
57 logic insert_dummy_instr;
58 logic [6:0] dummy_set;
59 logic [2:0] dummy_opcode;
60 logic [31:0] dummy_instr;
61 logic [31:0] dummy_instr_seed_q, dummy_instr_seed_d;
62
63 // Shift the LFSR every time we insert an instruction
64 assign lfsr_en = insert_dummy_instr & id_in_ready_i;
65
66 assign dummy_instr_seed_d = dummy_instr_seed_q ^ dummy_instr_seed_i;
67
68 always_ff @(posedge clk_i or negedge rst_ni) begin
69 if (!rst_ni) begin
70 dummy_instr_seed_q <= '0;
71 end else if (dummy_instr_seed_en_i) begin
72 dummy_instr_seed_q <= dummy_instr_seed_d;
73 end
74 end
75
76 prim_lfsr #(
77 .LfsrDw ( LfsrWidth ),
78 .StateOutDw ( LFSR_OUT_W ),
79 .DefaultSeed ( RndCnstLfsrSeed ),
80 .StatePermEn ( 1'b1 ),
81 .StatePerm ( RndCnstLfsrPerm )
82 ) lfsr_i (
83 .clk_i ( clk_i ),
84 .rst_ni ( rst_ni ),
85 .seed_en_i ( dummy_instr_seed_en_i ),
86 .seed_i ( dummy_instr_seed_d ),
87 .lfsr_en_i ( lfsr_en ),
88 .entropy_i ( '0 ),
89 .state_o ( lfsr_state )
90 );
91
92 // Extract fields from LFSR
93 assign lfsr_data = lfsr_data_t'(lfsr_state);
94
95 // Set count threshold for inserting a new instruction. This is the pseudo-random value from the
96 // LFSR with a mask applied (based on CSR config data) to shorten the period if required.
97 assign dummy_cnt_threshold = lfsr_data.cnt & {dummy_instr_mask_i,{TIMEOUT_CNT_W-3{1'b1}}};
98 assign dummy_cnt_incr = dummy_cnt_q + {{TIMEOUT_CNT_W-1{1'b0}},1'b1};
99 // Clear the counter everytime a new instruction is inserted
100 assign dummy_cnt_d = insert_dummy_instr ? '0 : dummy_cnt_incr;
101 // Increment the counter for each executed instruction while dummy instuctions are
102 // enabled.
103 assign dummy_cnt_en = dummy_instr_en_i & id_in_ready_i &
104 (fetch_valid_i | insert_dummy_instr);
105
106 always_ff @(posedge clk_i or negedge rst_ni) begin
107 if (!rst_ni) begin
108 dummy_cnt_q <= '0;
109 end else if (dummy_cnt_en) begin
110 dummy_cnt_q <= dummy_cnt_d;
111 end
112 end
113
114 // Insert a dummy instruction each time the counter hits the threshold
115 assign insert_dummy_instr = dummy_instr_en_i & (dummy_cnt_q == dummy_cnt_threshold);
116
117 // Encode instruction
118 always_comb begin
119 unique case (lfsr_data.instr_type)
120 DUMMY_ADD: begin
121 dummy_set = 7'b0000000;
122 dummy_opcode = 3'b000;
123 end
124 DUMMY_MUL: begin
125 dummy_set = 7'b0000001;
126 dummy_opcode = 3'b000;
127 end
128 DUMMY_DIV: begin
129 dummy_set = 7'b0000001;
130 dummy_opcode = 3'b100;
131 end
132 DUMMY_AND: begin
133 dummy_set = 7'b0000000;
134 dummy_opcode = 3'b111;
135 end
136 default: begin
137 dummy_set = 7'b0000000;
138 dummy_opcode = 3'b000;
139 end
140 endcase
141 end
142
143 // SET RS2 RS1 OP RD
144 assign dummy_instr = {dummy_set, lfsr_data.op_b, lfsr_data.op_a, dummy_opcode, 5'h00, 7'h33};
145
146 // Assign outputs
147 assign insert_dummy_instr_o = insert_dummy_instr;
148 assign dummy_instr_data_o = dummy_instr;
149
150endmodule
