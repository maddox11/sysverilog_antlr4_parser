5/**
6 * Branch Predictor
7 *
8 * This implements static branch prediction. It takes an instruction and its PC and determines if
9 * it's a branch or a jump and calculates its target. For jumps it will always predict taken. For
10 * branches it will predict taken if the PC offset is negative.
11 *
12 * This handles both compressed and uncompressed instructions. Compressed instructions must be in
13 * the lower 16-bits of instr.
14 *
15 * The predictor is entirely combinational but takes clk/rst_n signals for use by assertions.
16 */
17
18`include "prim_assert.sv"
19
20module ibex_branch_predict (
21 input logic clk_i,
22 input logic rst_ni,
23
24 // Instruction from fetch stage
25 input logic [31:0] fetch_rdata_i,
26 input logic [31:0] fetch_pc_i,
27 input logic fetch_valid_i,
28
29 // Prediction for supplied instruction
30 output logic predict_branch_taken_o,
31 output logic [31:0] predict_branch_pc_o
32);
33 import ibex_pkg::*;
34
35 logic [31:0] imm_j_type;
36 logic [31:0] imm_b_type;
37 logic [31:0] imm_cj_type;
38 logic [31:0] imm_cb_type;
39
40 logic [31:0] branch_imm;
41
42 logic [31:0] instr;
43
44 logic instr_j;
45 logic instr_b;
46 logic instr_cj;
47 logic instr_cb;
48
49 logic instr_b_taken;
50
51 // Provide short internal name for fetch_rdata_i due to reduce line wrapping
52 assign instr = fetch_rdata_i;
53
54 // Extract and sign-extend to 32-bit the various immediates that may be used to calculate the
55 // target
56
57 // Uncompressed immediates
58 assign imm_j_type = { {12{instr[31]}}, instr[19:12], instr[20], instr[30:21], 1'b0 };
59 assign imm_b_type = { {19{instr[31]}}, instr[31], instr[7], instr[30:25], instr[11:8], 1'b0 };
60
61 // Compressed immediates
62 assign imm_cj_type = { {20{instr[12]}}, instr[12], instr[8], instr[10:9], instr[6], instr[7],
63 instr[2], instr[11], instr[5:3], 1'b0 };
64
65 assign imm_cb_type = { {23{instr[12]}}, instr[12], instr[6:5], instr[2], instr[11:10],
66 instr[4:3], 1'b0};
67
68 // Determine if the instruction is a branch or a jump
69
70 // Uncompressed branch/jump
71 assign instr_b = opcode_e'(instr[6:0]) == OPCODE_BRANCH;
72 assign instr_j = opcode_e'(instr[6:0]) == OPCODE_JAL;
73
74 // Compressed branch/jump
75 assign instr_cb = (instr[1:0] == 2'b01) & ((instr[15:13] == 3'b110) | (instr[15:13] == 3'b111));
76 assign instr_cj = (instr[1:0] == 2'b01) & ((instr[15:13] == 3'b101) | (instr[15:13] == 3'b001));
77
78 // Select out the branch offset for target calculation based upon the instruction type
79 always_comb begin
80 branch_imm = imm_b_type;
81
82 unique case (1'b1)
83 instr_j : branch_imm = imm_j_type;
84 instr_b : branch_imm = imm_b_type;
85 instr_cj : branch_imm = imm_cj_type;
86 instr_cb : branch_imm = imm_cb_type;
87 default : ;
88 endcase
89 end
90
91 `ASSERT_IF(BranchInsTypeOneHot, $onehot0({instr_j, instr_b, instr_cj, instr_cb}), fetch_valid_i)
92
93 // Determine branch prediction, taken if offset is negative
94 assign instr_b_taken = (instr_b & imm_b_type[31]) | (instr_cb & imm_cb_type[31]);
95
96 // Always predict jumps taken otherwise take prediction from `instr_b_taken`
97 assign predict_branch_taken_o = fetch_valid_i & (instr_j | instr_cj | instr_b_taken);
98 // Calculate target
99 assign predict_branch_pc_o = fetch_pc_i + branch_imm;
100endmodule
