6/**
7 * RISC-V register file
8 *
9 * Register file with 31 or 15x 32 bit wide registers. Register 0 is fixed to 0.
10 * This register file is based on flip flops. Use this register file when
11 * targeting FPGA synthesis or Verilator simulation.
12 */
13module ibex_register_file_ff #(
14 parameter bit RV32E = 0,
15 parameter int unsigned DataWidth = 32,
16 parameter bit DummyInstructions = 0,
17 parameter bit WrenCheck = 0,
18 parameter bit RdataMuxCheck = 0,
19 parameter logic [DataWidth-1:0] WordZeroVal = '0
20) (
21 // Clock and Reset
22 input logic clk_i,
23 input logic rst_ni,
24
25 input logic test_en_i,
26 input logic dummy_instr_id_i,
27 input logic dummy_instr_wb_i,
28
29 //Read port R1
30 input logic [4:0] raddr_a_i,
31 output logic [DataWidth-1:0] rdata_a_o,
32
33 //Read port R2
34 input logic [4:0] raddr_b_i,
35 output logic [DataWidth-1:0] rdata_b_o,
36
37
38 // Write port W1
39 input logic [4:0] waddr_a_i,
40 input logic [DataWidth-1:0] wdata_a_i,
41 input logic we_a_i,
42
43 // This indicates whether spurious WE or non-one-hot encoded raddr are detected.
44 output logic err_o
45);
46
47 localparam int unsigned ADDR_WIDTH = RV32E ? 4 : 5;
48 localparam int unsigned NUM_WORDS = 2**ADDR_WIDTH;
49
50 logic [DataWidth-1:0] rf_reg [NUM_WORDS];
51 logic [NUM_WORDS-1:0] we_a_dec;
52
53 logic oh_raddr_a_err, oh_raddr_b_err, oh_we_err;
54
55 always_comb begin : we_a_decoder
56 for (int unsigned i = 0; i < NUM_WORDS; i++) begin
57 we_a_dec[i] = (waddr_a_i == 5'(i)) ? we_a_i : 1'b0;
58 end
59 end
60
61 // SEC_CM: DATA_REG_SW.GLITCH_DETECT
62 // This checks for spurious WE strobes on the regfile.
63 if (WrenCheck) begin : gen_wren_check
64 // Buffer the decoded write enable bits so that the checker
65 // is not optimized into the address decoding logic.
66 logic [NUM_WORDS-1:0] we_a_dec_buf;
67 prim_buf #(
68 .Width(NUM_WORDS)
69 ) u_prim_buf (
70 .in_i(we_a_dec),
71 .out_o(we_a_dec_buf)
72 );
73
74 prim_onehot_check #(
75 .AddrWidth(ADDR_WIDTH),
76 .AddrCheck(1),
77 .EnableCheck(1)
78 ) u_prim_onehot_check (
79 .clk_i,
80 .rst_ni,
81 .oh_i(we_a_dec_buf),
82 .addr_i(waddr_a_i),
83 .en_i(we_a_i),
84 .err_o(oh_we_err)
85 );
86 end else begin : gen_no_wren_check
87 logic unused_strobe;
88 assign unused_strobe = we_a_dec[0]; // this is never read from in this case
89 assign oh_we_err = 1'b0;
90 end
91
92 // No flops for R0 as it's hard-wired to 0
93 for (genvar i = 1; i < NUM_WORDS; i++) begin : g_rf_flops
94 logic [DataWidth-1:0] rf_reg_q;
95
96 always_ff @(posedge clk_i or negedge rst_ni) begin
97 if (!rst_ni) begin
98 rf_reg_q <= WordZeroVal;
99 end else if (we_a_dec[i]) begin
100 rf_reg_q <= wdata_a_i;
101 end
102 end
103
104 assign rf_reg[i] = rf_reg_q;
105 end
106
107 // With dummy instructions enabled, R0 behaves as a real register but will always return 0 for
108 // real instructions.
109 if (DummyInstructions) begin : g_dummy_r0
110 // SEC_CM: CTRL_FLOW.UNPREDICTABLE
111 logic we_r0_dummy;
112 logic [DataWidth-1:0] rf_r0_q;
113
114 // Write enable for dummy R0 register (waddr_a_i will always be 0 for dummy instructions)
115 assign we_r0_dummy = we_a_i & dummy_instr_wb_i;
116
117 always_ff @(posedge clk_i or negedge rst_ni) begin
118 if (!rst_ni) begin
119 rf_r0_q <= WordZeroVal;
120 end else if (we_r0_dummy) begin
121 rf_r0_q <= wdata_a_i;
122 end
123 end
124
125 // Output the dummy data for dummy instructions, otherwise R0 reads as zero
126 assign rf_reg[0] = dummy_instr_id_i ? rf_r0_q : WordZeroVal;
127
128 end else begin : g_normal_r0
129 logic unused_dummy_instr;
130 assign unused_dummy_instr = dummy_instr_id_i ^ dummy_instr_wb_i;
131
132 // R0 is nil
133 assign rf_reg[0] = WordZeroVal;
134 end
135
136 if (RdataMuxCheck) begin : gen_rdata_mux_check
137 // Encode raddr_a/b into one-hot encoded signals.
138 logic [NUM_WORDS-1:0] raddr_onehot_a, raddr_onehot_b;
139 logic [NUM_WORDS-1:0] raddr_onehot_a_buf, raddr_onehot_b_buf;
140 prim_onehot_enc #(
141 .OneHotWidth(NUM_WORDS)
142 ) u_prim_onehot_enc_raddr_a (
143 .in_i (raddr_a_i),
144 .en_i (1'b1),
145 .out_o (raddr_onehot_a)
146 );
147
148 prim_onehot_enc #(
149 .OneHotWidth(NUM_WORDS)
150 ) u_prim_onehot_enc_raddr_b (
151 .in_i (raddr_b_i),
152 .en_i (1'b1),
153 .out_o (raddr_onehot_b)
154 );
155
156 // Buffer the one-hot encoded signals so that the checkers
157 // are not optimized.
158 prim_buf #(
159 .Width(NUM_WORDS)
160 ) u_prim_buf_raddr_a (
161 .in_i (raddr_onehot_a),
162 .out_o(raddr_onehot_a_buf)
163 );
164
165 prim_buf #(
166 .Width(NUM_WORDS)
167 ) u_prim_buf_raddr_b (
168 .in_i (raddr_onehot_b),
169 .out_o(raddr_onehot_b_buf)
170 );
171
172 // SEC_CM: DATA_REG_SW.GLITCH_DETECT
173 // Check the one-hot encoded signals for glitches.
174 prim_onehot_check #(
175 .AddrWidth(ADDR_WIDTH),
176 .OneHotWidth(NUM_WORDS),
177 .AddrCheck(1),
178 // When AddrCheck=1 also EnableCheck needs to be 1.
179 .EnableCheck(1)
180 ) u_prim_onehot_check_raddr_a (
181 .clk_i,
182 .rst_ni,
183 .oh_i (raddr_onehot_a_buf),
184 .addr_i (raddr_a_i),
185 // Set enable=1 as address is always valid.
186 .en_i (1'b1),
187 .err_o (oh_raddr_a_err)
188 );
189
190 prim_onehot_check #(
191 .AddrWidth(ADDR_WIDTH),
192 .OneHotWidth(NUM_WORDS),
193 .AddrCheck(1),
194 // When AddrCheck=1 also EnableCheck needs to be 1.
195 .EnableCheck(1)
196 ) u_prim_onehot_check_raddr_b (
197 .clk_i,
198 .rst_ni,
199 .oh_i (raddr_onehot_b_buf),
200 .addr_i (raddr_b_i),
201 // Set enable=1 as address is always valid.
202 .en_i (1'b1),
203 .err_o (oh_raddr_b_err)
204 );
205
206 // MUX register to rdata_a/b_o according to raddr_a/b_onehot.
207 prim_onehot_mux #(
208 .Width(DataWidth),
209 .Inputs(NUM_WORDS)
210 ) u_rdata_a_mux (
211 .clk_i,
212 .rst_ni,
213 .in_i (rf_reg),
214 .sel_i (raddr_onehot_a),
215 .out_o (rdata_a_o)
216 );
217
218 prim_onehot_mux #(
219 .Width(DataWidth),
220 .Inputs(NUM_WORDS)
221 ) u_rdata_b_mux (
222 .clk_i,
223 .rst_ni,
224 .in_i (rf_reg),
225 .sel_i (raddr_onehot_b),
226 .out_o (rdata_b_o)
227 );
228 end else begin : gen_no_rdata_mux_check
229 assign rdata_a_o = rf_reg[raddr_a_i];
230 assign rdata_b_o = rf_reg[raddr_b_i];
231 assign oh_raddr_a_err = 1'b0;
232 assign oh_raddr_b_err = 1'b0;
233 end
234
235 assign err_o = oh_raddr_a_err || oh_raddr_b_err || oh_we_err;
236
237 // Signal not used in FF register file
238 logic unused_test_en;
239 assign unused_test_en = test_en_i;
240
241endmodule
