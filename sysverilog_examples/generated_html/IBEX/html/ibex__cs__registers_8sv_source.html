129 // Is a PMP config a locked one that allows M-mode execution when MSECCFG.MML is set (either
130 // M mode alone or shared M/U mode execution)?
131 function automatic logic is_mml_m_exec_cfg(ibex_pkg::pmp_cfg_t pmp_cfg);
132 logic unused_cfg = ^{pmp_cfg.mode};
133 logic value = 1'b0;
134
135 if (pmp_cfg.lock) begin
136 unique case ({pmp_cfg.read, pmp_cfg.write, pmp_cfg.exec})
137 3'b001, 3'b010, 3'b011, 3'b101: value = 1'b1;
138 default: value = 1'b0;
139 endcase
140 end
141
142 return value;
143 endfunction
