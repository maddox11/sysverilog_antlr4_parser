598 // When branch prediction is enabled a skid buffer between the IF and ID/EX stage is introduced.
599 // If an instruction in IF is predicted to be a taken branch and ID/EX is not ready the
600 // instruction in IF is moved to the skid buffer which becomes the output of the IF stage until
601 // the ID/EX stage accepts the instruction. The skid buffer is required as otherwise the ID/EX
602 // ready signal is coupled to the instr_req_o output which produces a feedthrough path from
603 // data_gnt_i -> instr_req_o (which needs to be avoided as for some interconnects this will
604 // result in a combinational loop).
605

627 instr_skid_data_q <= fetch_rdata;
628 instr_skid_addr_q <= fetch_addr;

635 instr_skid_data_q <= fetch_rdata;

652 // If there is an instruction in the skid buffer there must be no branch prediction.
653 // Instructions are only placed in the skid after they have been predicted to be a taken branch
654 // so with the skid valid any prediction has already occurred.
655 // Do not branch predict on instruction errors.
656 assign predict_branch_taken = predict_branch_taken_raw & ~instr_skid_valid_q & ~fetch_err;

658 assign if_instr_valid = fetch_valid | (instr_skid_valid_q & ~nt_branch_mispredict_i);

660 assign if_instr_addr = instr_skid_valid_q ? instr_skid_addr_q : fetch_addr;

663 // skid buffer.
664 assign if_instr_bus_err = ~instr_skid_valid_q & fetch_err;
665 assign instr_bp_taken_d = instr_skid_valid_q ? instr_skid_bp_taken_q : predict_branch_taken;

715 /**
716 * Checks for branch prediction interface to fetch_fifo/icache
717 *
718 * The interface has two signals:
719 * - predicted_branch_i: When set with a branch (branch_i) indicates the branch is a predicted
720 * one, it should be ignored when a branch_i isn't set.
721 * - branch_mispredict_i: Indicates the previously predicted branch was mis-predicted and
722 * execution should resume with the not-taken side of the branch (i.e. continue with the PC
723 * that followed the predicted branch). This must be raised before the instruction that is
724 * made available following a predicted branch is accepted (Following a cycle with branch_i
725 * & predicted_branch_i, branch_mispredict_i can only be asserted before or on the same cycle
726 * as seeing fetch_valid & fetch_ready). When branch_mispredict_i is asserted, fetch_valid may
727 * be asserted in response. If fetch_valid is asserted on the same cycle as
728 * branch_mispredict_i this indicates the fetch_fifo/icache has the not-taken side of the
729 * branch immediately ready for use
730 */
