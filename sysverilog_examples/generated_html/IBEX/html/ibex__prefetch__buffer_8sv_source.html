6/**
7 * Prefetcher Buffer for 32 bit memory interface
8 *
9 * Prefetch Buffer that caches instructions. This cuts overly long critical
10 * paths to the instruction cache.
11 */
12module ibex_prefetch_buffer #(
13 parameter bit ResetAll = 1'b0
14) (
15 input logic clk_i,
16 input logic rst_ni,
17
18 input logic req_i,
19
20 input logic branch_i,
21 input logic [31:0] addr_i,
22
23 input logic ready_i,
24 output logic valid_o,
25 output logic [31:0] rdata_o,
26 output logic [31:0] addr_o,
27 output logic err_o,
28 output logic err_plus2_o,
29
30 // goes to instruction memory / instruction cache
31 output logic instr_req_o,
32 input logic instr_gnt_i,
33 output logic [31:0] instr_addr_o,
34 input logic [31:0] instr_rdata_i,
35 input logic instr_err_i,
36 input logic instr_rvalid_i,
37
38 // Prefetch Buffer Status
39 output logic busy_o
40);
41
42 localparam int unsigned NUM_REQS = 2;
43
44 logic valid_new_req, valid_req;
45 logic valid_req_d, valid_req_q;
46 logic discard_req_d, discard_req_q;
47 logic [NUM_REQS-1:0] rdata_outstanding_n, rdata_outstanding_s, rdata_outstanding_q;
48 logic [NUM_REQS-1:0] branch_discard_n, branch_discard_s, branch_discard_q;
49 logic [NUM_REQS-1:0] rdata_outstanding_rev;
50
51 logic [31:0] stored_addr_d, stored_addr_q;
52 logic stored_addr_en;
53 logic [31:0] fetch_addr_d, fetch_addr_q;
54 logic fetch_addr_en;
55 logic [31:0] instr_addr, instr_addr_w_aligned;
56
57 logic fifo_valid;
58 logic [31:0] fifo_addr;
59 logic fifo_ready;
60 logic fifo_clear;
61 logic [NUM_REQS-1:0] fifo_busy;
62
63 ////////////////////////////
64 // Prefetch buffer status //
65 ////////////////////////////
66
67 assign busy_o = (|rdata_outstanding_q) | instr_req_o;
68
69 //////////////////////////////////////////////
70 // Fetch fifo - consumes addresses and data //
71 //////////////////////////////////////////////
72
73 // A branch will invalidate any previously fetched instructions.
74 // Note that the FENCE.I instruction relies on this flushing behaviour on branch. If it is
75 // altered the FENCE.I implementation may require changes.
76 assign fifo_clear = branch_i;
77
78 // Reversed version of rdata_outstanding_q which can be overlaid with fifo fill state
79 for (genvar i = 0; i < NUM_REQS; i++) begin : gen_rd_rev
80 assign rdata_outstanding_rev[i] = rdata_outstanding_q[NUM_REQS-1-i];
81 end
82
83 // The fifo is ready to accept a new request if it is not full - including space reserved for
84 // requests already outstanding.
85 // Overlay the fifo fill state with the outstanding requests to see if there is space.
86 assign fifo_ready = ~&(fifo_busy | rdata_outstanding_rev);
87
88 ibex_fetch_fifo #(
89 .NUM_REQS (NUM_REQS),
90 .ResetAll (ResetAll)
91 ) fifo_i (
92 .clk_i ( clk_i ),
93 .rst_ni ( rst_ni ),
94
95 .clear_i ( fifo_clear ),
96 .busy_o ( fifo_busy ),
97
98 .in_valid_i ( fifo_valid ),
99 .in_addr_i ( fifo_addr ),
100 .in_rdata_i ( instr_rdata_i ),
101 .in_err_i ( instr_err_i ),
102
103 .out_valid_o ( valid_o ),
104 .out_ready_i ( ready_i ),
105 .out_rdata_o ( rdata_o ),
106 .out_addr_o ( addr_o ),
107 .out_err_o ( err_o ),
108 .out_err_plus2_o ( err_plus2_o )
109 );
110
111 //////////////
112 // Requests //
113 //////////////
114
115 // Make a new request any time there is space in the FIFO, and space in the request queue
116 assign valid_new_req = req_i & (fifo_ready | branch_i) &
117 ~rdata_outstanding_q[NUM_REQS-1];
118
119 assign valid_req = valid_req_q | valid_new_req;
120
121 // Hold the request stable for requests that didn't get granted
122 assign valid_req_d = valid_req & ~instr_gnt_i;
123
124 // Record whether an outstanding bus request is cancelled by a branch
125 assign discard_req_d = valid_req_q & (branch_i | discard_req_q);
126
127 ////////////////
128 // Fetch addr //
129 ////////////////
130
131 // Two addresses are tracked in the prefetch buffer:
132 // 1. stored_addr_q - This is the address issued on the bus. It stays stable until
133 // the request is granted.
134 // 2. fetch_addr_q - This is our next address to fetch from. It is updated on branches to
135 // capture the new address, and then for each new request issued.
136 // A third address is tracked in the fetch FIFO itself:
137 // 3. instr_addr_q - This is the address at the head of the FIFO, efectively our oldest fetched
138 // address. This address is updated on branches, and does its own increment
139 // each time the FIFO is popped.
140
141 // 1. stored_addr_q
142
143 // Only update stored_addr_q for new ungranted requests
144 assign stored_addr_en = valid_new_req & ~valid_req_q & ~instr_gnt_i;
145
146 // Store whatever address was issued on the bus
147 assign stored_addr_d = instr_addr;
148
149 // CPU resets with a branch, so no need to reset these addresses
150 if (ResetAll) begin : g_stored_addr_ra
151 always_ff @(posedge clk_i or negedge rst_ni) begin
152 if (!rst_ni) begin
153 stored_addr_q <= '0;
154 end else if (stored_addr_en) begin
155 stored_addr_q <= stored_addr_d;
156 end
157 end
158 end else begin : g_stored_addr_nr
159 always_ff @(posedge clk_i) begin
160 if (stored_addr_en) begin
161 stored_addr_q <= stored_addr_d;
162 end
163 end
164 end
165 // 2. fetch_addr_q
166
167 // Update on a branch or as soon as a request is issued
168 assign fetch_addr_en = branch_i | (valid_new_req & ~valid_req_q);
169
170 assign fetch_addr_d = (branch_i ? addr_i : {fetch_addr_q[31:2], 2'b00}) +
171 // Current address + 4
172 {{29{1'b0}},(valid_new_req & ~valid_req_q),2'b00};
173
174 if (ResetAll) begin : g_fetch_addr_ra
175 always_ff @(posedge clk_i or negedge rst_ni) begin
176 if (!rst_ni) begin
177 fetch_addr_q <= '0;
178 end else if (fetch_addr_en) begin
179 fetch_addr_q <= fetch_addr_d;
180 end
181 end
182 end else begin : g_fetch_addr_nr
183 always_ff @(posedge clk_i) begin
184 if (fetch_addr_en) begin
185 fetch_addr_q <= fetch_addr_d;
186 end
187 end
188 end
189
190 // Address mux
191 assign instr_addr = valid_req_q ? stored_addr_q :
192 branch_i ? addr_i :
193 fetch_addr_q;
194
195 assign instr_addr_w_aligned = {instr_addr[31:2], 2'b00};
196
197 ///////////////////////////////
198 // Request outstanding queue //
199 ///////////////////////////////
200
201 for (genvar i = 0; i < NUM_REQS; i++) begin : g_outstanding_reqs
202 // Request 0 (always the oldest outstanding request)
203 if (i == 0) begin : g_req0
204 // A request becomes outstanding once granted, and is cleared once the rvalid is received.
205 // Outstanding requests shift down the queue towards entry 0.
206 assign rdata_outstanding_n[i] = (valid_req & instr_gnt_i) |
207 rdata_outstanding_q[i];
208 // If a branch is received at any point while a request is outstanding, it must be tracked
209 // to ensure we discard the data once received
210 assign branch_discard_n[i] = (valid_req & instr_gnt_i & discard_req_d) |
211 (branch_i & rdata_outstanding_q[i]) |
212 branch_discard_q[i];
213
214 end else begin : g_reqtop
215 // Entries > 0 consider the FIFO fill state to calculate their next state (by checking
216 // whether the previous entry is valid)
217
218 assign rdata_outstanding_n[i] = (valid_req & instr_gnt_i &
219 rdata_outstanding_q[i-1]) |
220 rdata_outstanding_q[i];
221 assign branch_discard_n[i] = (valid_req & instr_gnt_i & discard_req_d &
222 rdata_outstanding_q[i-1]) |
223 (branch_i & rdata_outstanding_q[i]) |
224 branch_discard_q[i];
225 end
226 end
227
228 // Shift the entries down on each instr_rvalid_i
229 assign rdata_outstanding_s = instr_rvalid_i ? {1'b0,rdata_outstanding_n[NUM_REQS-1:1]} :
230 rdata_outstanding_n;
231 assign branch_discard_s = instr_rvalid_i ? {1'b0,branch_discard_n[NUM_REQS-1:1]} :
232 branch_discard_n;
233
234 // Push a new entry to the FIFO once complete (and not cancelled by a branch)
235 assign fifo_valid = instr_rvalid_i & ~branch_discard_q[0];
236
237 assign fifo_addr = addr_i;
238
239 ///////////////
240 // Registers //
241 ///////////////
242
243 always_ff @(posedge clk_i or negedge rst_ni) begin
244 if (!rst_ni) begin
245 valid_req_q <= 1'b0;
246 discard_req_q <= 1'b0;
247 rdata_outstanding_q <= 'b0;
248 branch_discard_q <= 'b0;
249 end else begin
250 valid_req_q <= valid_req_d;
251 discard_req_q <= discard_req_d;
252 rdata_outstanding_q <= rdata_outstanding_s;
253 branch_discard_q <= branch_discard_s;
254 end
255 end
256
257 /////////////
258 // Outputs //
259 /////////////
260
261 assign instr_req_o = valid_req;
262 assign instr_addr_o = instr_addr_w_aligned;
263
264endmodule
